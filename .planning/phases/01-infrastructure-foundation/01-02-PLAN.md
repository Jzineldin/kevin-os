    RDS_SECRET_ARN=$(aws cloudformation describe-stack-resources --stack-name KosData --region "$REGION" \

        --query "SecretList[?starts_with(Name, 'KosData') && contains(Name, 'Credentials')].ARN | [0]" --output text)
    fi
    SECRET_JSON=$(aws secretsmanager get-secret-value --secret-id "$RDS_SECRET_ARN" --region "$REGION" --query SecretString --output text)
    DB_HOST=$(echo "$SECRET_JSON" | jq -r .host)
    DB_PORT=$(echo "$SECRET_JSON" | jq -r .port)
    DB_USER=$(echo "$SECRET_JSON" | jq -r .username)
