    import { Stack, StackProps, Duration } from 'aws-cdk-lib';
    import { Construct } from 'constructs';
    import { IVpc } from 'aws-cdk-lib/aws-ec2';
    import { EventBus } from 'aws-cdk-lib/aws-events';
    import { ISecret } from 'aws-cdk-lib/aws-secretsmanager';
    import { CfnSchedule } from 'aws-cdk-lib/aws-scheduler';
    import { Role, ServicePrincipal } from 'aws-cdk-lib/aws-iam';
    import { KosLambda } from '../constructs/kos-lambda';
    import * as path from 'path';
    import * as fs from 'fs';

    const NOTION_IDS = JSON.parse(
      fs.readFileSync(path.join(__dirname, '../../../../scripts/.notion-db-ids.json'), 'utf8'),
    ) as { entities: string; projects: string; kevinContext: string; legacyInbox: string; commandCenter: string };

    export interface IntegrationsStackProps extends StackProps {
      vpc: IVpc;
      rdsSecret: ISecret;
      rdsProxyEndpoint: string;
      notionTokenSecret: ISecret;
      captureBus: EventBus;
      scheduleGroupName: string;
    }

    export class IntegrationsStack extends Stack {
      public readonly notionIndexer: KosLambda;
      public readonly notionIndexerBackfill: KosLambda;

      constructor(scope: Construct, id: string, props: IntegrationsStackProps) {
        super(scope, id, props);

        this.notionIndexer = new KosLambda(this, 'NotionIndexer', {
          entry: path.join(__dirname, '../../../../services/notion-indexer/src/handler.ts'),
          timeout: Duration.minutes(2),
          memory: 512,
          environment: {
            NOTION_TOKEN_SECRET_ARN: props.notionTokenSecret.secretArn,
            RDS_SECRET_ARN: props.rdsSecret.secretArn,
            RDS_ENDPOINT: props.rdsProxyEndpoint,
            CAPTURE_BUS_NAME: props.captureBus.eventBusName,
          },
        });
        props.notionTokenSecret.grantRead(this.notionIndexer);
        props.rdsSecret.grantRead(this.notionIndexer);
        props.captureBus.grantPutEventsTo(this.notionIndexer);

        this.notionIndexerBackfill = new KosLambda(this, 'NotionIndexerBackfill', {
          entry: path.join(__dirname, '../../../../services/notion-indexer-backfill/src/handler.ts'),
          timeout: Duration.minutes(15),
          memory: 1024,
          environment: {
            NOTION_TOKEN_SECRET_ARN: props.notionTokenSecret.secretArn,
            RDS_SECRET_ARN: props.rdsSecret.secretArn,
            RDS_ENDPOINT: props.rdsProxyEndpoint,
          },
        });
        props.notionTokenSecret.grantRead(this.notionIndexerBackfill);
        props.rdsSecret.grantRead(this.notionIndexerBackfill);

        const schedulerRole = new Role(this, 'SchedulerRole', {
          assumedBy: new ServicePrincipal('scheduler.amazonaws.com', {
            conditions: {
              ArnLike: { 'aws:SourceArn': `arn:aws:scheduler:${Stack.of(this).region}:${Stack.of(this).account}:schedule/${props.scheduleGroupName}/*` },
            },
          }),
        });
        this.notionIndexer.grantInvoke(schedulerRole);

        const watched = [
          { key: 'Entities', dbId: NOTION_IDS.entities, dbKind: 'entities' },
          { key: 'Projects', dbId: NOTION_IDS.projects, dbKind: 'projects' },
          { key: 'KevinContext', dbId: NOTION_IDS.kevinContext, dbKind: 'kevin_context' },
          { key: 'CommandCenter', dbId: NOTION_IDS.commandCenter, dbKind: 'command_center' }, // D-11: 4th watched DB
        ];

        for (const w of watched) {
          new CfnSchedule(this, 'IndexerSchedule-' + w.key, {
            name: 'notion-indexer-' + w.key.toLowerCase(),
            groupName: props.scheduleGroupName,
            scheduleExpression: 'rate(5 minutes)',
            scheduleExpressionTimezone: 'Europe/Stockholm',
            flexibleTimeWindow: { mode: 'OFF' },
            target: {
              arn: this.notionIndexer.functionArn,
              roleArn: schedulerRole.roleArn,
              input: JSON.stringify({ dbId: w.dbId, dbKind: w.dbKind }),
            },
            state: 'ENABLED',
          });
        }
      }
    }
