       ```typescript
       import { TranscribeClient, CreateVocabularyCommand, UpdateVocabularyCommand, GetVocabularyCommand, DeleteVocabularyCommand } from '@aws-sdk/client-transcribe';
       import { S3Client, PutObjectCommand, GetObjectCommand } from '@aws-sdk/client-s3';
       import * as fs from 'fs';
       import * as path from 'path';

       const VOCAB_NAME = 'kos-sv-se-v1';
       const LANG = 'sv-SE';

       export async function handler(event: any): Promise<any> {
         const region = process.env.TRANSCRIBE_REGION!;
         const bucket = process.env.VOCAB_BUCKET!;
         const s3Key = process.env.VOCAB_S3_KEY!; // e.g. vocab/sv-se-v1.txt
         const transcribe = new TranscribeClient({ region });
         const s3 = new S3Client({ region });

         if (event.RequestType === 'Delete') {
           // Archive-not-delete: do not delete vocabulary on stack delete
           return { PhysicalResourceId: VOCAB_NAME };
         }

         // Read seed vocab from the CDK Asset (S3) and re-upload under the canonical VOCAB_BUCKET/VOCAB_S3_KEY
         // path so Transcribe can read it. Using the Asset avoids `cp -r` bundling which fails on Windows.
         const seedBucket = process.env.VOCAB_SEED_BUCKET!;
         const seedKey = process.env.VOCAB_SEED_KEY!;
         const seedObj = await s3.send(new (await import('@aws-sdk/client-s3')).GetObjectCommand({ Bucket: seedBucket, Key: seedKey }));
         const content = await seedObj.Body!.transformToString('utf-8');
         await s3.send(new PutObjectCommand({ Bucket: bucket, Key: s3Key, Body: content, ContentType: 'text/plain; charset=utf-8' }));

         const s3Uri = `s3://${bucket}/${s3Key}`;

         // If vocabulary already exists, call UpdateVocabulary; otherwise Create
         let exists = false;
         try {
           await transcribe.send(new GetVocabularyCommand({ VocabularyName: VOCAB_NAME }));
           exists = true;
         } catch (e: any) {
           if (e.name !== 'BadRequestException' && e.name !== 'NotFoundException') throw e;
         }

         if (exists) {
           await transcribe.send(new UpdateVocabularyCommand({
             VocabularyName: VOCAB_NAME,
             LanguageCode: LANG,
             VocabularyFileUri: s3Uri,
           }));
         } else {
           await transcribe.send(new CreateVocabularyCommand({
             VocabularyName: VOCAB_NAME,
             LanguageCode: LANG,
             VocabularyFileUri: s3Uri,
           }));
         }

         // Poll until READY or FAILED (max 5 min)
         const deadline = Date.now() + 5 * 60 * 1000;
         while (Date.now() < deadline) {
           const v = await transcribe.send(new GetVocabularyCommand({ VocabularyName: VOCAB_NAME }));
           if (v.VocabularyState === 'READY') {
             return { PhysicalResourceId: VOCAB_NAME, Data: { vocabularyName: VOCAB_NAME } };
           }
           if (v.VocabularyState === 'FAILED') {
             throw new Error(`Vocabulary FAILED: ${v.FailureReason}`);
           }
           await new Promise(r => setTimeout(r, 10000));
         }
         throw new Error('Vocabulary did not reach READY within 5 min');
       }
       ```

    import { CustomResource, Duration } from 'aws-cdk-lib';
    import { Provider } from 'aws-cdk-lib/custom-resources';
    import { Asset } from 'aws-cdk-lib/aws-s3-assets';

    // Bundle the seed vocab file as a CDK Asset (S3 upload at synth time). Avoids the cross-platform
    // `cp -r` bundling which fails on Windows (win32 dev hosts).
    const vocabAsset = new Asset(this, 'VocabSeedAsset', {
      path: path.join(__dirname, '../../../../vocab/sv-se-v1.txt'),
    });

    const vocabFn = new KosLambda(this, 'TranscribeVocabDeploy', {
      entry: path.join(__dirname, '../../../../services/transcribe-vocab-deploy/src/handler.ts'),
      timeout: Duration.minutes(10),
      memory: 512,
      environment: {
        TRANSCRIBE_REGION: props.transcribeRegion,
        VOCAB_BUCKET: props.blobsBucket.bucketName,
        VOCAB_S3_KEY: 'vocab/sv-se-v1.txt',
        // Lambda reads the seed vocab from this S3 asset URI instead of a bundled file
        VOCAB_SEED_S3_URI: vocabAsset.s3ObjectUrl,
        VOCAB_SEED_BUCKET: vocabAsset.s3BucketName,
        VOCAB_SEED_KEY: vocabAsset.s3ObjectKey,
      },
    });
    vocabAsset.grantRead(vocabFn);
    // Grants
    props.blobsBucket.grantReadWrite(vocabFn);
    vocabFn.addToRolePolicy(new PolicyStatement({
      actions: ['transcribe:CreateVocabulary', 'transcribe:UpdateVocabulary', 'transcribe:GetVocabulary'],
      resources: ['*'],
    }));

    const vocabProvider = new Provider(this, 'VocabProvider', { onEventHandler: vocabFn });
    new CustomResource(this, 'TranscribeVocabulary', {

      },
    });
