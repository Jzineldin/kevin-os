  ```typescript
  export interface PersonCandidate { name: string; context_snippet: string; confidence: 'high' | 'medium' | 'low'; source_hint: 'signature' | 'header' | 'body' }
  const BLOCKLIST = new Set(['kevin el-zarka', 'kevin elzarka', 'kevin', 'tale forge', 'outbehaving', 'kos', 'notion', 'google', 'aws', 'slack', 'bedrock', 'anthropic', 'damien lovell']);
  const NAME_2W_RE = /\b([A-ZÅÄÖ][a-zåäö]{2,})\s+([A-ZÅÄÖ][a-zåäö]{2,})\b/g;
  const FROM_HEADER_RE = /From:\s*"?([^"<]+?)"?\s*<([^>]+)>/i;
  const SIGN_OFF_RE = /(?:Best|Thanks|Regards|Mvh|Vänligen|Hälsningar|Cheers)[,!]?\s*\n\s*([A-ZÅÄÖ][a-zåäö]+\s+[A-ZÅÄÖ][a-zåäö]+)/i;

  export function extractPersonCandidates(text: string): PersonCandidate[] {
    const out = new Map<string, PersonCandidate>();
    const add = (c: PersonCandidate) => {
      const k = c.name.trim().toLowerCase();
      if (BLOCKLIST.has(k)) return;
      if (!out.has(k) || (c.confidence === 'high' && out.get(k)!.confidence !== 'high')) out.set(k, c);
    };
    // HIGH: From: header
    for (const m of text.matchAll(new RegExp(FROM_HEADER_RE, 'g'))) {
      const name = m[1].trim();
      if (/\s/.test(name)) add({ name, context_snippet: `From: ${m[0]}`.slice(0, 300), confidence: 'high', source_hint: 'header' });
    }
    // HIGH: sign-off
    const so = text.match(SIGN_OFF_RE);
    if (so) add({ name: so[1].trim(), context_snippet: so[0].slice(-300), confidence: 'high', source_hint: 'signature' });
    // MEDIUM: capitalised 2-word sequences
    for (const m of text.matchAll(NAME_2W_RE)) {
      const full = `${m[1]} ${m[2]}`;
      // Avoid "Monday April" / "Ping Damien" noise by requiring both words len ≥ 3
      if (m[1].length < 3 || m[2].length < 3) continue;
      const start = Math.max(0, m.index! - 40);
      const end = Math.min(text.length, m.index! + full.length + 40);
      add({ name: full, context_snippet: text.slice(start, end), confidence: 'medium', source_hint: 'body' });
    }
    return Array.from(out.values());
  }
  ```

  ```typescript
  import type { Client } from '@notionhq/client';

  export async function discoverTranskriptenDbId(client: Client): Promise<string> {
    const res = await client.search({ query: 'Transkripten', filter: { property: 'object', value: 'database' } });
    const dbs = res.results.filter((r) => (r as { object: string }).object === 'database');
    if (dbs.length === 0) throw new Error('Transkripten DB not found via notion.search — Kevin must create it OR set TRANSKRIPTEN_DB_ID env var');
    if (dbs.length > 1) throw new Error(`Expected 1 Transkripten DB; found ${dbs.length}. Set TRANSKRIPTEN_DB_ID env explicitly.`);
    return (dbs[0] as { id: string }).id;
  }

  export async function *readTranskripten(client: Client, dbId: string, daysBack = 90): AsyncGenerator<{ id: string; title: string; bodyText: string }> {
    const cutoff = new Date(Date.now() - daysBack * 86400_000).toISOString();
    let cursor: string | undefined;
    do {
      const res = await client.databases.query({
        database_id: dbId,
        filter: { property: 'last_edited_time', last_edited_time: { on_or_after: cutoff.slice(0, 10) } } as never,
        page_size: 100, start_cursor: cursor,
      });
      for (const row of res.results as Array<{ id: string; properties: Record<string, unknown> }>) {
        // Prefer a Transcript rich_text property if present, else concatenate block children
        const titleProp = row.properties.Name ?? row.properties.Title;
        const title = (titleProp as { title?: Array<{ plain_text: string }> })?.title?.[0]?.plain_text ?? '(untitled)';
        const transcriptProp = row.properties.Transcript ?? row.properties.Body;
        let bodyText = '';
        if (transcriptProp) {
          bodyText = ((transcriptProp as { rich_text?: Array<{ plain_text: string }> }).rich_text ?? []).map((t) => t.plain_text).join(' ');
        } else {
          // fetch block children (limit 100 top-level; good enough for 90d bulk)
          const children = await client.blocks.children.list({ block_id: row.id, page_size: 100 });
          bodyText = children.results.map((b) => JSON.stringify(b)).join(' ').slice(0, 50_000);
        }
        yield { id: row.id, title, bodyText };
      }
      cursor = res.has_more ? (res.next_cursor ?? undefined) : undefined;
    } while (cursor);
  }
  ```

  **Step D — `services/bulk-import-granola-gmail/src/gmail.ts`:**
  ```typescript
  import { google, type gmail_v1 } from 'googleapis';
  import { SecretsManagerClient, GetSecretValueCommand } from '@aws-sdk/client-secrets-manager';

  export async function buildGmailClient(): Promise<gmail_v1.Gmail> {
    const sm = new SecretsManagerClient({ region: process.env.AWS_REGION ?? 'eu-north-1' });
    const s = await sm.send(new GetSecretValueCommand({ SecretId: 'kos/gmail-oauth-tokens' }));
    const tokens = JSON.parse(s.SecretString!) as { client_id: string; client_secret: string; refresh_token: string };
    const oauth2 = new google.auth.OAuth2(tokens.client_id, tokens.client_secret);
    oauth2.setCredentials({ refresh_token: tokens.refresh_token });
    return google.gmail({ version: 'v1', auth: oauth2 });
  }

  export async function *readGmailSignatures(gmail: gmail_v1.Gmail, daysBack = 90): AsyncGenerator<{ from: string; snippet: string }> {
    let pageToken: string | undefined;
    const q = `newer_than:${daysBack}d`;
    do {
      const list = await gmail.users.messages.list({ userId: 'me', q, maxResults: 500, pageToken });
      for (const m of list.data.messages ?? []) {
        if (!m.id) continue;
        const msg = await gmail.users.messages.get({ userId: 'me', id: m.id, format: 'metadata', metadataHeaders: ['From'] });
        const fromHeader = msg.data.payload?.headers?.find((h) => h.name === 'From')?.value ?? '';
        const snippet = msg.data.snippet ?? '';
        if (fromHeader) yield { from: fromHeader, snippet };
      }
      pageToken = list.data.nextPageToken ?? undefined;
    } while (pageToken);
  }
  ```
