  ```typescript
  import * as Sentry from '@sentry/aws-serverless';
  import { SecretsManagerClient, GetSecretValueCommand } from '@aws-sdk/client-secrets-manager';

  let initialised = false;

  /** Initialise Sentry at cold start. Safe to call multiple times (idempotent).
   *  DSN fetched from Secrets Manager the first time; cached in module scope.
   *  Graceful degradation: if DSN unavailable (secret empty), runs without Sentry
   *  rather than blocking the Lambda. */
  export async function initSentry(): Promise<void> {
    if (initialised) return;
    const arn = process.env.SENTRY_DSN_SECRET_ARN;
    if (!arn) { initialised = true; return; }
    try {
      const sm = new SecretsManagerClient({ region: process.env.AWS_REGION ?? 'eu-north-1' });
      const r = await sm.send(new GetSecretValueCommand({ SecretId: arn }));
      const dsn = r.SecretString;
      if (!dsn || dsn === 'PLACEHOLDER') { initialised = true; return; }
      Sentry.init({
        dsn,
        environment: process.env.KOS_ENV ?? 'prod',
        tracesSampleRate: 0.0,
        sampleRate: 1.0,
        beforeBreadcrumb: (bc) => {
          // Scrub any breadcrumb with a key that smells like a token
          if (bc.data) for (const k of Object.keys(bc.data)) if (/TOKEN|SECRET|KEY/i.test(k)) delete bc.data[k];
          return bc;
        },
      });
    } catch {
      // Sentry init failure MUST NOT kill the Lambda
    }
    initialised = true;
  }

  export const wrapHandler = Sentry.wrapHandler;
  export { Sentry };
  ```

  import { Stack, Duration, type StackProps } from 'aws-cdk-lib';
  import type { Construct } from 'constructs';
  import { Alarm, ComparisonOperator, TreatMissingData } from 'aws-cdk-lib/aws-cloudwatch';
  import { SnsAction } from 'aws-cdk-lib/aws-cloudwatch-actions';
  import { Topic } from 'aws-cdk-lib/aws-sns';
  import type { IFunction } from 'aws-cdk-lib/aws-lambda';

  export interface ObservabilityStackProps extends StackProps {
    telegramBotFn: IFunction;
    agentLambdas: IFunction[];   // triage, voice-capture, resolver — cold-path; not bulk imports
  }

  export class ObservabilityStack extends Stack {
    public readonly alarmTopic: Topic;
    constructor(scope: Construct, id: string, props: ObservabilityStackProps) {
      super(scope, id, props);

      this.alarmTopic = new Topic(this, 'KosAlarmTopic', { displayName: 'KOS Observability alarms' });
      const action = new SnsAction(this.alarmTopic);

      // Resolved Open Q4 — telegram-bot cold-start monitoring
      new Alarm(this, 'TelegramBotP95LatencyHigh', {
        alarmName: 'kos-telegram-bot-p95-duration-high',
        alarmDescription: 'telegram-bot p95 duration > 1800ms over 5-min window → investigate cold starts; consider provisioned concurrency',
        metric: props.telegramBotFn.metricDuration({ statistic: 'p95', period: Duration.minutes(5) }),
        threshold: 1800,
        evaluationPeriods: 2,
        datapointsToAlarm: 2,
        comparisonOperator: ComparisonOperator.GREATER_THAN_THRESHOLD,
        treatMissingData: TreatMissingData.NOT_BREACHING,
      }).addAlarmAction(action);

      // Agent error rate — fires when ANY of the 3 agents' error rate > 5% over 15 min
      for (const fn of props.agentLambdas) {
        new Alarm(this, `AgentErrorRateHigh-${fn.node.id}`, {
          alarmName: `kos-${fn.node.id.toLowerCase()}-error-rate-high`,
          alarmDescription: `${fn.node.id} error rate > 5% over 15-min window`,
          metric: fn.metricErrors({ statistic: 'Sum', period: Duration.minutes(15) }),
          threshold: 5,         // 5 errors in 15 min ~= 5% at Phase 2 volume
          evaluationPeriods: 1,
          comparisonOperator: ComparisonOperator.GREATER_THAN_THRESHOLD,
          treatMissingData: TreatMissingData.NOT_BREACHING,
        }).addAlarmAction(action);
      }
    }
  }
  ```

  #!/usr/bin/env node
  // Verifies: (1) Langfuse cloud has ≥ 1 trace for a capture_id passed as --capture-id,
  // (2) Sentry has received at least one event in the last 60 min.
  import { SecretsManagerClient, GetSecretValueCommand } from '@aws-sdk/client-secrets-manager';

  const args = new Map(process.argv.slice(2).flatMap((a, i, arr) => a.startsWith('--') ? [[a.slice(2), arr[i + 1]]] : []));
  const captureId = args.get('capture-id');
  if (!captureId) { console.error('Usage: verify-observability.mjs --capture-id <ulid>'); process.exit(1); }

  const region = process.env.AWS_REGION ?? 'eu-north-1';
  const sm = new SecretsManagerClient({ region });
  const pub = (await sm.send(new GetSecretValueCommand({ SecretId: 'kos/langfuse-public-key' }))).SecretString;
  const sec = (await sm.send(new GetSecretValueCommand({ SecretId: 'kos/langfuse-secret-key' }))).SecretString;

  // Langfuse REST: GET /api/public/traces?name=<capture_id>... authenticated with basic auth (public:secret)
  const auth = Buffer.from(`${pub}:${sec}`).toString('base64');
  const base = process.env.LANGFUSE_BASE_URL ?? 'https://cloud.langfuse.com';
  const res = await fetch(`${base}/api/public/traces?sessionId=${encodeURIComponent(captureId)}&limit=10`, {
    headers: { Authorization: `Basic ${auth}` },
  });
  if (!res.ok) { console.error(`[ERR] Langfuse ${res.status}: ${await res.text()}`); process.exit(1); }
  const j = await res.json();
  const count = (j.data ?? []).length;
  console.log(`[i] Langfuse traces for session=${captureId}: ${count}`);
  if (count === 0) { console.error('[ERR] no Langfuse trace found — did the Lambda flush?'); process.exit(1); }
  console.log('[OK] observability verified');
  ```
