  **Step B — `scripts/verify-phase-2-e2e.mjs`** (operator script, long-running):
  ```javascript
  #!/usr/bin/env node
  // End-to-end Phase 2a gate.
  //   1. Generate ulid capture_id.
  //   2. Upload fixture audio to S3 + write meta sidecar.
  //   3. Publish capture.received (kind=voice) to kos.capture bus.
  //   4. Poll Transcribe job kos-<capture_id> → COMPLETED (max 90s).
  //   5. Poll mention_events + agent_runs (via RDS Proxy IAM tunnel) for capture_id until voice-capture OK row appears + ≥ 0 mention.resolved events (empty fixture may produce 0).
  //   6. Assert push-telegram invoked once with is_reply=true (via Lambda metric or CloudWatch Logs Insights on the voice-capture Lambda).
  //   7. Stopwatch total elapsed → assert ≤ 25_000 ms OR ≤ 45_000 ms with a warning log (per D-02 "SLO not deadline").
  //   8. Write `.planning/phases/02-minimum-viable-loop/02-11-e2e-result-<ts>.json` with {capture_id, elapsed_ms, transcript, mention_count, stage_breakdown, langfuse_trace_ids}.
  import { S3Client, PutObjectCommand } from '@aws-sdk/client-s3';
  import { EventBridgeClient, PutEventsCommand } from '@aws-sdk/client-eventbridge';
  import { TranscribeClient, GetTranscriptionJobCommand } from '@aws-sdk/client-transcribe';
  import { readFileSync, writeFileSync } from 'node:fs';
  import { ulid } from 'ulid';
  // (Lambda invoke + RDS poll helpers abbreviated; full implementation expands inline.)
  const region = process.env.AWS_REGION ?? 'eu-north-1';
  const startedAt = Date.now();
  const capture_id = ulid();
  const bucket = process.env.BLOBS_BUCKET;
  if (!bucket) { console.error('BLOBS_BUCKET env required'); process.exit(1); }
  const fixturePath = process.env.VERIFY_FIXTURE ?? 'scripts/fixtures/sample-sv-voice-memo.oga';
  const bytes = readFileSync(fixturePath);
  const s3 = new S3Client({ region });
  const eb = new EventBridgeClient({ region });
  const tr = new TranscribeClient({ region });
  const audioKey = `audio/${new Date().getUTCFullYear()}/${String(new Date().getUTCMonth() + 1).padStart(2, '0')}/${capture_id}.oga`;
  const metaKey = `audio/meta/${capture_id}.json`;
  await s3.send(new PutObjectCommand({ Bucket: bucket, Key: audioKey, Body: bytes, ContentType: 'audio/ogg' }));
  await s3.send(new PutObjectCommand({
    Bucket: bucket, Key: metaKey, ContentType: 'application/json',
    Body: JSON.stringify({
      raw_ref: { s3_bucket: bucket, s3_key: audioKey, duration_sec: 5, mime_type: 'audio/ogg' },
      sender: { id: Number(process.env.KEVIN_TELEGRAM_USER_ID ?? 111222333), display: 'Kevin' },
      received_at: new Date().toISOString(),
      telegram: { chat_id: Number(process.env.KEVIN_TELEGRAM_USER_ID ?? 111222333), message_id: 1 },
    }),
  }));
  await eb.send(new PutEventsCommand({ Entries: [{
    EventBusName: 'kos.capture', Source: 'kos.capture', DetailType: 'capture.received',
    Detail: JSON.stringify({
      capture_id, channel: 'telegram', kind: 'voice',
      raw_ref: { s3_bucket: bucket, s3_key: audioKey, duration_sec: 5, mime_type: 'audio/ogg' },
      sender: { id: Number(process.env.KEVIN_TELEGRAM_USER_ID ?? 111222333), display: 'Kevin' },
      received_at: new Date().toISOString(),
      telegram: { chat_id: Number(process.env.KEVIN_TELEGRAM_USER_ID ?? 111222333), message_id: 1 },
    }),
  }] }));

  // Poll Transcribe
  const jobName = `kos-${capture_id}`;
  const tDeadline = Date.now() + 90_000;
  let status = 'UNKNOWN';
  while (Date.now() < tDeadline) {
    try { const j = await tr.send(new GetTranscriptionJobCommand({ TranscriptionJobName: jobName })); status = j.TranscriptionJob?.TranscriptionJobStatus ?? 'UNKNOWN'; if (status === 'COMPLETED' || status === 'FAILED') break; } catch {}
    await new Promise((r) => setTimeout(r, 2000));
  }
  console.log(`[${status}] transcribe job=${jobName} after ${Math.round((Date.now() - startedAt) / 1000)}s`);

  // Poll agent_runs via pg (assumes SSM tunnel — same setup as scripts/db-migrate-0003.sh)
  // Implementation detail: for Phase 2a gate we defer this poll to a small helper `poll-agent-runs.mjs`
  // that wraps scripts/db-migrate-0003.sh's connection block; omitted inline for brevity.

  const elapsed = Date.now() - startedAt;
  console.log(`[*] total elapsed: ${elapsed} ms`);
  if (elapsed > 25_000) console.warn(`[WARN] exceeded 25s SLO (D-02)`);
  if (elapsed > 45_000) console.error(`[ERR] exceeded 45s max (D-02 handoff threshold)`);

  const ts = new Date().toISOString().slice(0, 19).replace(/:/g, '-');
  writeFileSync(`.planning/phases/02-minimum-viable-loop/02-11-e2e-result-${ts}.json`, JSON.stringify({
    capture_id, transcribe_status: status, elapsed_ms: elapsed, started_at: new Date(startedAt).toISOString(),
  }, null, 2));
  console.log(`[OK] wrote 02-11-e2e-result-${ts}.json`);
  ```

  import { EventBridgeClient, PutEventsCommand } from '@aws-sdk/client-eventbridge';
  import { readFileSync, writeFileSync } from 'node:fs';
  import { ulid } from 'ulid';
  // ...setup...
  const mentions = JSON.parse(readFileSync('scripts/fixtures/resolver-three-stage-mentions.json', 'utf8'));
  const results = [];
  for (const m of mentions) {
    const capture_id = ulid();
    await eb.send(new PutEventsCommand({ Entries: [{
      EventBusName: 'kos.agent', Source: 'kos.agent', DetailType: 'entity.mention.detected',
      Detail: JSON.stringify({
        capture_id, mention_text: m.mention, context_snippet: m.context,
        candidate_type: m.type ?? 'Person', source: 'telegram-text',
        occurred_at: new Date().toISOString(),
      }),
    }] }));
    // ...poll mention.resolved via CloudWatch Logs Insights on entity-resolver Lambda...
    results.push({ mention: m.mention, expected: m.expected_stage, actual: '<fill-in-from-Langfuse>', passed: null });
  }
  const ts = new Date().toISOString().slice(0, 10);
  const md = [
    `# Phase 2 Resolver Three-Stage Scoreboard — ${ts}`, '',
