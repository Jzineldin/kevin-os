    1. **Create `services/dashboard-listen-relay/src/buffer.ts`** — pure ring buffer:
       ```ts
       import type { SseEvent } from '@kos/contracts/dashboard';
       export class RingBuffer {
         private items: Array<SseEvent & { seq: number }> = [];
         private nextSeq = 1;
         constructor(private max = 256) {}
         push(ev: SseEvent) { const r = { ...ev, seq: this.nextSeq++ }; this.items.push(r); while (this.items.length > this.max) this.items.shift(); return r.seq; }
         since(cursor: number): Array<SseEvent & { seq: number }> { return this.items.filter(e => e.seq > cursor); }
         get size() { return this.items.length; }
         get maxSeq() { return this.nextSeq - 1; }
       }
       ```

       ```ts
       import createSubscriber from 'pg-listen';
       import { Signer } from '@aws-sdk/rds-signer';
       import { SseEventSchema } from '@kos/contracts/dashboard';
       import type { RingBuffer } from './buffer.js';

       export async function startSubscriber(buffer: RingBuffer) {
         const signer = new Signer({
           hostname: process.env.RDS_PROXY_ENDPOINT!,
           port: 5432,
           username: process.env.RDS_USER ?? 'dashboard_relay',
           region: process.env.AWS_REGION!,
         });
         const subscriber = createSubscriber({
           host: process.env.RDS_PROXY_ENDPOINT,
           port: 5432,
           user: process.env.RDS_USER ?? 'dashboard_relay',
           database: 'kos',
           ssl: { rejectUnauthorized: true },
         }, {
           // pg-listen accepts a retrieveGetPayload-style password via connect options; wrap in a pre-connect hook:
         });
         // pg-listen uses pg.Client under the hood and supports pg connection options.
         // The proxy IAM token refresh is handled by re-connecting after `error`.
         subscriber.notifications.on('kos_output', (raw: unknown) => {
           try {
             const payload = typeof raw === 'string' ? JSON.parse(raw) : raw;
             const ev = SseEventSchema.parse(payload);
             buffer.push(ev);
           } catch (err) {
             console.warn('[relay] dropped malformed NOTIFY', err);
           }
         });
         subscriber.events.on('error', (err) => { console.error('[relay] subscriber error, exiting for ECS restart', err); process.exit(1); });
         subscriber.events.on('connected', () => console.log('[relay] LISTEN connected'));
         // IAM password callback (pg-listen exposes underlying client via options; if not supported,
         // fall back to connection-string with freshly-signed token re-applied on each reconnect via
         // process.exit(1) → ECS restart (described in RESEARCH §13 gotchas)):
         (subscriber as any).connect = async function () {
           const password = await signer.getAuthToken();
           (subscriber as any).options.password = password;
           return subscriber.connect();
         };
         await subscriber.connect();
         await subscriber.listenTo('kos_output');
         return subscriber;
       }
       ```

    3. **Create `services/dashboard-listen-relay/src/index.ts`** — Fastify entry:
       ```ts
       import Fastify from 'fastify';
       import { RingBuffer } from './buffer.js';
       import { startSubscriber } from './subscriber.js';

       const app = Fastify({ logger: { level: 'info' }, disableRequestLogging: true });
       const buffer = new RingBuffer(256);
       let subscriberHealthy = false;

       app.get('/healthz', async (_req, reply) => {
         if (!subscriberHealthy) return reply.code(500).send({ ok: false, reason: 'subscriber not connected' });
         return { ok: true, buffered: buffer.size, max_seq: buffer.maxSeq };
       });

       app.get('/events', async (req, reply) => {
         const q = req.query as { cursor?: string; wait?: string };
         const cursor = Number(q.cursor ?? '0') || 0;
         const wait = Math.min(Math.max(Number(q.wait ?? '0') || 0, 0), 25);
         const immediate = buffer.since(cursor);
         if (immediate.length > 0 || wait === 0) {
           return { events: immediate, cursor: buffer.maxSeq };
         }
         const deadline = Date.now() + wait * 1000;
         while (Date.now() < deadline) {
           await new Promise(r => setTimeout(r, 500));
           const pending = buffer.since(cursor);
           if (pending.length) return { events: pending, cursor: buffer.maxSeq };
         }
         return { events: [], cursor: buffer.maxSeq };
       });

       const sub = await startSubscriber(buffer);
       sub.events.on('connected', () => { subscriberHealthy = true; });
       sub.events.on('error', () => { subscriberHealthy = false; });

       await app.listen({ host: '0.0.0.0', port: Number(process.env.PORT ?? 8080) });
       ```

       ```ts
       import type { EventBridgeHandler } from 'aws-lambda';
       import pg from 'pg';
       import { Signer } from '@aws-sdk/rds-signer';
       import { SseEventSchema } from '@kos/contracts/dashboard';

       const ALLOWED_KINDS = new Set(['inbox_item','entity_merge','capture_ack','draft_ready','timeline_event'] as const);

       export const handler: EventBridgeHandler<string, any, { ok: true }> = async (event) => {
         const detailType = event['detail-type'];
         if (!ALLOWED_KINDS.has(detailType as any)) {
           console.warn('[dashboard-notify] ignored detail-type', detailType);
           return { ok: true };
         }
         const payload = SseEventSchema.parse({

           id: String(event.detail?.id ?? event.id),
           entity_id: event.detail?.entity_id,
           ts: event.detail?.ts ?? new Date().toISOString(),
         });

         const signer = new Signer({
           hostname: process.env.RDS_PROXY_ENDPOINT!,
           port: 5432,
           username: 'dashboard_notify',
           region: process.env.AWS_REGION!,
         });
         const client = new pg.Client({
           host: process.env.RDS_PROXY_ENDPOINT,
           port: 5432,
           user: 'dashboard_notify',
           database: 'kos',
           ssl: { rejectUnauthorized: true },
           password: await signer.getAuthToken(),
         });
         await client.connect();
         try {
           await client.query('SELECT pg_notify($1, $2)', ['kos_output', JSON.stringify(payload)]);
         } finally {
           await client.end();
         }
         return { ok: true };
       };
       ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| EventBridge→dashboard-notify Lambda | AWS-internal; EventBridge rule is the only invocation path |
| dashboard-notify Lambda→RDS | IAM auth as user `dashboard_notify` with NOTIFY-only privileges |
| RDS LISTEN→dashboard-listen-relay | Fargate task reads every NOTIFY payload; payload size ≤ 8KB (Postgres cap) ≤ 256 bytes typical |
| dashboard-listen-relay→Vercel /api/stream | Public HTTPS via Function URL wrapper (Plan 05 Option B); IAM-authenticated |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-3-03-01 | Tampering | NOTIFY payload | mitigate | SseEventSchema.parse at BOTH emit (dashboard-notify) AND receive (relay subscriber); malformed dropped with warn log. |
| T-3-03-02 | Information disclosure | NOTIFY payload | mitigate | Per D-25, payload contains ONLY pointers (kind/id/entity_id/ts) — no PII. Full rows fetched via authenticated API layer. |
| T-3-03-03 | DoS | long-poll `/events` endpoint | mitigate | Long-poll wait capped at 25s per RESEARCH §13. Single-user threat model; Plan 05 IAM authorizer limits callers. |
| T-3-03-04 | Spoofing | /events endpoint public ingress | mitigate | Plan 05 wraps relay behind Function URL with AWS_IAM auth; only Vercel's `kos-dashboard-caller` IAM user can invoke. |
| T-3-03-05 | DoS | dashboard-notify Lambda storms | accept | Single-user volume (<100 events/day per RESEARCH §R-05); Lambda concurrency cap default is 1000 — more than enough. |
| T-3-03-06 | Elevation | RDS user privileges | mitigate | `dashboard_notify` user has ONLY `rds-db:connect` + `EXECUTE` on `pg_notify` (no SELECT/INSERT). Plan 05 enforces. |
</threat_model>
