Compose Wave 1 infrastructure into a new `DashboardStack`. Stands up dashboard-api Lambda (Function URL + AWS_IAM), dashboard-notify Lambda (EventBridge rule on `kos.output`), dashboard-listen-relay ECS Fargate service (ARM64, singleton task, on existing kos-cluster), a thin relay-proxy Lambda Function URL (Option B ingress per phase scope guidance), dashboard IAM users + caller policies, Secrets Manager placeholders (`kos/dashboard-bearer-token`, `kos/sentry-dsn-dashboard`, `kos/dashboard-caller-access-keys`), and the `NOTION_TODAY_PAGE_ID` env wiring.

Output: `pnpm --filter @kos/cdk synth` produces a clean template; unit tests green; `pnpm --filter @kos/cdk deploy DashboardStack --require-approval never` dry-run succeeds.
</objective>

       ```ts
       import { FunctionUrlAuthType, InvokeMode, HttpMethod, LambdaFunctionUrlCorsOptions } from 'aws-cdk-lib/aws-lambda';
       import { Rule, IEventBus } from 'aws-cdk-lib/aws-events';

       import { KosLambda } from '../constructs/kos-lambda.js';
       import * as path from 'node:path';

       export interface DashboardStackProps extends StackProps {
         vpc: IVpc;
         outputBus: IEventBus;
         rdsProxyEndpoint: string;
         rdsProxySecurityGroup: SecurityGroup;
         notionTokenSecretArn: string;
         vercelOriginUrl: string; // e.g. https://kos-dashboard-kevin-elzarka.vercel.app

       export class DashboardStack extends Stack {
         public readonly dashboardApiFunctionUrl: string;
         public readonly relayProxyFunctionUrl: string;

         constructor(scope: Construct, id: string, props: DashboardStackProps) {
           super(scope, id, props);
           const sg = new SecurityGroup(this, 'DashboardLambdaSG', { vpc: props.vpc, description: 'dashboard-api + dashboard-notify egress to RDS Proxy', allowAllOutbound: true });

           // dashboard-api Lambda
           const dashboardApi = new KosLambda(this, 'DashboardApi', {
             entry: path.join(__dirname, '../../../..', 'services/dashboard-api/src/index.ts'),
             timeout: Duration.seconds(30),
             memory: 1024,
             vpc: props.vpc,
             vpcSubnets: { subnetType: SubnetType.PRIVATE_ISOLATED },
       ```

       ```ts
       import type { LambdaFunctionURLHandler } from 'aws-lambda';
       export const handler: LambdaFunctionURLHandler = async (event) => {
         const target = process.env.RELAY_INTERNAL_URL!;
         const method = event.requestContext.http.method;
         const path = event.rawPath + (event.rawQueryString ? `?${event.rawQueryString}` : '');
         const url = `${target}${path}`;
         const res = await fetch(url, {
           method,
           headers: { 'accept': 'application/json' },
           body: method === 'POST' || method === 'PUT' ? event.body ?? '' : undefined,
           // Long-poll timeout: Fargate task replies within <25s; give Lambda 30s leeway
           signal: AbortSignal.timeout(30_000),
         });
         const body = await res.text();
         return { statusCode: res.status, headers: { 'content-type': res.headers.get('content-type') ?? 'application/json' }, body };
       };
       ```

    2. **Create `packages/cdk/lib/stacks/integrations-dashboard.ts`** — helper holding the Fargate + NLB + relay-proxy composition (keeps dashboard-stack.ts manageable). Key CDK shape (reference `capture-stack.ts` for existing kos-cluster reuse; if the cluster is created in a different stack, import it via `Cluster.fromClusterAttributes`):
       ```ts
       import { Construct } from 'constructs';
       import { Duration } from 'aws-cdk-lib';
       import { Cluster, FargateTaskDefinition, ContainerImage, AwsLogDriver, FargateService, CpuArchitecture, OperatingSystemFamily } from 'aws-cdk-lib/aws-ecs';
       import { NetworkLoadBalancer, NetworkTargetGroup, Protocol } from 'aws-cdk-lib/aws-elasticloadbalancingv2';
       import { RetentionDays, LogGroup } from 'aws-cdk-lib/aws-logs';
       import { IVpc, SubnetType, SecurityGroup, Port } from 'aws-cdk-lib/aws-ec2';
       import { PolicyStatement, Effect } from 'aws-cdk-lib/aws-iam';
       import { FunctionUrlAuthType, InvokeMode } from 'aws-cdk-lib/aws-lambda';
       import { KosLambda } from '../constructs/kos-lambda.js';
       import * as path from 'node:path';

       export interface DashboardRelayProps {
         vpc: IVpc;
         cluster: Cluster;
         rdsProxyEndpoint: string;
         rdsProxySecurityGroup: SecurityGroup;
       }

       export function buildRelayStack(scope: Construct, props: DashboardRelayProps): { relayProxyFunctionArn: string; relayProxyUrl: string; } {
         // Fargate task definition (ARM64, 256/512 per RESEARCH §13)
         const taskDef = new FargateTaskDefinition(scope, 'RelayTaskDef', {
           cpu: 256, memoryLimitMiB: 512,
           runtimePlatform: { cpuArchitecture: CpuArchitecture.ARM64, operatingSystemFamily: OperatingSystemFamily.LINUX },
         });
         taskDef.addContainer('Relay', {
           image: ContainerImage.fromAsset(path.join(__dirname, '../../../..', 'services/dashboard-listen-relay')),
           logging: new AwsLogDriver({ streamPrefix: 'relay', logGroup: new LogGroup(scope, 'RelayLog', { logGroupName: '/ecs/dashboard-listen-relay', retention: RetentionDays.ONE_MONTH }) }),
           environment: { RDS_PROXY_ENDPOINT: props.rdsProxyEndpoint, AWS_REGION: 'eu-north-1', RDS_USER: 'dashboard_relay' },
           portMappings: [{ containerPort: 8080 }],
           healthCheck: { command: ['CMD-SHELL', 'wget -qO- http://127.0.0.1:8080/healthz || exit 1'], interval: Duration.seconds(30), retries: 3, timeout: Duration.seconds(3) },
         });
         taskDef.taskRole.addToPrincipalPolicy(new PolicyStatement({ effect: Effect.ALLOW, actions: ['rds-db:connect'], resources: [`arn:aws:rds-db:*:*:dbuser:*/dashboard_relay`] }));

         const relaySg = new SecurityGroup(scope, 'RelaySG', { vpc: props.vpc, allowAllOutbound: true });
         props.rdsProxySecurityGroup.addIngressRule(relaySg, Port.tcp(5432));

         const service = new FargateService(scope, 'RelayService', {
           cluster: props.cluster, taskDefinition: taskDef, desiredCount: 1,
           vpcSubnets: { subnetType: SubnetType.PRIVATE_ISOLATED }, securityGroups: [relaySg],
           assignPublicIp: false,
         });

         // Internal NLB in front of the service
         const nlb = new NetworkLoadBalancer(scope, 'RelayNlb', { vpc: props.vpc, internetFacing: false, vpcSubnets: { subnetType: SubnetType.PRIVATE_ISOLATED } });
         const tg = new NetworkTargetGroup(scope, 'RelayTg', { vpc: props.vpc, port: 8080, protocol: Protocol.TCP, targets: [service.loadBalancerTarget({ containerName: 'Relay', containerPort: 8080 })] });
         nlb.addListener('RelayListener', { port: 8080, protocol: Protocol.TCP, defaultTargetGroups: [tg] });

         // relay-proxy Lambda (Option B: Lambda Function URL → internal NLB)
         const proxyFn = new KosLambda(scope, 'RelayProxy', {
           entry: path.join(__dirname, '../../../..', 'services/dashboard-listen-relay/src/proxy.ts'),
           timeout: Duration.seconds(30),
           memory: 256,
           vpc: props.vpc,
           vpcSubnets: { subnetType: SubnetType.PRIVATE_ISOLATED },
           securityGroups: [relaySg],
           environment: { RELAY_INTERNAL_URL: `http://${nlb.loadBalancerDnsName}:8080` },
         });
         const url = proxyFn.addFunctionUrl({ authType: FunctionUrlAuthType.AWS_IAM, invokeMode: InvokeMode.BUFFERED });
         return { relayProxyFunctionArn: proxyFn.functionArn, relayProxyUrl: url.url };
       }
       ```
