       export function constantTimeEqual(a: string, b: string): boolean {
         if (typeof a !== 'string' || typeof b !== 'string') return false;
         if (a.length !== b.length) return false;
         let out = 0;
         for (let i = 0; i < a.length; i++) out |= a.charCodeAt(i) ^ b.charCodeAt(i);
         return out === 0;
       }

       import { constantTimeEqual } from '@/lib/constant-time';

       const PUBLIC_PATHS = ['/login', '/api/auth/login'];
       const COOKIE = 'kos_session';

       export function middleware(req: NextRequest) {
         const { pathname } = req.nextUrl;
         if (
           PUBLIC_PATHS.some(p => pathname === p || pathname.startsWith(p + '/')) ||
           pathname.startsWith('/_next') ||
           pathname === '/favicon.ico' ||
           pathname === '/sw.js' ||
           pathname === '/manifest.webmanifest' ||
           pathname.startsWith('/icons/')
         ) {
           return NextResponse.next();
         }
         const token = req.cookies.get(COOKIE)?.value;
         const expected = process.env.KOS_DASHBOARD_BEARER_TOKEN;
         if (!token || !expected || !constantTimeEqual(token, expected)) {
           const url = req.nextUrl.clone();
           url.pathname = '/login';
           url.searchParams.set('return', pathname);
           return NextResponse.redirect(url);
         }
         return NextResponse.next();
       }

       export const config = {
         matcher: ['/((?!_next/static|_next/image|favicon.ico|sw.js|manifest\\.webmanifest|icons/).*)'],
       };

       ```ts
       import { AwsClient } from 'aws4fetch';
       import type { z } from 'zod';

       let _client: AwsClient | null = null;
       function getClient() {
         if (_client) return _client;
         _client = new AwsClient({
           accessKeyId: process.env.AWS_ACCESS_KEY_ID_DASHBOARD!,
           secretAccessKey: process.env.AWS_SECRET_ACCESS_KEY_DASHBOARD!,
           region: process.env.AWS_REGION ?? 'eu-north-1',
           service: 'lambda',
         });
         return _client;
       }
       const BASE = () => process.env.KOS_DASHBOARD_API_URL!;

       export async function callApi<T>(path: string, init: RequestInit, schema: z.ZodSchema<T>): Promise<T> {
         const res = await getClient().fetch(`${BASE()}${path}`, {
           ...init,
           headers: { 'content-type': 'application/json', ...(init.headers ?? {}) },
         });
         if (!res.ok) throw new Error(`dashboard-api ${path} → ${res.status}: ${await res.text()}`);
         return schema.parse(await res.json());
       }

       // Relay long-poll — different SigV4 target but same client (both service=lambda)
       export async function callRelay(path: string, init: RequestInit): Promise<Response> {
         const base = process.env.KOS_DASHBOARD_RELAY_URL!;
         const res = await getClient().fetch(`${base}${path}`, {
           ...init,
           headers: { ...(init.headers ?? {}) },
         });
         return res;
       }
       ```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-3-05-01 | Information disclosure | Bearer token in logs | mitigate | Token NEVER rendered in React tree (form input is `type=password`; token is cookie-only after login). Sentry scrubber adds `cookie` + `authorization` to scrubbed headers (Plan 06 wires Sentry). |
| T-3-05-02 | Tampering (CSRF) | /api/auth/login | mitigate | SameSite=Lax cookie; Next Server Actions include origin check. /api/auth/login POST requires JSON body; form-encoded CSRF attempts rejected by schema parse. |
| T-3-05-03 | Spoofing | Cookie forgery | mitigate | `constantTimeEqual` cookie compare prevents timing leak (V2). 90-day maxAge + rotation procedure (Plan 11) limits window. |
| T-3-05-04 | Information disclosure | SigV4 access keys | mitigate | Keys stored as encrypted Vercel env vars; never `NEXT_PUBLIC_*` prefixed (P-01 + RESEARCH §16). Unit test linter asserts no `NEXT_PUBLIC_AWS` pattern anywhere. |
| T-3-05-05 | Elevation | Middleware bypass | mitigate | Explicit matcher allowlist; all non-static routes go through middleware; Plan 03-00 ESLint bans Node imports in middleware.ts to prevent accidental runtime demotion. |
| T-3-05-06 | Tampering | Login body injection | mitigate | `LoginRequestSchema.safeParse` rejects anything other than `{ token: string }`; large payloads via Next's default body-size limit (1MB). |
</threat_model>
