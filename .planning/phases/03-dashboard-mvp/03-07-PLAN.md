       ```ts
       import type { NextRequest } from 'next/server';
       import { callRelay } from '@/lib/dashboard-api';
       import { SseEventSchema } from '@kos/contracts/dashboard';

       export const runtime = 'nodejs';
       export const maxDuration = 300;
       export const dynamic = 'force-dynamic';

       const HEARTBEAT_MS = 15_000;
       const STREAM_DEADLINE_MS = 280_000; // close before Vercel's 300s hard cap (P-09)
       const LONG_POLL_WAIT_S = 25;
       const BACKOFF_MIN_MS = 500;
       const BACKOFF_MAX_MS = 5_000;

       export async function GET(req: NextRequest) {
         const encoder = new TextEncoder();
         const started = Date.now();
         let cursor = 0;
         let backoff = BACKOFF_MIN_MS;

         const stream = new ReadableStream({
           async start(controller) {
             const write = (s: string) => {
               try { controller.enqueue(encoder.encode(s)); } catch {}
             };
             write(`: connected at ${new Date().toISOString()}\n\n`);
             const heartbeat = setInterval(() => write(`: heartbeat\n\n`), HEARTBEAT_MS);
             const closeSoon = setTimeout(() => { clearInterval(heartbeat); try { controller.close(); } catch {} }, STREAM_DEADLINE_MS);
             req.signal.addEventListener('abort', () => { clearInterval(heartbeat); clearTimeout(closeSoon); try { controller.close(); } catch {} });

             try {
               while (!req.signal.aborted && Date.now() - started < STREAM_DEADLINE_MS) {
                 try {
                   const res = await callRelay(`/events?cursor=${cursor}&wait=${LONG_POLL_WAIT_S}`, { method: 'GET', signal: req.signal });
                   if (!res.ok) { write(`: relay ${res.status}, retrying\n\n`); await sleep(backoff); backoff = Math.min(backoff * 2, BACKOFF_MAX_MS); continue; }
                   backoff = BACKOFF_MIN_MS;
                   const body = await res.json() as { events: Array<{ seq: number } & unknown>; cursor: number };
                   for (const rawEv of body.events ?? []) {
                     try {
                       const ev = SseEventSchema.parse(rawEv);
                       const seq = (rawEv as any).seq;
                       write(`id: ${seq}\ndata: ${JSON.stringify(ev)}\n\n`);
                     } catch { /* skip malformed */ }
                   }
                   if (typeof body.cursor === 'number' && body.cursor > cursor) cursor = body.cursor;
                 } catch (err: any) {
                   if (err?.name === 'AbortError') break;
                   write(`: reconnecting\n\n`);
                   await sleep(backoff);
                   backoff = Math.min(backoff * 2, BACKOFF_MAX_MS);
                 }
               }
             } finally {
               clearInterval(heartbeat);
               clearTimeout(closeSoon);
             }
           },
           cancel() {},
         });

         return new Response(stream, {
           headers: {
             'Content-Type': 'text/event-stream; charset=utf-8',
             'Cache-Control': 'no-cache, no-transform',
             'Connection': 'keep-alive',
             'X-Accel-Buffering': 'no',
           },
         });
       }

       function sleep(ms: number) { return new Promise(r => setTimeout(r, ms)); }
       ```

    2. **Create `apps/dashboard/src/components/system/SseProvider.tsx`**:
       ```tsx
       'use client';
       import { createContext, useContext, useEffect, useRef, useState } from 'react';
       import type { SseEvent } from '@kos/contracts/dashboard';
       import { BACKOFF_MIN, BACKOFF_MAX, nextBackoff, parseMessage } from '@/lib/sse-client';

       type Handler = (ev: SseEvent) => void;
       type Ctx = { subscribe: (kind: SseEvent['kind'], h: Handler) => () => void; status: 'idle'|'connecting'|'open'|'closed' };
       const C = createContext<Ctx>({ subscribe: () => () => {}, status: 'idle' });
       export const useSse = () => useContext(C);
       export function useSseKind(kind: SseEvent['kind'], handler: Handler) {
         const { subscribe } = useSse();
         useEffect(() => subscribe(kind, handler), [kind, handler, subscribe]);
       }

       export function SseProvider({ children }: { children: React.ReactNode }) {
         const [status, setStatus] = useState<Ctx['status']>('idle');
         const subsRef = useRef<Map<string, Set<Handler>>>(new Map());
         const esRef = useRef<EventSource | null>(null);
         const backoffRef = useRef<number>(BACKOFF_MIN);

         useEffect(() => {
           let cancelled = false;
           const connect = () => {
             if (cancelled) return;
             setStatus('connecting');
             const es = new EventSource('/api/stream');
             esRef.current = es;
             es.onopen = () => { setStatus('open'); backoffRef.current = BACKOFF_MIN; };
             es.onmessage = (e) => {
               const ev = parseMessage(e.data);
               if (!ev) return;
               const subs = subsRef.current.get(ev.kind);
               if (subs) subs.forEach(h => { try { h(ev); } catch (err) { console.warn(err); } });
             };
             es.onerror = () => {
               setStatus('closed');
               es.close();
               const wait = backoffRef.current;
               backoffRef.current = nextBackoff(backoffRef.current);
               setTimeout(() => { if (!cancelled) connect(); }, wait);
             };
           };
           connect();
           return () => { cancelled = true; esRef.current?.close(); };
         }, []);

         return (
           <C.Provider value={{
             status,
             subscribe: (kind, h) => {
               let set = subsRef.current.get(kind);
               if (!set) { set = new Set(); subsRef.current.set(kind, set); }
               set.add(h);
               return () => set!.delete(h);
             }
           }}>{children}</C.Provider>
         );
       }
       ```

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-3-07-01 | DoS | /api/stream hold open | mitigate | Stream auto-closes at 280s; Vercel Pro concurrency headroom >>1 user; acceptable. Long-poll upstream capped at 25s — no infinite waits. |
| T-3-07-02 | Information disclosure | SSE payload to browser | mitigate | Payload is pointer-only per D-25; full row fetched via authenticated API layer (Plan 08/09 handlers). EventSource endpoint is cookie-authed via middleware. |

| T-3-07-04 | Tampering | CSRF against /api/stream | mitigate | EventSource does NOT send cross-origin cookies by default; SameSite=Lax cookie also blocks. |
| T-3-07-05 | Replay | Stream reconnect | accept | Browser may re-receive same events; client handlers must be idempotent (planned in Plan 08 — uses event `id` as dedup key). |
