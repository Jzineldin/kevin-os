    2. **Create `apps/dashboard/src/app/(app)/today/Composer.tsx`**:
       ```tsx
       'use client';
       import { useState, useTransition, useRef, useCallback } from 'react';
       import { Textarea } from '@/components/ui/textarea';
       import { Button } from '@/components/ui/button';
       import { toast } from 'sonner';
       import { PulseDot } from '@/components/system/PulseDot';
       import { useSseKind } from '@/components/system/SseProvider';
       import { captureText } from './actions';

       export function Composer() {
         const [text, setText] = useState('');
         const [pending, startTransition] = useTransition();
         const [waitingFor, setWaitingFor] = useState<string | null>(null);
         const [dotTone, setDotTone] = useState<'success'|'warning'|'accent'>('accent');
         const timeoutRef = useRef<number | null>(null);

         useSseKind('capture_ack', useCallback((ev) => {
           if (ev.id === waitingFor) {
             setDotTone('success'); setWaitingFor(null);
             if (timeoutRef.current) { clearTimeout(timeoutRef.current); timeoutRef.current = null; }
           }
         }, [waitingFor]));

         function onSubmit(e: React.FormEvent) {
           e.preventDefault();
           if (!text.trim()) return;
           const payload = text;
           setText('');
           setDotTone('warning');
           startTransition(async () => {
             try {
               const res = await captureText(payload);
               setWaitingFor(res.capture_id);
               toast.success(<span>Captured <code className="mono">{res.capture_id}</code></span>, { duration: 3000 });
               timeoutRef.current = window.setTimeout(() => { setDotTone('warning'); }, 5000);
             } catch (err) {
               toast.error("Capture didn't reach KOS. Retry?", { duration: Infinity, action: { label: 'Retry', onClick: () => { setText(payload); } } });
               setDotTone('accent');
             }
           });
         }

         return (
           <div className="side-card">
             <div className="h-section">CAPTURE</div>
             <form onSubmit={onSubmit} className="flex flex-col gap-3">
               <Textarea value={text} onChange={e => setText(e.target.value)} placeholder="Dumpa allt — en tanke, en idé, ett möte. KOS sorterar." rows={4} className="bg-[var(--color-surface-2)]" />
               <div className="flex items-center justify-between">
                 <PulseDot tone={dotTone} />
                 <Button type="submit" disabled={pending || !text.trim()}>Skicka</Button>
               </div>
             </form>
           </div>
         );
       }
       ```

|----------|-------------|
| Composer→Server Action→dashboard-api | Same-origin; SameSite cookie + SigV4 IAM both gate |
