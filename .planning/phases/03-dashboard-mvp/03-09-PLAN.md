       ```tsx
       'use client';
       import { useMemo, useOptimistic, useState, useTransition, useEffect, useCallback, useRef } from 'react';
       import { AnimatePresence, motion } from 'framer-motion';
       import { toast } from 'sonner';
       import type { InboxItem } from '@kos/contracts/dashboard';
       import { approveInbox, skipInbox, editInbox } from './actions';
       import { ItemRow } from './ItemRow';
       import { ItemDetail } from './ItemDetail';
       import { PulseDot } from '@/components/system/PulseDot';
       import { useKeys, isTypingInField } from '@/lib/tinykeys';
       import { useLiveRegion } from '@/components/system/LiveRegion';
       import { useSseKind } from '@/components/system/SseProvider';
       import { useRouter } from 'next/navigation';

       export function InboxClient({ initialItems, focusId }: { initialItems: InboxItem[]; focusId: string | null }) {
         const [items, setItems] = useState<InboxItem[]>(initialItems);
         const [optimistic, applyOptimistic] = useOptimistic(items, (cur, removeId: string) => cur.filter(i => i.id !== removeId));
         const initialIndex = useMemo(() => focusId ? Math.max(0, items.findIndex(i => i.id === focusId || i.merge_id === focusId.replace(/^resume-/, ''))) : 0, [items, focusId]);
         const [selectedIdx, setSelectedIdx] = useState(initialIndex);
         const [editMode, setEditMode] = useState(false);
         const [pending, startTransition] = useTransition();
         const { announce } = useLiveRegion();
         const router = useRouter();

         const selected = optimistic[selectedIdx] ?? null;

         // SSE — new inbox_item pushes → router.refresh() so RSC reloads
         const onNew = useCallback(() => { router.refresh(); announce('New inbox item'); }, [announce, router]);
         useSseKind('inbox_item', onNew);

         useKeys({
           'j': (e) => { if (isTypingInField(e.target)) return; e.preventDefault(); setSelectedIdx(i => Math.min(i + 1, optimistic.length - 1)); setEditMode(false); },
           'k': (e) => { if (isTypingInField(e.target)) return; e.preventDefault(); setSelectedIdx(i => Math.max(i - 1, 0)); setEditMode(false); },
           'Enter': (e) => {
             if (isTypingInField(e.target)) return; e.preventDefault();
             if (!selected) return;
             startTransition(async () => {
               applyOptimistic(selected.id);
               try { await approveInbox(selected.id); setItems(cur => cur.filter(i => i.id !== selected.id)); announce(`Approved ${selected.title}`); }
               catch { toast.error('Already handled elsewhere.', { duration: 4000 }); }
             });
           },
           's': (e) => {
             if (isTypingInField(e.target)) return; e.preventDefault();
             if (!selected) return;
             startTransition(async () => {
               applyOptimistic(selected.id);
               try { await skipInbox(selected.id); setItems(cur => cur.filter(i => i.id !== selected.id)); announce(`Skipped ${selected.title}`); }
               catch { toast.error('Already handled elsewhere.', { duration: 4000 }); }
             });
           },
           'e': (e) => { if (isTypingInField(e.target)) return; e.preventDefault(); if (selected) setEditMode(true); },
           'Escape': () => { if (editMode) setEditMode(false); },
         });

         if (optimistic.length === 0) {
           return (
             <div className="h-full grid place-items-center">
               <div className="text-center space-y-3">
                 <div className="flex items-center justify-center gap-2"><PulseDot tone="success" /><span className="text-[var(--color-text)] font-medium">Inbox clear. ✅</span></div>
                 <p className="text-sm text-[var(--color-text-3)]">Nothing to review. KOS surfaces drafts as they arrive.</p>
               </div>
             </div>
           );
         }

         return (
           <div className="grid h-[calc(100vh-52px-64px)]" style={{ gridTemplateColumns: '280px 1fr', borderTop: '1px solid var(--color-border)' }}>
             <aside className="overflow-auto border-r border-[var(--color-border)]">
               <div className="sticky top-0 bg-[var(--color-surface-1)] border-b border-[var(--color-border)] p-3 text-xs text-[var(--color-text-3)]">
                 <AnimatePresence mode="popLayout"><motion.span key={optimistic.length} initial={{ opacity: 0 }} animate={{ opacity: 1 }}>{optimistic.length} pending</motion.span></AnimatePresence>
               </div>
               <AnimatePresence initial={false}>
                 {optimistic.map((item, idx) => (
                   <motion.div key={item.id} initial={{ opacity: 0, y: 4 }} animate={{ opacity: 1, y: 0 }} exit={{ opacity: 0 }} transition={{ duration: 0.18, ease: [0.16, 1, 0.3, 1] }}>
                     <ItemRow item={item} selected={idx === selectedIdx} onClick={() => { setSelectedIdx(idx); setEditMode(false); }} />
                   </motion.div>
                 ))}
               </AnimatePresence>
               <div className="p-3 text-[11px] text-[var(--color-text-4)] font-mono">J / K to nav · Enter approve · E edit · S skip</div>
             </aside>
             <section className="overflow-auto">
               {selected ? <ItemDetail item={selected} editMode={editMode} onEditDone={() => setEditMode(false)} /> : <div className="p-8 text-[var(--color-text-3)]">Select an item</div>}
             </section>
           </div>
         );
       }
       ```

       'use client';
       import type { InboxItem } from '@kos/contracts/dashboard';
       import { Button } from '@/components/ui/button';
       import { toast } from 'sonner';
       export function ResumeMergeCard({ item }: { item: InboxItem }) {
         async function resume() {
           if (!item.merge_id) return;
           try {
             const r = await fetch(`/api/merge-resume?merge_id=${item.merge_id}`, { method: 'POST' });
             if (!r.ok) throw new Error(await r.text());
             toast.success('Merge resumed');
           } catch (err) { toast.error(`Resume failed: ${String(err)}`); }
         }
         return (
           <div className="p-8 space-y-6">
             <div><h2 className="h-page">Resume merge?</h2><p className="h-page-meta">A previous merge failed partway through. You can resume, revert, or cancel.</p></div>
             <div className="flex items-center gap-2 text-[var(--color-text-3)]"><span className="text-xs uppercase">merge_id</span><code className="mono text-xs">{item.merge_id}</code></div>
             <div className="flex gap-2">
               <Button onClick={resume}>Resume</Button>
               <Button variant="ghost" onClick={() => toast('Revert not implemented until Plan 10')} className="text-[var(--color-text-2)] hover:text-[var(--color-danger)]">Revert</Button>
               <Button variant="ghost" onClick={() => toast('Cancelled')}>Cancel</Button>

       import { callApi } from '@/lib/dashboard-api';
       import { MergeResponseSchema, MergeResumeRequestSchema } from '@kos/contracts/dashboard';
       export const runtime = 'nodejs';
       export async function POST(req: Request) {
         const url = new URL(req.url);
         const merge_id = url.searchParams.get('merge_id');
         if (!merge_id) return NextResponse.json({ error: 'missing merge_id' }, { status: 400 });
         // For Phase 3 Plan 09: call the target-entity-agnostic endpoint; Plan 10 adds the full handler.

           return NextResponse.json(res);
         } catch (err) {
           return NextResponse.json({ error: String(err) }, { status: 502 });
         }
       }
