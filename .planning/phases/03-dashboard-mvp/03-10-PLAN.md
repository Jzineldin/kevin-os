       ```tsx
       'use client';
       import { useCallback, useMemo, useState } from 'react';
       import { List, type RowComponentProps } from 'react-window';
       import { AnimatePresence, motion } from 'framer-motion';
       import type { TimelineRow, TimelinePage } from '@kos/contracts/dashboard';
       import { useSseKind } from '@/components/system/SseProvider';

       const ICON: Record<string, string> = { email: '✉', transcript: '🎙', doc: '📄', task: '✓', decision: '⚡', merge: '⟲', mention: '·', agent_run: '·' };

       function Row({ index, style, rows }: RowComponentProps<{ rows: TimelineRow[] }>) {
         const r = rows[index];
         return (
           <a href={r.href ?? '#'} style={style} className="grid items-start px-0" role="listitem">
             <div className="timeline-row grid" style={{ gridTemplateColumns: '20px 88px 1fr', gap: 12, padding: '10px 0', borderBottom: '1px dashed var(--color-border)' }}>
               <div className="mono text-[11px] text-[var(--color-text-3)]">{ICON[r.kind] ?? '·'}</div>
               <time className="mono text-[11px] text-[var(--color-text-3)]">{new Date(r.occurred_at).toLocaleString('sv-SE')}</time>
               <div className="min-w-0"><div className="text-sm">{r.source}</div><div className="text-xs text-[var(--color-text-2)] line-clamp-2 italic">{r.context}</div></div>
             </div>
           </a>
         );
       }

       export function Timeline({ entityId, initial }: { entityId: string; initial: TimelinePage }) {
         const [rows, setRows] = useState<TimelineRow[]>(initial.rows);
         const [cursor, setCursor] = useState<string | null>(initial.next_cursor);
         const [loading, setLoading] = useState(false);

         const loadMore = useCallback(async () => {
           if (loading || !cursor) return;
           setLoading(true);
           try {
             const res = await fetch(`/api/entities/${entityId}/timeline?cursor=${encodeURIComponent(cursor)}`);
             const page = (await res.json()) as TimelinePage;
             setRows(prev => [...prev, ...page.rows]);
             setCursor(page.next_cursor);
           } finally { setLoading(false); }
         }, [cursor, entityId, loading]);

         useSseKind('timeline_event', useCallback((ev) => {
           if (ev.entity_id !== entityId) return;
           // Re-fetch first page to pick up new row at top (cheap; ≤ 50 rows); dedupe by id
           fetch(`/api/entities/${entityId}/timeline`).then(r => r.json()).then((p: TimelinePage) => {
             setRows(prev => {
               const existing = new Set(prev.map(r => r.id));
               const newRows = p.rows.filter(r => !existing.has(r.id));
               return [...newRows, ...prev];
             });
           }).catch(() => {});
         }, [entityId]));

         return (
           <div style={{ height: 600 }} role="list">
             <AnimatePresence initial={false}>
               <List
                 rowCount={rows.length}
                 rowHeight={72}
                 rowComponent={Row as any}
                 rowProps={{ rows }}
                 onRowsRendered={({ stopIndex }: { stopIndex: number }) => {
                   if (stopIndex >= rows.length - 10) loadMore();
                 }}
               />
             </AnimatePresence>
           </div>
         );
       }
       ```

       ```ts
       register('GET', '/calendar/week', async ({ query }) => {
       ```
