       ```ts
       import { ulid } from 'ulid';
       import { eq, and } from 'drizzle-orm';
       import { register } from '../router.js';
       import { getDb } from '../db.js';
       import { getNotion } from '../notion.js';
       import { publishOutput } from '../events.js';
       import { OWNER_ID, ownerScoped } from '../owner-scoped.js';
       import { MergeRequestSchema, MergeResumeRequestSchema, MergeResponseSchema } from '@kos/contracts/dashboard';
       import { entityMergeAudit, inboxIndex, entityIndex, mentionEvents, agentRuns } from '@kos/db/schema';
       import { copyRelations, archiveNotionPage, unarchiveNotionPage, sourceNotionPageId } from './notion-merge.js';

       const STATE_ORDER = ['initiated','notion_relations_copied','notion_archived','rds_updated','complete'] as const;
       type State = typeof STATE_ORDER[number];

       async function updateState(db: any, merge_id: string, state: State, patch: Record<string, any> = {}) {
         await db.update(entityMergeAudit).set({ state, ...patch }).where(eq(entityMergeAudit.merge_id, merge_id));
       }
       async function markFailed(db: any, merge_id: string, lastState: string, err: unknown) {
         await db.update(entityMergeAudit).set({ state: `failed_at_${lastState}`, error_message: String(err) }).where(eq(entityMergeAudit.merge_id, merge_id));
         await db.insert(inboxIndex).values({
           id: `merge-${merge_id}`, kind: 'merge_resume', title: `Resume merge (${merge_id.slice(-6)})`,
           preview: `Merge paused at ${lastState}: ${String(err).slice(0, 140)}`,
           merge_id, payload: { failed_at: lastState, error: String(err) }, status: 'pending',
         });
       }

       async function runMergeSteps(db: any, notion: any, merge_id: string, source: string, target: string, startFrom: State): Promise<void> {
         if (STATE_ORDER.indexOf(startFrom) < STATE_ORDER.indexOf('notion_relations_copied')) {
           await copyRelations(notion, source, target);
           await updateState(db, merge_id, 'notion_relations_copied');
           await new Promise(r => setTimeout(r, 400)); // P-06
         }
         if (STATE_ORDER.indexOf(startFrom) < STATE_ORDER.indexOf('notion_archived')) {
           await archiveNotionPage(notion, source);
           await updateState(db, merge_id, 'notion_archived', { notion_archived_at: new Date() });
         }
         if (STATE_ORDER.indexOf(startFrom) < STATE_ORDER.indexOf('rds_updated')) {
           await db.transaction(async (tx: any) => {
             await tx.update(mentionEvents).set({ entity_id: target }).where(and(ownerScoped(mentionEvents), eq(mentionEvents.entity_id, source)));
             await tx.update(entityIndex).set({ status: 'merged_into' }).where(and(ownerScoped(entityIndex), eq(entityIndex.id, source)));
           });
           await updateState(db, merge_id, 'rds_updated', { rds_updated_at: new Date() });
         }
         if (STATE_ORDER.indexOf(startFrom) < STATE_ORDER.indexOf('complete')) {
           await db.insert(agentRuns).values({
             agent_name: 'entity_merge_manual', status: 'ok', capture_id: null,
             output_json: { source_id: source, target_id: target, merge_id, initiated_by: 'kevin' },
           });
           await updateState(db, merge_id, 'complete', { completed_at: new Date() });
           await publishOutput('entity_merge', { id: merge_id, entity_id: target, ts: new Date().toISOString() });
         }
       }

       register('POST', '/entities/:id/merge', async ({ params, body }) => {
         const parsed = MergeRequestSchema.parse(JSON.parse(body ?? '{}'));
         const merge_id = parsed.merge_id ?? ulid();
         const db = await getDb();
         const notion = getNotion();
         try {
           await db.insert(entityMergeAudit).values({
             merge_id, source_entity_id: parsed.source_id, target_entity_id: params.id,
             state: 'initiated', diff: parsed.diff,
           });
         } catch (err) {
           // Duplicate merge_id (ULID replay) — idempotent return
           return { statusCode: 409, body: JSON.stringify(MergeResponseSchema.parse({ ok: false, merge_id, resumable: true })) };
         }
         try {
           await runMergeSteps(db, notion, merge_id, parsed.source_id, params.id, 'initiated');
           return { statusCode: 200, body: JSON.stringify(MergeResponseSchema.parse({ ok: true, merge_id })) };
         } catch (err) {
           const [row] = await db.select().from(entityMergeAudit).where(eq(entityMergeAudit.merge_id, merge_id)).limit(1);
           await markFailed(db, merge_id, row?.state ?? 'initiated', err);
           return { statusCode: 500, body: JSON.stringify(MergeResponseSchema.parse({ ok: false, merge_id, resumable: true })) };
         }
       });

       register('POST', '/entities/:id/merge/resume', async ({ body, query }) => {
         const parsed = MergeResumeRequestSchema.parse(JSON.parse(body ?? '{}'));
         const action = query.action ?? 'resume';
         const db = await getDb();
         const [row] = await db.select().from(entityMergeAudit).where(eq(entityMergeAudit.merge_id, parsed.merge_id)).limit(1);
         if (!row) return { statusCode: 404, body: JSON.stringify({ error: 'not_found' }) };
         if (action === 'cancel') { await updateState(db, parsed.merge_id, 'cancelled' as any); return { statusCode: 200, body: JSON.stringify({ ok: true, merge_id: parsed.merge_id }) }; }
         if (action === 'revert') {
           const notion = getNotion();
           if (row.state === 'notion_archived' || row.state === 'rds_updated' || row.state === 'complete') await unarchiveNotionPage(notion, row.source_entity_id);
           if (row.state === 'rds_updated' || row.state === 'complete') {
             await db.transaction(async (tx: any) => {
               await tx.update(mentionEvents).set({ entity_id: row.source_entity_id }).where(eq(mentionEvents.entity_id, row.target_entity_id));
               await tx.update(entityIndex).set({ status: 'active' }).where(eq(entityIndex.id, row.source_entity_id));
             });
           }
           await updateState(db, parsed.merge_id, 'reverted' as any);
           return { statusCode: 200, body: JSON.stringify({ ok: true, merge_id: parsed.merge_id }) };
         }
         // default resume
         if (row.state === 'complete') return { statusCode: 200, body: JSON.stringify({ ok: true, merge_id: parsed.merge_id }) }; // idempotent
         const lastOk = row.state.replace(/^failed_at_/, '') as State;
         try {
           await runMergeSteps(db, getNotion(), parsed.merge_id, row.source_entity_id, row.target_entity_id, lastOk);
           return { statusCode: 200, body: JSON.stringify({ ok: true, merge_id: parsed.merge_id }) };
         } catch (err) {
           const [row2] = await db.select().from(entityMergeAudit).where(eq(entityMergeAudit.merge_id, parsed.merge_id)).limit(1);
           await markFailed(db, parsed.merge_id, row2?.state ?? lastOk, err);
           return { statusCode: 500, body: JSON.stringify({ ok: false, merge_id: parsed.merge_id, resumable: true }) };
         }
       });
       ```
