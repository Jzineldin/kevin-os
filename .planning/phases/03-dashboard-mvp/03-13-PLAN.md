       #!/usr/bin/env tsx
       import { CloudFormationClient, DescribeStacksCommand } from '@aws-sdk/client-cloudformation';
       import { SecretsManagerClient, GetSecretValueCommand } from '@aws-sdk/client-secrets-manager';
       import { execSync } from 'node:child_process';

       const REGION = 'eu-north-1';
       const STACK = 'DashboardStack';
       const cf = new CloudFormationClient({ region: REGION });
       const sm = new SecretsManagerClient({ region: REGION });

       async function cfOutput(name: string): Promise<string> {
         const out = await cf.send(new DescribeStacksCommand({ StackName: STACK }));
         const o = out.Stacks?.[0]?.Outputs?.find(x => x.OutputKey === name);
         if (!o?.OutputValue) throw new Error(`missing CF output ${name}`);
         return o.OutputValue;
       }
       async function secret(name: string): Promise<string> {
         const r = await sm.send(new GetSecretValueCommand({ SecretId: name }));
         if (!r.SecretString) throw new Error(`empty secret ${name}`);
         return r.SecretString;
       }
       function setVercel(key: string, value: string) {
         console.log(`Setting ${key}…`);
         try { execSync(`vercel env rm ${key} production -y`, { stdio: 'ignore' }); } catch {}
         const child = require('node:child_process').spawnSync('vercel', ['env','add',key,'production'], { input: value + '\n', encoding: 'utf-8' });
         if (child.status !== 0) throw new Error(`vercel env add ${key} failed: ${child.stderr}`);
       }

       (async () => {
         const apiUrl = await cfOutput('DashboardApiFunctionUrl');
         const relayUrl = await cfOutput('RelayProxyFunctionUrl');
         const bearerRaw = await secret('kos/dashboard-bearer-token');
         const bearer = JSON.parse(bearerRaw).token ?? bearerRaw;
         if (bearer === '{"placeholder":true}' || bearer.startsWith('{"placeholder')) {
           throw new Error('Bearer secret still contains placeholder JSON — populate it first:\n  aws secretsmanager put-secret-value --secret-id kos/dashboard-bearer-token --secret-string \'{"token":"<generate with `openssl rand -base64 32`>"}\'');
         }
         const callerRaw = await secret('kos/dashboard-caller-access-keys');
         const caller = JSON.parse(callerRaw);
         if (!caller.AccessKeyId) throw new Error('Caller access keys not populated — run `aws iam create-access-key --user-name kos-dashboard-caller` then store result into the secret.');
         const sentry = await secret('kos/sentry-dsn-dashboard');

         setVercel('KOS_DASHBOARD_API_URL', apiUrl);
         setVercel('KOS_DASHBOARD_RELAY_URL', relayUrl);
         setVercel('KOS_DASHBOARD_BEARER_TOKEN', bearer);
         setVercel('AWS_ACCESS_KEY_ID_DASHBOARD', caller.AccessKeyId);
         setVercel('AWS_SECRET_ACCESS_KEY_DASHBOARD', caller.SecretAccessKey);
         setVercel('AWS_REGION', 'eu-north-1');
         setVercel('SENTRY_DSN', sentry);
         setVercel('NEXT_PUBLIC_SENTRY_DSN', sentry);
         console.log('All env vars synced.');
       })().catch(err => { console.error(err); process.exit(1); });
       ```

       #!/usr/bin/env tsx
       import { execSync } from 'node:child_process';
       import fs from 'node:fs';

       const BASE = process.env.DEPLOY_URL ?? 'https://kos-dashboard-kevin-elzarka.vercel.app';
       const TOKEN = process.env.KOS_DASHBOARD_BEARER_TOKEN!;
       if (!TOKEN) { console.error('Set KOS_DASHBOARD_BEARER_TOKEN'); process.exit(1); }

       type Result = { name: string; pass: boolean; detail: string; reqs: string[] };
       const results: Result[] = [];
       let cookies = '';

       async function login() {
         const res = await fetch(`${BASE}/api/auth/login`, { method: 'POST', headers: { 'content-type': 'application/json' }, body: JSON.stringify({ token: TOKEN }) });
         if (!res.ok) throw new Error(`login failed ${res.status}`);
         cookies = res.headers.get('set-cookie')?.split(';')[0] ?? '';
       }
       async function check(path: string): Promise<{ ok: boolean; status: number; body: string }> {
         const res = await fetch(`${BASE}${path}`, { headers: { cookie: cookies } });
         return { ok: res.ok, status: res.status, body: await res.text() };
       }

       (async () => {
         await login();

         // SC1 — UI-01 INF-12
         const today = await check('/today');
         results.push({ name: 'SC1 Today view + auth', pass: today.ok && today.body.includes('h-page'), detail: `status=${today.status}`, reqs: ['UI-01','INF-12'] });

         // SC2 — UI-02 ENT-08
         // For SC2 we need a seeded entity id. Document that it must be passed via env var.
         if (process.env.SEED_ENTITY_ID) {
           const ent = await check(`/entities/${process.env.SEED_ENTITY_ID}`);
           results.push({ name: 'SC2 Entity dossier + timeline', pass: ent.ok, detail: `status=${ent.status}`, reqs: ['UI-02','ENT-08'] });
         } else {
           results.push({ name: 'SC2 Entity dossier + timeline', pass: false, detail: 'SEED_ENTITY_ID not set', reqs: ['UI-02','ENT-08'] });
         }

         // SC3 — ENT-07
         if (process.env.SEED_ENTITY_ID && process.env.SEED_SOURCE_ID) {
           const merge = await check(`/entities/${process.env.SEED_ENTITY_ID}/merge?source=${process.env.SEED_SOURCE_ID}`);
           results.push({ name: 'SC3 Merge review page', pass: merge.ok && /Confirm merge/.test(merge.body), detail: `status=${merge.status}`, reqs: ['ENT-07'] });
         } else {
           results.push({ name: 'SC3 Merge review page', pass: false, detail: 'SEED_ENTITY_ID+SEED_SOURCE_ID not set', reqs: ['ENT-07'] });
         }

         // SC4 — UI-06 (SSE) — soft-check: /api/stream returns text/event-stream within 5s
         const streamRes = await fetch(`${BASE}/api/stream`, { headers: { cookie: cookies } });
         const streamOk = streamRes.headers.get('content-type')?.includes('text/event-stream');
         results.push({ name: 'SC4 SSE stream content-type', pass: !!streamOk, detail: `content-type=${streamRes.headers.get('content-type')}`, reqs: ['UI-06'] });
         streamRes.body?.cancel();

         // SC5 — UI-05 PWA manifest + sw
         const manifest = await check('/manifest.webmanifest');
         const sw = await check('/sw.js');
         results.push({ name: 'SC5 PWA manifest + sw', pass: manifest.ok && manifest.body.includes('Kevin OS') && sw.ok, detail: `manifest=${manifest.status} sw=${sw.status}`, reqs: ['UI-05'] });

         // SC6 — UI-03 UI-04
         const inbox = await check('/inbox');
         const cal = await check('/calendar');
         results.push({ name: 'SC6 Inbox + Calendar', pass: inbox.ok && cal.ok, detail: `inbox=${inbox.status} calendar=${cal.status}`, reqs: ['UI-03','UI-04'] });

         // Summary
         const failed = results.filter(r => !r.pass);
         console.log('\n=== Phase 3 Verification ===');
         for (const r of results) console.log(`${r.pass ? 'PASS' : 'FAIL'}  ${r.name}  [${r.reqs.join(',')}]  ${r.detail}`);
         console.log(`\nTotal: ${results.length - failed.length}/${results.length} passed.`);
         fs.mkdirSync('.planning/phases/03-dashboard-mvp/.ephemeral', { recursive: true });
         fs.writeFileSync('.planning/phases/03-dashboard-mvp/.ephemeral/verify-report.json', JSON.stringify({ results, ts: new Date().toISOString() }, null, 2));
         process.exit(failed.length ? 1 : 0);
       })().catch(err => { console.error(err); process.exit(1); });
       ```
