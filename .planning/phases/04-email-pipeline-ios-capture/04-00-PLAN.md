CREATE TABLE email_drafts (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  owner_id uuid NOT NULL,
  capture_id text NOT NULL,
  account_id text NOT NULL,                           -- "kevin-elzarka" | "kevin-taleforge" | "forward"
  message_id text NOT NULL,
  from_email text NOT NULL,
  to_email text[] NOT NULL DEFAULT '{}',
  subject text,
  classification text NOT NULL CHECK (classification IN ('urgent','important','informational','junk','pending_triage')),
  draft_body text,
  draft_subject text,
  status text NOT NULL DEFAULT 'draft' CHECK (status IN ('pending_triage','draft','edited','approved','skipped','sent','failed')),
  received_at timestamptz NOT NULL,
  triaged_at timestamptz,
  approved_at timestamptz,
  sent_at timestamptz,
  sent_message_id text,
  CONSTRAINT email_drafts_account_message_uidx UNIQUE (account_id, message_id)
);
CREATE INDEX email_drafts_owner_status_idx ON email_drafts (owner_id, status, received_at DESC);
CREATE INDEX email_drafts_owner_classification_idx ON email_drafts (owner_id, classification, received_at DESC);

CREATE TABLE email_send_authorizations (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  owner_id uuid NOT NULL,
  draft_id uuid NOT NULL REFERENCES email_drafts(id) ON DELETE CASCADE,
  approved_by text NOT NULL DEFAULT 'kevin',
  approved_at timestamptz NOT NULL DEFAULT now(),
  consumed_at timestamptz,                            -- set by email-sender after SES call
  send_result jsonb
);
CREATE INDEX email_send_authorizations_owner_draft_idx ON email_send_authorizations (owner_id, draft_id);

CREATE TABLE agent_dead_letter (

  <action>
    Implement `withTimeoutAndRetry` per the signature in `<interfaces>`. Keep the implementation pure (no Lambda-specific imports; accepts injected pool + eventBridge for tests).

    **Core algorithm**:
    1. For attempt 0..maxRetries:
       - Race `fn()` against `setTimeout(reject(TimeoutError), timeoutMs)`.
       - On success: return value.
       - On caught error: if `shouldRetry(err)` AND attempt < maxRetries → wait `2^attempt * 1000 ms` (1s, 2s), continue. Else break.
    2. On final failure: call `writeDeadLetter(opts, lastErr)` — INSERT into `agent_dead_letter` AND emit `kos.output / inbox.dead_letter` event. Both calls wrapped in their own try/catch (do NOT re-throw from dead-letter-write path to avoid infinite loop — P-11). Log to console on dead-letter write failure.
    3. Re-throw the original error (caller decides whether to return a partial value or propagate).

    **Default shouldRetry**: return true for:
    - Error message matches `/timeout/i` (TimeoutError from our race)
    - Error object has `.name === 'ThrottlingException'`
    - Error object has `.statusCode >= 500`
    - Error object has `.code in ('ECONNRESET', 'ETIMEDOUT', 'EAI_AGAIN')`
    Otherwise return false (do not retry on 4xx, auth errors, schema errors).

    **writeDeadLetter implementation**: takes `pool` (RDS Proxy IAM auth — delegate to same pattern as services/triage/src/persist.ts; do NOT re-implement the pool bootstrap — export a `getPool()` helper that callers inject OR lazy-load). For Wave 0 scaffold, the function body is complete but callers (email-triage etc.) will inject the pool from their existing bootstrap.

    **Tests** (>=15):
    1. Happy path: fn resolves on first try → return value; no retry; no dead letter.
    2. Timeout on first attempt → retry once → succeeds on attempt 2 → return value.
    3. Throttle on first two attempts → succeeds on attempt 3 (maxRetries=2 means 3 total attempts: 0,1,2).
    4. Timeout on all 3 attempts → dead letter written; error rethrown.
