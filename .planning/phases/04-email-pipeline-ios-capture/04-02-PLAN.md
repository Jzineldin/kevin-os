    ```typescript
    import { simpleParser } from 'mailparser';

    export async function parseRawEmail(rfc822: Buffer): Promise<ParsedEmail> {
      if (!rfc822 || rfc822.length === 0) throw new Error('empty MIME buffer');
      const parsed = await simpleParser(rfc822);
      const messageId = (parsed.messageId ?? '').replace(/^<|>$/g, '').trim();
      if (!messageId) throw new Error('missing Message-ID header');

      const fromAddr = parsed.from?.value?.[0]?.address ?? parsed.from?.text ?? '';
      const toAddrs = (parsed.to && 'value' in parsed.to ? parsed.to.value : []).map(a => a.address).filter(Boolean) as string[];
      const ccAddrs = parsed.cc ? ('value' in parsed.cc ? parsed.cc.value.map(a => a.address).filter(Boolean) as string[] : []) : undefined;
      const subject = parsed.subject ?? '';
      const bodyText = parsed.text ?? '';
      const bodyHtml = typeof parsed.html === 'string' ? parsed.html : undefined;
      const receivedAt = parsed.date?.toISOString() ?? new Date().toISOString();

      return { messageId, from: fromAddr, to: toAddrs, cc: ccAddrs, subject, bodyText, bodyHtml, receivedAt };
    }
    ```

    **Note on capture_id determinism**: SES retries can cause double-invocation. We want identical capture_id so downstream dedupe works. Formula:
    ```typescript
    import { createHash } from 'node:crypto';
    import { ulid } from 'ulid';

    function deterministicCaptureId(messageId: string): string {
      // ULID time component = hash first 6 bytes mapped to 48-bit ms; random component = hash next 10 bytes mapped.
      // Simpler: use ulidFromHash — but `ulid` npm doesn't expose that. Use ulid() then if tests demand determinism, swap to:
      //   const hex = createHash('sha256').update(messageId).digest('hex');
      //   return crockfordFromHex(hex.slice(0, 52));
      // Pragmatic: compute a 26-char Crockford-base32 string from the sha256 of messageId, matching ULID alphabet + length.
      const CROCKFORD = '0123456789ABCDEFGHJKMNPQRSTVWXYZ';
      const hash = createHash('sha256').update(messageId).digest();
      let out = '';
      for (let i = 0; i < 26; i++) out += CROCKFORD[hash[i] % 32];
      return out;
    }
    ```
    Document in code comment that this is NOT a true ULID (timestamp not embedded); it satisfies the `UlidRegex` shape (`[0-9A-HJKMNP-TV-Z]{26}`) used by contracts. Acceptable for idempotency purposes.

    ```typescript
    import { initSentry, wrapHandler } from '../../_shared/sentry.js';
    import { setupOtelTracingAsync, flush as langfuseFlush, tagTraceWithCaptureId } from '../../_shared/tracing.js';
    import { EventBridgeClient, PutEventsCommand } from '@aws-sdk/client-eventbridge';
    import { S3Client, GetObjectCommand } from '@aws-sdk/client-s3';
    import { CaptureReceivedEmailForwardSchema } from '@kos/contracts';
    import { parseRawEmail } from './parse.js';
    import { createHash } from 'node:crypto';

    if (!process.env.AWS_REGION) process.env.AWS_REGION = 'eu-north-1';

    const eb = new EventBridgeClient({ region: 'eu-north-1' });
    const s3EuWest = new S3Client({ region: 'eu-west-1' });  // CROSS-REGION — P-1

    interface SesEvent { Records: Array<{ ses: { mail: { commonHeaders: { messageId?: string } }; receipt: { action: { bucketName?: string; objectKey?: string; type: string; } } } }>; }

    export const handler = wrapHandler(async (event: SesEvent) => {
      await initSentry();
      await setupOtelTracingAsync();

      if (!event.Records?.length) throw new Error('ses-inbound: no Records in event');

      // SES may batch — loop (rarely >1 in practice).
      const results = [];
      for (const rec of event.Records) {
        const { bucketName, objectKey, type } = rec.ses.receipt.action;
        if (type !== 'S3' || !bucketName || !objectKey) {
          console.warn('[ses-inbound] skipping non-S3 action', { type });
          continue;
        }

        const messageIdHint = rec.ses.mail.commonHeaders.messageId ?? '';

        // Fetch raw email from eu-west-1 bucket
        const obj = await s3EuWest.send(new GetObjectCommand({ Bucket: bucketName, Key: objectKey }));
        const chunks: Buffer[] = [];
        for await (const c of obj.Body as AsyncIterable<Uint8Array>) chunks.push(Buffer.from(c));
        const raw = Buffer.concat(chunks);

        const parsed = await parseRawEmail(raw);
        const captureId = deterministicCaptureIdFromMessageId(parsed.messageId);
        tagTraceWithCaptureId(captureId);

        const detail = CaptureReceivedEmailForwardSchema.parse({
          capture_id: captureId,
          channel: 'email-forward',
          kind: 'email_forward',
          email: {
            message_id: parsed.messageId,
            from: parsed.from,
            to: parsed.to,
            cc: parsed.cc,
            subject: parsed.subject,
            body_text: parsed.bodyText,
            body_html: parsed.bodyHtml,
            s3_ref: { bucket: bucketName, key: objectKey, region: 'eu-west-1' },
            received_at: parsed.receivedAt,
          },
          received_at: new Date().toISOString(),
        });

        await eb.send(new PutEventsCommand({
          Entries: [{
            EventBusName: 'kos.capture',
            Source: 'kos.capture',
            DetailType: 'capture.received',
            Detail: JSON.stringify(detail),
          }],
        }));

        results.push({ capture_id: captureId, message_id: parsed.messageId });
      }

      await langfuseFlush();
      return { processed: results.length, records: results };
    });

    function deterministicCaptureIdFromMessageId(messageId: string): string { /* per <action> formula */ }
    ```

    ```typescript
    export interface WireSesInboundProps {
      captureBus: EventBus;
      blobsBucket: Bucket;     // NOT used — ses-inbound doesn't write here; kept for interface symmetry
    }
    export interface SesInboundWiring { sesInboundFunction: KosLambda; }

    export function wireSesInbound(scope: Construct, props: WireSesInboundProps): SesInboundWiring {
      const fn = new KosLambda(scope, 'SesInbound', {
        entry: path.join(REPO_ROOT, 'services/ses-inbound/src/handler.ts'),
        memory: 512,
        timeout: Duration.seconds(30),
        environment: {
          KEVIN_OWNER_ID: '9e4be978-cc7d-571b-98ec-a1e92373682c',
          SES_INBOUND_BUCKET_NAME: 'kos-ses-inbound-euw1',  // documented operator-created
        },
      });

      // Cross-region S3 read on kos-ses-inbound-euw1-<account>/incoming/* (account-level wildcard)
      fn.addToRolePolicy(new PolicyStatement({
        effect: Effect.ALLOW,
        actions: ['s3:GetObject'],
        resources: [`arn:aws:s3:::kos-ses-inbound-euw1-*/incoming/*`],
      }));

      props.captureBus.grantPutEventsTo(fn);

      // Allow SES service principal in eu-west-1 to invoke this Lambda.
      fn.addPermission('AllowSesInvoke', {
        principal: new ServicePrincipal('ses.amazonaws.com'),
        action: 'lambda:InvokeFunction',
        sourceAccount: Stack.of(scope).account,
      });

      return { sesInboundFunction: fn };
    }
    ```
