    ```typescript
    import { initSentry, wrapHandler } from '../../_shared/sentry.js';
    import { setupOtelTracingAsync, flush as langfuseFlush, tagTraceWithCaptureId } from '../../_shared/tracing.js';
    import { EventBridgeClient, PutEventsCommand } from '@aws-sdk/client-eventbridge';
    import { CaptureReceivedEmailInboxSchema } from '@kos/contracts';
    import { getWebhookSecret } from './secrets.js';
    import { timingSafeEqual, createHash } from 'node:crypto';

    if (!process.env.AWS_REGION) process.env.AWS_REGION = 'eu-north-1';
    const eb = new EventBridgeClient({ region: 'eu-north-1' });

    interface FnUrlEvent {
      requestContext: { http: { method: string } };
      headers: Record<string, string>;
      body?: string;
      isBase64Encoded?: boolean;
    }

    export const handler = wrapHandler(async (event: FnUrlEvent) => {
      await initSentry();
      await setupOtelTracingAsync();
      try {
        if (event.requestContext.http.method !== 'POST') return { statusCode: 405, body: JSON.stringify({ error: 'method' }) };

        const secret = await getWebhookSecret();
        const provided = event.headers['x-ee-secret'] ?? event.headers['X-EE-Secret'];
        if (!provided) return { statusCode: 401, body: JSON.stringify({ error: 'missing_secret' }) };
        const a = Buffer.from(secret); const b = Buffer.from(provided);
        if (a.length !== b.length || !timingSafeEqual(a, b)) return { statusCode: 401, body: JSON.stringify({ error: 'bad_secret' }) };

        if (!event.body) return { statusCode: 400, body: JSON.stringify({ error: 'empty_body' }) };
        const raw = event.isBase64Encoded ? Buffer.from(event.body, 'base64').toString('utf8') : event.body;
        let payload: any;
        try { payload = JSON.parse(raw); } catch { return { statusCode: 400, body: JSON.stringify({ error: 'invalid_json' }) }; }

        // Only messageNew triggers capture; others acknowledged + ignored.
        if (payload.event !== 'messageNew') {
          return { statusCode: 200, body: JSON.stringify({ skipped: payload.event }) };
        }

        const messageId: string | undefined = payload.data?.messageId;
        if (!messageId) return { statusCode: 400, body: JSON.stringify({ error: 'missing_messageId' }) };

        const accountId: string = payload.account;
        const captureId = deterministicCaptureId(accountId, messageId);
        tagTraceWithCaptureId(captureId);

        const detail = CaptureReceivedEmailInboxSchema.parse({
          capture_id: captureId,
          channel: 'email-inbox',
          kind: 'email_inbox',
          email: {
            account_id: accountId,
            message_id: messageId,
            from: payload.data.from?.address ?? '',
            to: (payload.data.to ?? []).map((x: any) => x.address),
            cc: payload.data.cc?.map((x: any) => x.address),
            subject: payload.data.subject ?? '',
            body_text: payload.data.text?.plain ?? '',
            body_html: payload.data.text?.html,
            received_at: payload.data.date ?? new Date().toISOString(),
            imap_uid: payload.data.uid,
          },
          received_at: new Date().toISOString(),
        });

        await eb.send(new PutEventsCommand({
          Entries: [{ EventBusName: 'kos.capture', Source: 'kos.capture', DetailType: 'capture.received', Detail: JSON.stringify(detail) }],
        }));

        return { statusCode: 200, body: JSON.stringify({ capture_id: captureId }) };
      } finally {
        await langfuseFlush();
      }
    });

    function deterministicCaptureId(accountId: string, messageId: string): string {
      const CROCKFORD = '0123456789ABCDEFGHJKMNPQRSTVWXYZ';
      const hash = createHash('sha256').update(`${accountId}|${messageId}`).digest();
      let out = ''; for (let i = 0; i < 26; i++) out += CROCKFORD[hash[i] % 32];
      return out;
    }
    ```

    ```typescript
    import { initSentry, wrapHandler } from '../../_shared/sentry.js';
    import { SecretsManagerClient, GetSecretValueCommand } from '@aws-sdk/client-secrets-manager';
    import { z } from 'zod';

    const AdminCommandSchema = z.object({
      command: z.enum(['register-account', 'unregister-account', 'list-accounts']),
      account: z.string().optional(),
      accountSecretArn: z.string().optional(),   // for register: ARN of kos/emailengine-imap-<name> secret
    });

    const sm = new SecretsManagerClient({ region: process.env.AWS_REGION ?? 'eu-north-1' });

    export const handler = wrapHandler(async (event: { body: string; isBase64Encoded?: boolean }) => {
      await initSentry();
      const base = process.env.EE_REST_URL;
      const eeKey = process.env.EE_API_KEY;  // admin api key for EmailEngine REST
      if (!base) throw new Error('EE_REST_URL not set');

      const raw = event.isBase64Encoded ? Buffer.from(event.body, 'base64').toString() : event.body;
      const cmd = AdminCommandSchema.parse(JSON.parse(raw));

      if (cmd.command === 'list-accounts') {
        const r = await fetch(`${base}/v1/accounts`, { headers: { Authorization: `Bearer ${eeKey}` } });
        return { statusCode: r.status, body: await r.text() };
      }

      if (cmd.command === 'unregister-account') {
        if (!cmd.account) return { statusCode: 400, body: JSON.stringify({ error: 'account required' }) };
        const r = await fetch(`${base}/v1/account/${cmd.account}`, { method: 'DELETE', headers: { Authorization: `Bearer ${eeKey}` } });
        return { statusCode: r.status, body: await r.text() };
      }

      // register-account
      if (!cmd.account || !cmd.accountSecretArn) return { statusCode: 400, body: JSON.stringify({ error: 'account + accountSecretArn required' }) };
      const credR = await sm.send(new GetSecretValueCommand({ SecretId: cmd.accountSecretArn }));
      const cred = JSON.parse(credR.SecretString ?? '{}');  // { email, app_password }
      if (!cred.email || !cred.app_password) return { statusCode: 500, body: JSON.stringify({ error: 'secret missing email/app_password' }) };
      const payload = {
        account: cmd.account,
        name: cmd.account,
        email: cred.email,
        imap: { host: 'imap.gmail.com', port: 993, secure: true, auth: { user: cred.email, pass: cred.app_password } },
        smtp: false,                  // outbound via SES, not EE
        webhooks: true,
      };
      const r = await fetch(`${base}/v1/account`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json', Authorization: `Bearer ${eeKey}` },
        body: JSON.stringify(payload),
      });
      return { statusCode: r.status, body: await r.text() };
    });
    ```
