    import { SESClient, SendRawEmailCommand } from '@aws-sdk/client-ses';
    const ses = new SESClient({ region: process.env.AWS_REGION ?? 'eu-north-1' });

    export function buildRawMessage(i: RawMessageInput): string {
      const boundary = 'KOS-BOUNDARY-' + Math.random().toString(36).slice(2);
      const date = new Date().toUTCString();
      const messageId = `<${Date.now()}.${Math.random().toString(36).slice(2)}@kos.tale-forge.app>`;
      const headers: string[] = [
        `From: ${i.from}`,
        `To: ${i.to.join(', ')}`,
        ...(i.cc?.length ? [`Cc: ${i.cc.join(', ')}`] : []),
        `Subject: ${i.subject}`,
        `Date: ${date}`,
        `Message-ID: ${messageId}`,
        'MIME-Version: 1.0',
      ];
      if (i.inReplyTo) headers.push(`In-Reply-To: ${i.inReplyTo}`);
      if (i.references?.length) headers.push(`References: ${i.references.join(' ')}`);

      let body: string;
      if (i.bodyHtml) {
        headers.push(`Content-Type: multipart/alternative; boundary="${boundary}"`);
        body = [
          `--${boundary}`, 'Content-Type: text/plain; charset=utf-8', '', i.bodyText, '',
          `--${boundary}`, 'Content-Type: text/html; charset=utf-8', '', i.bodyHtml, '',
          `--${boundary}--`,
        ].join('\r\n');
      } else {
        headers.push('Content-Type: text/plain; charset=utf-8');
        body = i.bodyText;
      }
      return headers.join('\r\n') + '\r\n\r\n' + body;
    }

    export async function sendRawEmail(raw: string): Promise<{ messageId: string }> {
      const r = await ses.send(new SendRawEmailCommand({ RawMessage: { Data: Buffer.from(raw, 'utf8') } }));
      return { messageId: r.MessageId ?? '' };
    }

    import { initSentry, wrapHandler } from '../../_shared/sentry.js';
    import { setupOtelTracingAsync, flush as langfuseFlush, tagTraceWithCaptureId } from '../../_shared/tracing.js';
    import { withTimeoutAndRetry } from '../../_shared/with-timeout-retry.js';
    import { EventBridgeClient, PutEventsCommand } from '@aws-sdk/client-eventbridge';
    import { EmailApprovedSchema } from '@kos/contracts';
    import { loadDraftForSend, markDraftSent, markAuthorizationConsumed, markDraftFailed, getPool } from './persist.js';
    import { buildRawMessage, sendRawEmail } from './ses.js';

    const eb = new EventBridgeClient({ region: 'eu-north-1' });

    export const handler = wrapHandler(async (event: any) => {
      await initSentry();
      await setupOtelTracingAsync();
      const ownerId = process.env.KEVIN_OWNER_ID!;
      try {
        if (event['detail-type'] !== 'email.approved') return { skipped: event['detail-type'] };
        const detail = EmailApprovedSchema.parse(event.detail);
        tagTraceWithCaptureId(detail.capture_id);

        const pool = await getPool();
        const draft = await loadDraftForSend(pool, detail.draft_id, detail.authorization_id);
        if (!draft) {
          console.warn('[email-sender] draft not found or already consumed', { draft_id: detail.draft_id });
          return { skipped: 'not_found_or_consumed' };
        }

        const raw = buildRawMessage({
          from: draft.from_account_email,   // e.g. kevin@tale-forge.app
          to: [draft.reply_to],
          subject: draft.draft_subject,
          bodyText: draft.draft_body,
          inReplyTo: draft.in_reply_to,
          references: draft.references,
        });

        try {
          const sent = await withTimeoutAndRetry(
            () => sendRawEmail(raw),
            { timeoutMs: 10_000, maxRetries: 2, toolName: 'ses.send_raw', captureId: detail.capture_id, ownerId }
          );
          await markDraftSent(pool, detail.draft_id, sent.messageId);
          await markAuthorizationConsumed(pool, detail.authorization_id, { messageId: sent.messageId });
          await eb.send(new PutEventsCommand({
            Entries: [{ EventBusName: 'kos.output', Source: 'kos.output', DetailType: 'email.sent', Detail: JSON.stringify({ capture_id: detail.capture_id, draft_id: detail.draft_id, ses_message_id: sent.messageId, sent_at: new Date().toISOString() }) }],
          }));
          return { sent: sent.messageId };
        } catch (err) {
          await markDraftFailed(pool, detail.draft_id, String(err));
          throw err;  // withTimeoutAndRetry already dead-lettered
        }
      } finally {
        await langfuseFlush();
      }
    });

    import { z } from 'zod';
    import { EventBridgeClient, PutEventsCommand } from '@aws-sdk/client-eventbridge';
    import { randomUUID } from 'node:crypto';
    import { EmailApprovedSchema } from '@kos/contracts';
    import { loadDraftById, insertAuthorization, updateDraftForEdit, updateDraftSkip } from '../email-drafts-persist.js';

    const eb = new EventBridgeClient({ region: 'eu-north-1' });

    export async function approveEmailDraft(req: { draftId: string; ownerId: string }) {
      const pool = await getPool();
      const draft = await loadDraftById(pool, req.draftId, req.ownerId);
      if (!draft) return { status: 404, body: { error: 'not_found' } };
      if (draft.status !== 'draft' && draft.status !== 'edited') return { status: 409, body: { error: `cannot approve from status ${draft.status}` } };

      const authId = randomUUID();
      await insertAuthorization(pool, { id: authId, ownerId: req.ownerId, draftId: req.draftId });

      const payload = EmailApprovedSchema.parse({
        capture_id: draft.capture_id, draft_id: req.draftId, authorization_id: authId, approved_at: new Date().toISOString(),
      });
      await eb.send(new PutEventsCommand({
        Entries: [{ EventBusName: 'kos.output', Source: 'kos.output', DetailType: 'email.approved', Detail: JSON.stringify(payload) }],
      }));
      return { status: 200, body: { authorization_id: authId } };
    }

    const EditSchema = z.object({ body: z.string().min(1).max(10_000), subject: z.string().min(1).max(300) });

    export async function editEmailDraft(req: { draftId: string; ownerId: string; body: unknown }) {
      const pool = await getPool();
      const draft = await loadDraftById(pool, req.draftId, req.ownerId);
      if (!draft) return { status: 404, body: { error: 'not_found' } };
      if (draft.status !== 'draft' && draft.status !== 'edited') return { status: 409, body: { error: `cannot edit from status ${draft.status}` } };
      const parsed = EditSchema.safeParse(req.body);
      if (!parsed.success) return { status: 400, body: { error: 'invalid_body', issues: parsed.error.issues } };
      await updateDraftForEdit(pool, req.draftId, parsed.data.body, parsed.data.subject);
      await eb.send(new PutEventsCommand({
        Entries: [{ EventBusName: 'kos.output', Source: 'kos.output', DetailType: 'draft_edited', Detail: JSON.stringify({ capture_id: draft.capture_id, draft_id: req.draftId }) }],
      }));
      return { status: 200, body: { status: 'edited' } };
    }

    export async function skipEmailDraft(req: { draftId: string; ownerId: string }) {
      const pool = await getPool();
      const draft = await loadDraftById(pool, req.draftId, req.ownerId);
      if (!draft) return { status: 404, body: { error: 'not_found' } };
      if (draft.status === 'sent' || draft.status === 'approved') return { status: 409, body: { error: `cannot skip from status ${draft.status}` } };
      await updateDraftSkip(pool, req.draftId);
      await eb.send(new PutEventsCommand({
        Entries: [{ EventBusName: 'kos.output', Source: 'kos.output', DetailType: 'draft_skipped', Detail: JSON.stringify({ capture_id: draft.capture_id, draft_id: req.draftId }) }],
      }));
      return { status: 200, body: { status: 'skipped' } };
    }

    export async function listInbox(args: { ownerId: string; limit?: number }) {

         WHERE owner_id = $1 AND status IN ('draft','edited')
         ORDER BY received_at DESC LIMIT $2`,
