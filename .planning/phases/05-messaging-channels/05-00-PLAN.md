    ```sql
    -- phase 5 messaging
    -- Phase 5 Plan 05-00: WhatsApp session keys (Baileys pluggable auth), system alerts, sync status
    -- See .planning/phases/05-messaging-channels/05-CONTEXT.md D-06, D-18, D-19

    BEGIN;

    -- WhatsApp session keys — Baileys signal-protocol state persisted per owner
    -- owner_id + key_id is the composite PK; one row per (account, signal key id)
    -- Value stored as JSONB (serialised by Baileys pluggable auth provider)
    CREATE TABLE whatsapp_session_keys (
      owner_id UUID NOT NULL,
      key_id TEXT NOT NULL,                       -- e.g., 'creds', 'pre-key-1', 'session-46XXX@s.whatsapp.net'
      key_type TEXT NOT NULL,                     -- 'creds' | 'pre-key' | 'session' | 'sender-key' | ...
      value_jsonb JSONB NOT NULL,
      updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
      PRIMARY KEY (owner_id, key_id)
    );
    CREATE INDEX idx_whatsapp_session_keys_type ON whatsapp_session_keys(owner_id, key_type);

    -- System alerts — capture-side Dashboard alerts (no Telegram, no notification cap)
    CREATE TABLE system_alerts (
      id UUID NOT NULL DEFAULT gen_random_uuid() PRIMARY KEY,
      owner_id UUID NOT NULL,
      source TEXT NOT NULL,                       -- chrome | linkedin | whatsapp | discord | baileys | ...
      severity TEXT NOT NULL,                     -- info | warn | error | auth_fail | unusual_activity
      message TEXT NOT NULL,
      raised_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
      ack_at TIMESTAMPTZ                          -- NULL until Kevin acks in Dashboard
    );
    CREATE INDEX idx_system_alerts_owner_raised ON system_alerts(owner_id, raised_at DESC);
    CREATE INDEX idx_system_alerts_unacked ON system_alerts(owner_id, raised_at DESC) WHERE ack_at IS NULL;

    -- Sync status — per-channel last-healthy timestamp + queue depth for graceful-degrade
    CREATE TABLE sync_status (
      owner_id UUID NOT NULL,
      channel TEXT NOT NULL,                      -- chrome | linkedin | whatsapp | discord | emailengine
      last_healthy_at TIMESTAMPTZ,
      queue_depth INTEGER NOT NULL DEFAULT 0,
      paused_until TIMESTAMPTZ,                   -- set when channel silently disabled (e.g., LinkedIn 24h backoff)
      updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
      PRIMARY KEY (owner_id, channel)
    );

    COMMIT;
    ```
