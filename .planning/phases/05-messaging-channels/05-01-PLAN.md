    ```typescript
    // Web Crypto HMAC-SHA256 — available in service_worker + content scripts under MV3.
    export async function hmacSha256Hex(secret: string, msg: string): Promise<string> {
      const key = await crypto.subtle.importKey(
        'raw',
        new TextEncoder().encode(secret),
        { name: 'HMAC', hash: 'SHA-256' },
        false,
        ['sign'],
      );
      const sig = await crypto.subtle.sign('HMAC', key, new TextEncoder().encode(msg));
      return Array.from(new Uint8Array(sig)).map(b => b.toString(16).padStart(2, '0')).join('');
    }

    export interface SignedRequest { timestamp: number; signature: string; }

    export async function signRequest(body: string, secret: string): Promise<SignedRequest> {
      const t = Math.floor(Date.now() / 1000);
      const canonical = `${secret}.${t}.${body}`;
      const signature = await hmacSha256Hex(secret, canonical);
      return { timestamp: t, signature };
    }

    export function formatSignatureHeader(s: SignedRequest): string {
      return `t=${s.timestamp},v1=${s.signature}`;
    }
    ```

      webhookUrl: string;       // base URL; /highlight, /linkedin, /linkedin/alert appended at call sites

    **`apps/chrome-extension/src/background.ts`** — MV3 service_worker:
    ```typescript
    import { signRequest, formatSignatureHeader } from './lib/hmac.js';
    import { loadConfig, isConfigured } from './lib/storage.js';

    // Install / startup: register context menu
    chrome.runtime.onInstalled.addListener(() => {
      chrome.contextMenus.create({
        id: 'kos-send-to-kos',
        title: 'Send to KOS',
        contexts: ['selection'],
      });
    });

    chrome.contextMenus.onClicked.addListener(async (info, tab) => {
      if (info.menuItemId !== 'kos-send-to-kos') return;
      if (!info.selectionText || !tab?.url) return;
      if (!(await isConfigured())) {
        console.warn('[KOS] extension not configured — open Options');
        return;
      }
      const cfg = (await loadConfig()) as Required<Awaited<ReturnType<typeof loadConfig>>>;
      const captureId = generateUlid();
      const body = JSON.stringify({
        capture_id: captureId,
        channel: 'chrome',
        kind: 'chrome_highlight',
        text: info.selectionText,
        source_url: tab.url,
        source_title: tab.title ?? null,
        selected_at: new Date().toISOString(),
      });
      const signed = await signRequest(body, cfg.hmacSecret!);
      try {
        const r = await fetch(`${cfg.webhookUrl}/highlight`, {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json',
            Authorization: `Bearer ${cfg.bearer}`,
            'X-KOS-Signature': formatSignatureHeader(signed),
          },
          body,
        });
        if (!r.ok) console.warn('[KOS] webhook POST failed', r.status);
      } catch (e) {
        console.warn('[KOS] fetch threw', (e as Error).message);
      }
    });

    // ULID shim — Web Crypto random + Crockford base32 encode, 48-bit ts + 80-bit rand.
    function generateUlid(): string {
      const CROCKFORD = '0123456789ABCDEFGHJKMNPQRSTVWXYZ';
      const ts = Date.now();
      const tsBytes = new Uint8Array(6);
      for (let i = 5; i >= 0; i--) { tsBytes[i] = ts % 256; /* eslint-disable-next-line */ (Math.floor(ts / 256)); }
      // Simpler: produce 26-char ULID directly.
      let out = '';
      let rem = ts;
      for (let i = 9; i >= 0; i--) { out = CROCKFORD[rem % 32] + out; rem = Math.floor(rem / 32); }
      const rand = new Uint8Array(16);
      crypto.getRandomValues(rand);
      for (let i = 0; i < 16; i++) out += CROCKFORD[rand[i] % 32];
      return out.slice(0, 26);
    }
    ```

    **`apps/chrome-extension/src/options.ts`**:
    ```typescript
    import { loadConfig, saveConfig } from './lib/storage.js';
    import { signRequest, formatSignatureHeader } from './lib/hmac.js';

    async function init(): Promise<void> {
      const cfg = await loadConfig();
      (document.getElementById('webhookUrl') as HTMLInputElement).value = cfg.webhookUrl ?? '';
      (document.getElementById('bearer') as HTMLInputElement).value = cfg.bearer ?? '';
      (document.getElementById('hmacSecret') as HTMLInputElement).value = cfg.hmacSecret ?? '';
    }

    function showStatus(msg: string, ok: boolean): void {
      const el = document.getElementById('status')!;
      el.textContent = msg;
      el.style.background = ok ? '#e7f9e8' : '#fde7e7';
    }

    async function onSave(): Promise<void> {
      const webhookUrl = (document.getElementById('webhookUrl') as HTMLInputElement).value.trim();
      const bearer = (document.getElementById('bearer') as HTMLInputElement).value.trim();
      const hmacSecret = (document.getElementById('hmacSecret') as HTMLInputElement).value.trim();
      if (!webhookUrl || !bearer || !hmacSecret) {
        showStatus('All three fields are required', false);
        return;
      }
      try {
        new URL(webhookUrl);
      } catch {
        showStatus('Webhook URL is not a valid URL', false);
        return;
      }
      await saveConfig({ webhookUrl, bearer, hmacSecret });
      showStatus('Saved.', true);
    }

    async function onTest(): Promise<void> {
      const cfg = await loadConfig();
      if (!cfg.webhookUrl || !cfg.bearer || !cfg.hmacSecret) {
        showStatus('Save config first', false);
        return;
      }
      const body = JSON.stringify({ test_ping: true, at: new Date().toISOString() });
      const signed = await signRequest(body, cfg.hmacSecret!);
      try {
        const r = await fetch(`${cfg.webhookUrl}/highlight`, {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json',
            Authorization: `Bearer ${cfg.bearer}`,
            'X-KOS-Signature': formatSignatureHeader(signed),
          },
          body,
        });
        showStatus(`Webhook responded ${r.status} ${r.ok ? '(ok)' : '(failed)'}`, r.ok);
      } catch (e) {
        showStatus(`Fetch threw: ${(e as Error).message}`, false);
      }
    }

    document.getElementById('save')!.addEventListener('click', () => { void onSave(); });
    document.getElementById('test')!.addEventListener('click', () => { void onTest(); });
    void init();
    ```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-01-01 | Spoofing | Webpage JS forges chrome.runtime.sendMessage | mitigate | Content script isolation; chrome.runtime.onMessage validates sender.id === chrome.runtime.id; not exposed as postMessage target |
| T-05-01-02 | Tampering | Webpage JS reads window.chrome.storage | mitigate | chrome.storage is not exposed to page world; only extension context |
| T-05-01-03 | Information disclosure | Options page Bearer input visible in DOM | accept | type="password"; field value cleared from DOM on save? No — options page is tab-only, same trust boundary as Chrome |
| T-05-01-04 | Repudiation | User clicks "Send to KOS" accidentally | accept | No undo in extension; server-side capture is idempotent on capture_id; dashboard allows archive/delete |
| T-05-01-05 | DoS | Attacker spams context-menu clicks | accept | Server-side per-IP throttle (Plan 05-02); extension itself rate-limits at one click / user interaction |
| T-05-01-06 | Elevation | Malicious webpage manipulates window.getSelection to include attacker data | accept | User controls selection explicitly; worst case is polluted capture; entity-resolver flags weird content |
