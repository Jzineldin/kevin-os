    import { SecretsManagerClient, GetSecretValueCommand } from '@aws-sdk/client-secrets-manager';
    const sm = new SecretsManagerClient({ region: process.env.AWS_REGION ?? 'eu-north-1' });
    interface Cache { bearer: string; hmacSecret: string; }
    let cached: Cache | null = null;
    export async function getSecrets(): Promise<Cache> {
      if (cached) return cached;
      const bearerArn = process.env.BEARER_SECRET_ARN;
      const hmacArn = process.env.HMAC_SECRET_ARN;
      if (!bearerArn || !hmacArn) throw new Error('BEARER_SECRET_ARN or HMAC_SECRET_ARN missing');
      const [b, h] = await Promise.all([
        sm.send(new GetSecretValueCommand({ SecretId: bearerArn })),
        sm.send(new GetSecretValueCommand({ SecretId: hmacArn })),
      ]);
      const bearer = b.SecretString;
      const hmacSecret = h.SecretString;
      if (!bearer || bearer === 'PLACEHOLDER') throw new Error('chrome-extension-bearer PLACEHOLDER');
      if (!hmacSecret || hmacSecret === 'PLACEHOLDER') throw new Error('chrome-extension-hmac-secret PLACEHOLDER');
      cached = { bearer, hmacSecret };
      return cached;
    }
    export function __resetForTests(): void { cached = null; }

    **`services/chrome-webhook/src/handler.ts`**:
    ```typescript
    import { initSentry, wrapHandler } from '../../_shared/sentry.js';
    import { setupOtelTracingAsync, flush as langfuseFlush, tagTraceWithCaptureId } from '../../_shared/tracing.js';
    import { EventBridgeClient, PutEventsCommand } from '@aws-sdk/client-eventbridge';
    import { CaptureReceivedChromeHighlightSchema } from '@kos/contracts';
    import { getSecrets } from './secrets.js';
    import { createHmac, timingSafeEqual } from 'node:crypto';

    if (!process.env.AWS_REGION) process.env.AWS_REGION = 'eu-north-1';
    const eb = new EventBridgeClient({ region: 'eu-north-1' });
    const REPLAY_WINDOW_SEC = 300;

    interface FnUrlEvent {
      requestContext: { http: { method: string; path: string } };
      headers: Record<string, string>;
      body?: string;
      isBase64Encoded?: boolean;
    }

    export const handler = wrapHandler(async (event: FnUrlEvent) => {
      await initSentry();
      await setupOtelTracingAsync();
      try {
        const { http } = event.requestContext;
        if (http.method !== 'POST') return resp(405, { error: 'method' });
        if (http.path !== '/highlight') return resp(404, { error: 'not_found' });

        const body = readBody(event);
        if (!body) return resp(400, { error: 'empty_body' });

        const { bearer, hmacSecret } = await getSecrets();

        // Bearer
        const auth = event.headers['authorization'] ?? event.headers['Authorization'] ?? '';
        const m = auth.match(/^Bearer (.+)$/);
        if (!m) return resp(401, { error: 'missing_bearer' });
        const provided = Buffer.from(m[1]);
        const expected = Buffer.from(bearer);
        if (provided.length !== expected.length || !timingSafeEqual(provided, expected)) return resp(401, { error: 'bad_bearer' });

        // HMAC + replay
        const sigHeader = event.headers['x-kos-signature'] ?? event.headers['X-KOS-Signature'];
        if (!sigHeader) return resp(400, { error: 'missing_signature' });
        const sm = sigHeader.match(/^t=(\d+),v1=([0-9a-f]{64})$/);
        if (!sm) return resp(400, { error: 'bad_signature_format' });
        const ts = parseInt(sm[1], 10);
        const v1 = sm[2];
        const now = Math.floor(Date.now() / 1000);
        if (Math.abs(now - ts) > REPLAY_WINDOW_SEC) return resp(401, { error: 'replay_window' });
        const canonical = `${hmacSecret}.${ts}.${body}`;
        const calc = createHmac('sha256', hmacSecret).update(canonical).digest('hex');
        const calcBuf = Buffer.from(calc);
        const v1Buf = Buffer.from(v1);
        if (calcBuf.length !== v1Buf.length || !timingSafeEqual(calcBuf, v1Buf)) return resp(401, { error: 'bad_signature' });

        // Zod parse
        let payload: unknown;
        try { payload = JSON.parse(body); } catch { return resp(400, { error: 'invalid_json' }); }
        const parse = CaptureReceivedChromeHighlightSchema.safeParse({
          ...(payload as object),
          received_at: new Date().toISOString(),
        });
        if (!parse.success) return resp(400, { error: 'schema', issues: parse.error.issues });

        const detail = parse.data;
        tagTraceWithCaptureId(detail.capture_id);

        await eb.send(new PutEventsCommand({
          Entries: [{
            EventBusName: 'kos.capture',
            Source: 'kos.capture',
            DetailType: 'capture.received',
            Detail: JSON.stringify(detail),
          }],
        }));

        return resp(200, { capture_id: detail.capture_id });
      } finally {
        await langfuseFlush();
      }
    });

    function resp(statusCode: number, body: unknown): { statusCode: number; headers: Record<string, string>; body: string } {
      return { statusCode, headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(body) };
    }
    function readBody(event: FnUrlEvent): string | null {
      if (!event.body) return null;
      return event.isBase64Encoded ? Buffer.from(event.body, 'base64').toString('utf8') : event.body;
    }
    ```

    ```typescript
    import type { Construct } from 'constructs';
    import { Duration } from 'aws-cdk-lib';
    import { FunctionUrlAuthType, InvokeMode } from 'aws-cdk-lib/aws-lambda';
    import { Secret } from 'aws-cdk-lib/aws-secretsmanager';
    import type { IEventBus } from 'aws-cdk-lib/aws-events';
    import { KosLambda } from '../constructs/kos-lambda';
    import * as path from 'node:path';

    export interface WireChromeWebhookProps {
      captureBus: IEventBus;
      kevinOwnerId: string;
    }

    export function wireChromeWebhook(scope: Construct, props: WireChromeWebhookProps) {
      // 2 Secrets (operator seeds values post-deploy)
      const bearerSecret = new Secret(scope, 'ChromeExtensionBearer', {
        secretName: 'kos/chrome-extension-bearer',
        description: 'Bearer token for Chrome extension → chrome-webhook',
        generateSecretString: { passwordLength: 48, excludePunctuation: true },
      });
      const hmacSecret = new Secret(scope, 'ChromeExtensionHmacSecret', {
        secretName: 'kos/chrome-extension-hmac-secret',
        description: 'HMAC secret for Chrome extension → chrome-webhook',
        generateSecretString: { passwordLength: 64, excludePunctuation: true },
      });

      const fn = new KosLambda(scope, 'ChromeWebhook', {
        entry: path.join(__dirname, '..', '..', '..', '..', 'services/chrome-webhook/src/handler.ts'),
        memory: 256,
        timeout: Duration.seconds(10),
        environment: {
          BEARER_SECRET_ARN: bearerSecret.secretArn,
          HMAC_SECRET_ARN: hmacSecret.secretArn,
          KEVIN_OWNER_ID: props.kevinOwnerId,
        },
      });
      bearerSecret.grantRead(fn);
      hmacSecret.grantRead(fn);
      props.captureBus.grantPutEventsTo(fn);

      const url = fn.addFunctionUrl({ authType: FunctionUrlAuthType.NONE, invokeMode: InvokeMode.BUFFERED });
      return { fn, url, bearerSecret, hmacSecret };
    }
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Chrome extension → Function URL (public) | Bearer + HMAC-signed body + 5-min replay window |
| Lambda → Secrets Manager | IAM role scoped to 2 secret arns only |
| Lambda → kos.capture | PutEvents IAM; emits only capture.received |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-02-01 | Spoofing | Attacker POSTs to Function URL | mitigate | Bearer + HMAC + 5-min replay window; any fail = 401 |
| T-05-02-02 | Tampering | Attacker replays captured signed body | mitigate | 5-min window blocks most replays; same-body within window = duplicate capture_id, downstream idempotent |
| T-05-02-03 | Information disclosure | Lambda logs leak HMAC secret | mitigate | Secret loaded via Secrets Manager cache; never logged; Sentry env-scrub covers accidents |
| T-05-02-04 | DoS | Attacker spams Function URL | accept | Function URL has no throttle; Bearer+HMAC rejects 401 cheaply; cost per 401 ~$0.0000002; budget alarm from Phase 1 catches spike |
| T-05-02-05 | Repudiation | capture_id collision | accept | ULID collision probability negligible; downstream dedupes on capture_id |
| T-05-02-06 | Elevation | Lambda role abused to call other buses | mitigate | IAM scoped to captureBus.grantPutEventsTo only |
</threat_model>
