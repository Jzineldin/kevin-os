    **`apps/chrome-extension/src/content-linkedin.ts`**:
    ```typescript
    import { signRequest, formatSignatureHeader } from './lib/hmac.js';
    import { loadConfig, isConfigured } from './lib/storage.js';

    const POLL_INTERVAL_MS = 30 * 60 * 1000;
    const BACKOFF_24H_MS = 24 * 60 * 60 * 1000;

    interface StorageState {
      linkedin_last_poll_at?: number;
      linkedin_disabled_until?: number;
      linkedin_last_event_urn?: string;
    }

    async function getState(): Promise<StorageState> {
      return (await chrome.storage.local.get([
        'linkedin_last_poll_at', 'linkedin_disabled_until', 'linkedin_last_event_urn',
      ])) as StorageState;
    }

    function sleep(ms: number): Promise<void> { return new Promise(r => setTimeout(r, ms)); }
    function jitter(minMs: number, maxMs: number): number { return minMs + Math.floor(Math.random() * (maxMs - minMs)); }

    chrome.runtime.onMessage.addListener((msg, _sender, sendResponse) => {
      if (msg?.type === 'linkedin-poll-tick') {
        void poll().finally(() => sendResponse({ ok: true }));
        return true;  // async response
      }
      return false;
    });

    async function poll(): Promise<void> {
      if (document.visibilityState !== 'visible') { console.debug('[KOS-LI] tab hidden, skip'); return; }
      if (!(await isConfigured())) { console.debug('[KOS-LI] not configured, skip'); return; }
      const state = await getState();
      const now = Date.now();
      if (state.linkedin_disabled_until && state.linkedin_disabled_until > now) {
        console.debug('[KOS-LI] disabled until', new Date(state.linkedin_disabled_until).toISOString());
        return;
      }
      if (state.linkedin_last_poll_at && now - state.linkedin_last_poll_at < POLL_INTERVAL_MS) {
        console.debug('[KOS-LI] last poll too recent');
        return;
      }
      await chrome.storage.local.set({ linkedin_last_poll_at: now });

      try {
        await sleep(jitter(2000, 15000));
        const convRes = await fetch('https://www.linkedin.com/voyager/api/messaging/conversations?count=20&keyVersion=LEGACY_INBOX', {
          credentials: 'include',
          headers: { 'x-restli-protocol-version': '2.0.0', accept: 'application/json' },
        });
        if (convRes.status === 401 || convRes.status === 403) {
          await disableFor24h(convRes.status);
          return;
        }
        if (!convRes.ok) {
          console.warn('[KOS-LI] conversations status', convRes.status);
          return;
        }
        const convData = await convRes.json();
        const elements = Array.isArray(convData?.elements) ? convData.elements : [];

        let newestUrn: string | undefined = state.linkedin_last_event_urn;

        for (const conv of elements) {
          await sleep(jitter(2000, 15000));
          const convUrn: string | undefined = conv?.entityUrn;
          if (!convUrn) continue;
          const eventsUrl = `https://www.linkedin.com/voyager/api/messaging/conversations/${encodeURIComponent(convUrn)}/events?count=20`;
          const evRes = await fetch(eventsUrl, {
            credentials: 'include',
            headers: { 'x-restli-protocol-version': '2.0.0', accept: 'application/json' },
          });
          if (evRes.status === 401 || evRes.status === 403) { await disableFor24h(evRes.status); return; }
          if (!evRes.ok) continue;
          const evData = await evRes.json();
          const eventElements = Array.isArray(evData?.elements) ? evData.elements : [];
          for (const ev of eventElements) {
            const evUrn: string | undefined = ev?.entityUrn;
            if (!evUrn) continue;
            if (state.linkedin_last_event_urn && evUrn <= state.linkedin_last_event_urn) continue;  // simple monotonic skip; URN is time-ordered
            const msg = ev?.eventContent?.['com.linkedin.voyager.messaging.event.MessageEvent'];
            const body: string | undefined = msg?.body?.text ?? msg?.body;
            if (!body) continue;
            const from = ev?.from?.messagingMember?.miniProfile;
            const sender = from ? `${from.firstName ?? ''} ${from.lastName ?? ''}`.trim() : 'unknown';
            const sentAt = ev?.createdAt ? new Date(ev.createdAt).toISOString() : new Date().toISOString();
            await forwardToKos({
              capture_id: generateUlid(),
              channel: 'linkedin',
              kind: 'linkedin_dm',
              conversation_urn: convUrn,
              message_urn: evUrn,
              from: { name: sender, li_public_id: from?.publicIdentifier },
              body,
              sent_at: sentAt,
              received_at: new Date().toISOString(),
            });
            if (!newestUrn || evUrn > newestUrn) newestUrn = evUrn;
          }
        }
        if (newestUrn) await chrome.storage.local.set({ linkedin_last_event_urn: newestUrn });
      } catch (e) {
        console.warn('[KOS-LI] poll threw', (e as Error).message);
      }
    }

    async function disableFor24h(status: number): Promise<void> {
      const until = Date.now() + BACKOFF_24H_MS;
      await chrome.storage.local.set({ linkedin_disabled_until: until });
      await forwardAlert({
        source: 'linkedin',
        severity: status === 401 ? 'auth_fail' : 'auth_fail',
        message: `LinkedIn Voyager returned ${status}. Polling disabled until ${new Date(until).toISOString()}.`,
        raised_at: new Date().toISOString(),
      });
    }

    async function forwardToKos(detail: Record<string, unknown>): Promise<void> {
      const cfg = await loadConfig();
      if (!cfg.webhookUrl || !cfg.bearer || !cfg.hmacSecret) return;
      const body = JSON.stringify(detail);
      const signed = await signRequest(body, cfg.hmacSecret);
      await fetch(`${cfg.webhookUrl}/linkedin`, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          Authorization: `Bearer ${cfg.bearer}`,
          'X-KOS-Signature': formatSignatureHeader(signed),
        },
        body,
      });
    }

    async function forwardAlert(alert: Record<string, unknown>): Promise<void> {
      const cfg = await loadConfig();
      if (!cfg.webhookUrl || !cfg.bearer || !cfg.hmacSecret) return;
      const body = JSON.stringify(alert);
      const signed = await signRequest(body, cfg.hmacSecret);
      await fetch(`${cfg.webhookUrl}/linkedin/alert`, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          Authorization: `Bearer ${cfg.bearer}`,
          'X-KOS-Signature': formatSignatureHeader(signed),
        },
        body,
      });
    }

    function generateUlid(): string {
      // same impl as background.ts — duplicated intentionally to keep content script self-contained
      const CROCKFORD = '0123456789ABCDEFGHJKMNPQRSTVWXYZ';
      const ts = Date.now();
      let out = '';
      let rem = ts;
      for (let i = 9; i >= 0; i--) { out = CROCKFORD[rem % 32] + out; rem = Math.floor(rem / 32); }
      const rand = new Uint8Array(16);
      crypto.getRandomValues(rand);
      for (let i = 0; i < 16; i++) out += CROCKFORD[rand[i] % 32];
      return out.slice(0, 26);
    }

    console.debug('[KOS-LI] content script loaded at', new Date().toISOString());
    ```

    ```javascript
    #!/usr/bin/env node
    // Counts system_alerts source=linkedin severity IN ('auth_fail','unusual_activity') over last 14 days.
    // Writes evidence to .planning/phases/05-messaging-channels/05-07-LINKEDIN-14-DAY-evidence-template.md
    // Exit 0 if zero alerts, 1 otherwise.
    //
    // Usage: AWS_REGION=eu-north-1 DATABASE_URL=postgres://... node scripts/verify-linkedin-observation.mjs
    import pg from 'pg';

    const { Client } = pg;
    const DATABASE_URL = process.env.DATABASE_URL;
    if (!DATABASE_URL) { console.error('Set DATABASE_URL'); process.exit(2); }

    const client = new Client({ connectionString: DATABASE_URL });
    await client.connect();
    try {
      const r = await client.query(`
        SELECT COUNT(*)::int AS c, MAX(raised_at) AS latest
        FROM system_alerts
        WHERE source = 'linkedin'
          AND severity IN ('auth_fail','unusual_activity')
          AND raised_at > NOW() - INTERVAL '14 days'
      `);
      const count = r.rows[0].c;
      const latest = r.rows[0].latest;
      console.log(`[verify-linkedin] 14-day LinkedIn alert count: ${count}. Latest: ${latest ?? 'none'}`);
      if (count === 0) { console.log('PASS — zero alerts in 14-day window'); process.exit(0); }
      else { console.log('FAIL — alerts present; see system_alerts table'); process.exit(1); }
    } finally { await client.end(); }
    ```
