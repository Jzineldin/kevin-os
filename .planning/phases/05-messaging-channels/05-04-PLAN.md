    // Read-only Baileys wrapper. Rejects 7 write methods at call-site + logs EMF metric.
    // Rationale: defense-in-depth layer (a) of 5. Even if upstream baileys-api does something
    // unexpected, our wrapper throws before the library touches the WebSocket.

    import makeWASocket from '@whiskeysockets/baileys';  // or the fazer-ai adapter re-export
    import type { AuthenticationState } from '@whiskeysockets/baileys';

    const BANNED_METHODS = [
      'sendMessage',
      'updateStatus',
      'groupCreate',
      'groupParticipantsUpdate',
      'chatModify',
      'readMessages',
      'sendPresenceUpdate',
    ] as const;
    type Banned = typeof BANNED_METHODS[number];

    export class BaileysWriteRejectedError extends Error {
      constructor(method: string) { super(`BAILEYS_WRITE_REJECTED: ${method}`); this.name = 'BaileysWriteRejectedError'; }
    }

    function emitWriteAttemptMetric(method: string): void {
      // CloudWatch Embedded Metric Format — one log line = one metric emit.
      const payload = {
        _aws: {
          Timestamp: Date.now(),
          CloudWatchMetrics: [{
            Namespace: 'KOS/Baileys',
            Dimensions: [['Method']],
            Metrics: [{ Name: 'whatsapp_write_calls_total', Unit: 'Count' }],
          }],
        },
        Method: method,
        whatsapp_write_calls_total: 1,
        log_line: 'BAILEYS_WRITE_REJECTED',
      };
      console.log(JSON.stringify(payload));
    }

    export function wrapReadOnly<S extends Record<string, any>>(sock: S): S {
      return new Proxy(sock, {
        get(target, prop: string, receiver) {
          if ((BANNED_METHODS as readonly string[]).includes(prop)) {
            return (..._args: unknown[]) => {
              emitWriteAttemptMetric(prop);
              throw new BaileysWriteRejectedError(prop);
            };
          }
          return Reflect.get(target, prop, receiver);
        },
      });
    }

    export interface ReadOnlyBaileysOptions {
      auth: AuthenticationState;
      printQRInTerminal?: boolean;
      browser?: [string, string, string];
    }

    export function createReadOnlyBaileys(options: ReadOnlyBaileysOptions): ReturnType<typeof makeWASocket> {
      const sock = makeWASocket({
        auth: options.auth,
        printQRInTerminal: options.printQRInTerminal ?? true,
        browser: options.browser ?? ['Kevin OS', 'Chrome', '130'],
      });
      return wrapReadOnly(sock) as typeof sock;
    }

    **`services/baileys-fargate/src/use-postgres-auth-state.ts`**:
    ```typescript
    // Baileys AuthenticationState + SignalKeyStore backed by RDS whatsapp_session_keys.
    // Pattern: one row per (owner_id, key_id). key_id encodes the signal key type + identifier
    // (e.g., 'creds', 'pre-key-1', 'session-46700000000@s.whatsapp.net').
    // See 05-CONTEXT.md D-06, 05-RESEARCH.md §3.

    import type { Pool } from 'pg';
    import type { AuthenticationState, AuthenticationCreds, SignalKeyStore, SignalDataTypeMap } from '@whiskeysockets/baileys';
    import { initAuthCreds } from '@whiskeysockets/baileys';

    function encodeKeyId(type: string, id: string): string { return `${type}:${id}`; }

    export async function usePostgresAuthState(pool: Pool, ownerId: string): Promise<{ state: AuthenticationState; saveCreds: () => Promise<void>; }> {
      // Load creds row (or initialise)
      async function loadCreds(): Promise<AuthenticationCreds> {
        const r = await pool.query<{ value_jsonb: AuthenticationCreds }>(
          `SELECT value_jsonb FROM whatsapp_session_keys WHERE owner_id = $1 AND key_id = 'creds'`,
          [ownerId],
        );
        if (r.rowCount === 0) return initAuthCreds();
        return r.rows[0].value_jsonb;
      }

      let creds = await loadCreds();

      async function saveCreds(): Promise<void> {
        await pool.query('BEGIN');
        try {
          await pool.query('SELECT pg_advisory_xact_lock(hashtext($1))', [ownerId]);
          await pool.query(
            `INSERT INTO whatsapp_session_keys (owner_id, key_id, key_type, value_jsonb, updated_at)
             VALUES ($1, 'creds', 'creds', $2::jsonb, NOW())
             ON CONFLICT (owner_id, key_id) DO UPDATE SET value_jsonb = EXCLUDED.value_jsonb, updated_at = NOW()`,
            [ownerId, JSON.stringify(creds)],
          );
          await pool.query('COMMIT');
        } catch (e) { await pool.query('ROLLBACK'); throw e; }
      }

      const keys: SignalKeyStore = {
        async get(type, ids) {
          if (ids.length === 0) return {};
          const encoded = ids.map(id => encodeKeyId(type, id));
          const r = await pool.query<{ key_id: string; value_jsonb: unknown }>(
            `SELECT key_id, value_jsonb FROM whatsapp_session_keys WHERE owner_id = $1 AND key_id = ANY($2::text[])`,
            [ownerId, encoded],
          );
          const out: Record<string, any> = {};
          for (const row of r.rows) {
            const id = row.key_id.slice(type.length + 1);  // strip "<type>:" prefix
            out[id] = row.value_jsonb;
          }
          return out as SignalDataTypeMap[keyof SignalDataTypeMap];
        },
        async set(data) {
          const client = await pool.connect();
          try {
            await client.query('BEGIN');
            await client.query('SELECT pg_advisory_xact_lock(hashtext($1))', [ownerId]);
            for (const [type, entries] of Object.entries(data)) {
              for (const [id, value] of Object.entries(entries ?? {})) {
                const keyId = encodeKeyId(type, id);
                if (value === null || value === undefined) {
                  await client.query(
                    `DELETE FROM whatsapp_session_keys WHERE owner_id = $1 AND key_id = $2`,
                    [ownerId, keyId],
                  );
                } else {
                  await client.query(
                    `INSERT INTO whatsapp_session_keys (owner_id, key_id, key_type, value_jsonb, updated_at)
                     VALUES ($1, $2, $3, $4::jsonb, NOW())
                     ON CONFLICT (owner_id, key_id) DO UPDATE SET value_jsonb = EXCLUDED.value_jsonb, updated_at = NOW()`,
                    [ownerId, keyId, type, JSON.stringify(value)],
                  );
                }
              }
            }
            await client.query('COMMIT');
          } catch (e) { await client.query('ROLLBACK'); throw e; }
          finally { client.release(); }
        },
        async clear() {
          await pool.query(`DELETE FROM whatsapp_session_keys WHERE owner_id = $1`, [ownerId]);
        },
      };

      return { state: { creds, keys }, saveCreds };
    }
    ```

    **`services/baileys-fargate/src/entrypoint.ts`**:
    ```typescript
    // Boot sequence:
    // 1. Load DB creds from Secrets Manager → open pg Pool.
    // 2. Load pg-auth (pre-existing session OR fresh QR).
    // 3. createReadOnlyBaileys — wrapped socket.
    // 4. On event messages.upsert → POST to BAILEYS_WEBHOOK_URL with X-BAILEYS-Secret.
    // 5. On session rejection → exponential backoff 1h → 4h cap.
    // 6. On SIGTERM → graceful logout + keys saved + exit 0.

    import { Pool } from 'pg';
    import { SecretsManagerClient, GetSecretValueCommand } from '@aws-sdk/client-secrets-manager';
    import { DisconnectReason } from '@whiskeysockets/baileys';
    import { usePostgresAuthState } from './use-postgres-auth-state.js';
    import { createReadOnlyBaileys } from './wa-socket.js';

    const OWNER_ID = process.env.KEVIN_OWNER_ID!;
    const WEBHOOK_URL = process.env.BAILEYS_WEBHOOK_URL!;
    const WEBHOOK_SECRET_ARN = process.env.BAILEYS_WEBHOOK_SECRET_ARN!;
    const DATABASE_URL_ARN = process.env.DATABASE_URL_SECRET_ARN!;
    const BACKOFF_STEPS_MS = [1 * 3600_000, 2 * 3600_000, 4 * 3600_000, 4 * 3600_000];

    const sm = new SecretsManagerClient({ region: process.env.AWS_REGION ?? 'eu-north-1' });

    async function secret(arn: string): Promise<string> {
      const r = await sm.send(new GetSecretValueCommand({ SecretId: arn }));
      if (!r.SecretString) throw new Error(`secret missing: ${arn}`);
      return r.SecretString;
    }

    async function bootOnce(rejectionStep: number): Promise<number> {
      const webhookSecret = await secret(WEBHOOK_SECRET_ARN);
      const databaseUrl = await secret(DATABASE_URL_ARN);
      const pool = new Pool({ connectionString: databaseUrl, max: 4 });
      const { state, saveCreds } = await usePostgresAuthState(pool, OWNER_ID);
      const sock = createReadOnlyBaileys({ auth: state });

      sock.ev.on('creds.update', () => { void saveCreds(); });
      sock.ev.on('messages.upsert', async (evt) => {
        try {
          await fetch(WEBHOOK_URL, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json', 'X-BAILEYS-Secret': webhookSecret },
            body: JSON.stringify({ event: 'messages.upsert', data: evt }),
          });
        } catch (e) { console.warn('webhook POST failed', (e as Error).message); }
      });

      return new Promise<number>((resolve) => {
        sock.ev.on('connection.update', (upd) => {
          if (upd.connection === 'close') {
            const reason = (upd.lastDisconnect?.error as any)?.output?.statusCode;
            if (reason === DisconnectReason.loggedOut) {
              console.error('[baileys] logged out — full restart with fresh QR needed');
              resolve(rejectionStep + 1);
            } else {
              console.warn('[baileys] connection closed', reason);
              resolve(rejectionStep + 1);
            }
          }
        });
      });
    }

    let step = 0;
    while (true) {
      step = await bootOnce(step);
      const backoff = BACKOFF_STEPS_MS[Math.min(step - 1, BACKOFF_STEPS_MS.length - 1)] ?? 4 * 3600_000;
      console.log(`[baileys] backing off ${backoff / 3600_000}h before reconnect`);
      await new Promise(r => setTimeout(r, backoff));
    }
    ```
