    ```typescript
    import { S3Client, PutObjectCommand } from '@aws-sdk/client-s3';
    const s3 = new S3Client({ region: process.env.AWS_REGION ?? 'eu-north-1' });

    export async function putAudio(captureId: string, bytes: Uint8Array, mimeType: string): Promise<{ bucket: string; key: string; }> {
      const bucket = process.env.BLOBS_BUCKET;
      if (!bucket) throw new Error('BLOBS_BUCKET missing');
      const d = new Date();
      const yyyy = d.getUTCFullYear(); const mm = String(d.getUTCMonth() + 1).padStart(2, '0');
      const ext = mimeType.includes('opus') || mimeType.includes('ogg') ? 'ogg' : 'bin';
      const key = `audio/${yyyy}/${mm}/${captureId}.${ext}`;
      await s3.send(new PutObjectCommand({ Bucket: bucket, Key: key, Body: bytes, ContentType: mimeType }));
      return { bucket, key };
    }
    ```

    ```typescript
    import { initSentry, wrapHandler } from '../../_shared/sentry.js';
    import { setupOtelTracingAsync, flush as langfuseFlush, tagTraceWithCaptureId } from '../../_shared/tracing.js';
    import { EventBridgeClient, PutEventsCommand } from '@aws-sdk/client-eventbridge';
    import {
      CaptureReceivedWhatsappTextSchema,
      CaptureReceivedWhatsappVoiceSchema,
    } from '@kos/contracts';
    import { getWebhookSecret } from './secrets.js';
    import { putAudio } from './s3.js';
    import { timingSafeEqual, createHash } from 'node:crypto';

    if (!process.env.AWS_REGION) process.env.AWS_REGION = 'eu-north-1';
    const eb = new EventBridgeClient({ region: 'eu-north-1' });

    // Deterministic ULID-like id from (chat_jid, message_key_id).
    function deterministicCaptureId(chatJid: string, messageKeyId: string): string {
      const CROCKFORD = '0123456789ABCDEFGHJKMNPQRSTVWXYZ';
      const hash = createHash('sha256').update(`${chatJid}|${messageKeyId}`).digest();
      let out = ''; for (let i = 0; i < 26; i++) out += CROCKFORD[hash[i] % 32];
      return out;
    }

    interface FnUrlEvent { requestContext: { http: { method: string } }; headers: Record<string, string>; body?: string; isBase64Encoded?: boolean; }

    export const handler = wrapHandler(async (event: FnUrlEvent) => {
      await initSentry();
      await setupOtelTracingAsync();
      try {
        if (event.requestContext.http.method !== 'POST') return resp(405, { error: 'method' });
        const secret = await getWebhookSecret();
        const provided = event.headers['x-baileys-secret'] ?? event.headers['X-BAILEYS-Secret'];
        if (!provided) return resp(401, { error: 'missing_secret' });
        const a = Buffer.from(secret); const b = Buffer.from(provided);
        if (a.length !== b.length || !timingSafeEqual(a, b)) return resp(401, { error: 'bad_secret' });

        if (!event.body) return resp(400, { error: 'empty_body' });
        const raw = event.isBase64Encoded ? Buffer.from(event.body, 'base64').toString('utf8') : event.body;
        const payload = JSON.parse(raw);

        if (payload.event !== 'messages.upsert') return resp(200, { skipped: payload.event });
        const messages = payload.data?.messages ?? [];
        const results: string[] = [];

        for (const m of messages) {
          if (m.key?.fromMe) continue;  // skip Kevin's own outbound (shouldn't exist — read-only)
          const chatJid: string | undefined = m.key?.remoteJid;
          const keyId: string | undefined = m.key?.id;
          if (!chatJid || !keyId) continue;
          const captureId = deterministicCaptureId(chatJid, keyId);
          tagTraceWithCaptureId(captureId);
          const isGroup = chatJid.endsWith('@g.us');
          const sentAt = m.messageTimestamp ? new Date(Number(m.messageTimestamp) * 1000).toISOString() : new Date().toISOString();
          const pushName: string | undefined = m.pushName;

          const text = m.message?.conversation ?? m.message?.extendedTextMessage?.text;
          const audio = m.message?.audioMessage;

          if (text) {
            const detail = CaptureReceivedWhatsappTextSchema.parse({
              capture_id: captureId,
              channel: 'whatsapp',
              kind: 'whatsapp_text',
              jid: chatJid,
              chat_jid: chatJid,
              from_name: pushName,
              body: text,
              is_group: isGroup,
              sent_at: sentAt,
              received_at: new Date().toISOString(),
            });
            await eb.send(new PutEventsCommand({
              Entries: [{ EventBusName: 'kos.capture', Source: 'kos.capture', DetailType: 'capture.received', Detail: JSON.stringify(detail) }],
            }));
            results.push(captureId);
          } else if (audio) {
            // Audio: the Baileys container side exposes an endpoint to fetch the decrypted bytes
            // (fazer-ai/baileys-api: GET /media/{messageId}) — configured via MEDIA_BASE_URL env.
            const mediaBase = process.env.BAILEYS_MEDIA_BASE_URL;
            if (!mediaBase) { console.warn('no BAILEYS_MEDIA_BASE_URL; skipping voice'); continue; }
            const r = await fetch(`${mediaBase}/${encodeURIComponent(keyId)}`, {
              headers: { 'X-BAILEYS-Secret': secret },  // same secret, bidirectional trust
            });
            if (!r.ok) { console.warn('media fetch failed', r.status); continue; }
            const bytes = new Uint8Array(await r.arrayBuffer());
            const mime: string = audio.mimetype ?? 'audio/ogg; codecs=opus';
            const { bucket, key } = await putAudio(captureId, bytes, mime);
            const duration: number = Number(audio.seconds ?? 0);
            const detail = CaptureReceivedWhatsappVoiceSchema.parse({
              capture_id: captureId,
              channel: 'whatsapp',
              kind: 'whatsapp_voice',
              jid: chatJid,
              chat_jid: chatJid,
              from_name: pushName,
              raw_ref: { s3_bucket: bucket, s3_key: key, duration_sec: duration, mime_type: mime },
              is_group: isGroup,
              sent_at: sentAt,
              received_at: new Date().toISOString(),
            });
            await eb.send(new PutEventsCommand({
              Entries: [{ EventBusName: 'kos.capture', Source: 'kos.capture', DetailType: 'capture.received', Detail: JSON.stringify(detail) }],
            }));
            results.push(captureId);
          }
        }

        return resp(200, { capture_ids: results });
      } finally {
        await langfuseFlush();
      }
    });

    function resp(statusCode: number, body: unknown) { return { statusCode, headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(body) }; }
    ```

    ```typescript
    import type { Construct } from 'constructs';
    import { Duration } from 'aws-cdk-lib';
    import { FunctionUrlAuthType, InvokeMode } from 'aws-cdk-lib/aws-lambda';
    import type { ISecret } from 'aws-cdk-lib/aws-secretsmanager';
    import type { IBucket } from 'aws-cdk-lib/aws-s3';
    import type { IEventBus } from 'aws-cdk-lib/aws-events';
    import { KosLambda } from '../constructs/kos-lambda';
    import * as path from 'node:path';

    export interface WireBaileysSidecarProps {
      captureBus: IEventBus;
      blobsBucket: IBucket;
      baileysWebhookSecret: ISecret;   // from Plan 05-04's wireBaileys()
      kevinOwnerId: string;
      baileysMediaBaseUrl: string;     // e.g., http://baileys.kos-internal.local:3025/media (from Plan 05-04 if service discovery added)
    }

    export function wireBaileysSidecar(scope: Construct, props: WireBaileysSidecarProps) {
      const fn = new KosLambda(scope, 'BaileysSidecar', {
        entry: path.join(__dirname, '..', '..', '..', '..', 'services/baileys-sidecar/src/handler.ts'),
        memory: 512,
        timeout: Duration.seconds(30),
        environment: {
          BAILEYS_WEBHOOK_SECRET_ARN: props.baileysWebhookSecret.secretArn,
          BLOBS_BUCKET: props.blobsBucket.bucketName,
          BAILEYS_MEDIA_BASE_URL: props.baileysMediaBaseUrl,
          KEVIN_OWNER_ID: props.kevinOwnerId,
        },
      });
      props.baileysWebhookSecret.grantRead(fn);
      props.blobsBucket.grantPut(fn, 'audio/*');  // scoped prefix
      props.captureBus.grantPutEventsTo(fn);

      const url = fn.addFunctionUrl({ authType: FunctionUrlAuthType.NONE, invokeMode: InvokeMode.BUFFERED });
      return { fn, url };
    }
    ```
