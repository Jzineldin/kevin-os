import { z } from 'zod';

const UlidRegex = /^[0-9A-HJKMNP-TV-Z]{26}$/;

// kos.capture / transcript.available — emitted by services/granola-poller (Plan 06-01).
// Consumed by services/transcript-extractor (Plan 06-02). capture_id = transcript_id (Notion page id).

  capture_id: z.string().regex(UlidRegex).or(z.string().length(36)), // ULID OR Notion page id (UUID-like)
  transcript_id: z.string(),
  title: z.string().min(1).max(500),
  transcript_text: z.string().min(1).max(64000), // 64 KB cap per D-04
  recorded_at: z.string().datetime(),
  attendees: z.array(z.string()).optional(),
  notion_url: z.string().url(),
  detected_at: z.string().datetime(),
});

export const ContextFullDossierRequestedSchema = z.object({
  capture_id: z.string().regex(UlidRegex),

  semanticChunks: z.array(SearchHitSchema),
  linkedProjects: z.array(z.object({
    id: z.string().uuid(),
