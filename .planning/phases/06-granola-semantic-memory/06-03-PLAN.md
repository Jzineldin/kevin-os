    import { SecretsManagerClient, GetSecretValueCommand } from '@aws-sdk/client-secrets-manager';

    let cached: { endpoint: string; adminKey: string } | null = null;
    export async function getAzureCreds(): Promise<{ endpoint: string; adminKey: string }> {
      if (cached) return cached;
      const arn = process.env.AZURE_SEARCH_SECRET_ARN;
      if (!arn) throw new Error('AZURE_SEARCH_SECRET_ARN not set');
      const sm = new SecretsManagerClient({ region: process.env.AWS_REGION ?? 'eu-north-1' });
      const r = await sm.send(new GetSecretValueCommand({ SecretId: arn }));
      const { endpoint, adminKey } = JSON.parse(r.SecretString!);
      cached = { endpoint, adminKey };
      return cached;
    }
    export const API_VERSION = '2025-09-01';
    export async function azureFetch(path: string, init: RequestInit & { body?: string }): Promise<Response> {
      const { endpoint, adminKey } = await getAzureCreds();
      const url = `${endpoint}${path}${path.includes('?') ? '&' : '?'}api-version=${API_VERSION}`;
      const headers = { 'Content-Type': 'application/json', 'api-key': adminKey, ...(init.headers as Record<string, string> | undefined) };
      return fetch(url, { ...init, headers });
    }

    **`packages/azure-search/src/query.ts`** — hybrid query (RESEARCH §9):
    ```ts
    import { ulid } from 'ulid';
    import { embedBatch } from '@kos/resolver';
    import { INDEX_NAME, SEMANTIC_CONFIG_NAME, VECTOR_FIELD } from './schema.js';
    import { azureFetch } from './client.js';
    import type { SearchHit } from '@kos/contracts';

    export interface HybridQueryInput {
      query: string;
      ownerId: string;
      entityIds?: string[];
      sources?: ('entity' | 'project' | 'transcript' | 'daily_brief')[];
      topK?: number;
      semanticRerank?: boolean;
    }
    export interface HybridQueryResult { results: SearchHit[]; elapsedMs: number; queryId: string; azureRequestId?: string; }

    function buildFilter(ownerId: string, entityIds?: string[], sources?: string[]): string {
      const clauses: string[] = [`owner_id eq '${ownerId.replace(/'/g, "''")}'`];
      if (sources?.length) clauses.push('(' + sources.map(s => `source eq '${s}'`).join(' or ') + ')');
      if (entityIds?.length) clauses.push('(' + entityIds.map(id => `entity_ids/any(e: e eq '${id.replace(/'/g, "''")}')`).join(' or ') + ')');
      return clauses.join(' and ');
    }

    export async function hybridQuery(input: HybridQueryInput): Promise<HybridQueryResult> {
      const queryId = ulid();
      const t0 = Date.now();
      const topK = input.topK ?? 10;
      const semantic = input.semanticRerank ?? true;

      // Cohere v4 query embedding via @kos/resolver
      const [vector] = await embedBatch([input.query], 'search_query');
      if (!vector) throw new Error('embedBatch returned empty vector');

      const body = {
        search: input.query,
        vectorQueries: [{ kind: 'vector', vector, k: 50, fields: VECTOR_FIELD }],
        filter: buildFilter(input.ownerId, input.entityIds, input.sources),
        ...(semantic ? { queryType: 'semantic', semanticConfiguration: SEMANTIC_CONFIG_NAME, captions: 'extractive', answers: 'none' } : {}),
        top: topK,
      };
      const resp = await azureFetch(`/indexes/${INDEX_NAME}/docs/search`, { method: 'POST', body: JSON.stringify(body) });
      const azureRequestId = resp.headers.get('x-request-id') ?? undefined;
      if (!resp.ok) throw new Error(`Azure hybridQuery ${resp.status}: ${await resp.text()}`);
      const json = await resp.json() as { value: Array<Record<string, unknown>> };

      const results: SearchHit[] = json.value.map((doc) => ({
        id: String(doc[FIELDS_ID] ?? doc.id),
        source: (doc.source ?? 'entity') as SearchHit['source'],
        content: String(doc.content ?? ''),
        score: Number(doc['@search.score'] ?? 0),
        rerankerScore: doc['@search.rerankerScore'] != null ? Number(doc['@search.rerankerScore']) : undefined,
        entityIds: (doc.entity_ids as string[] | undefined) ?? undefined,
        occurredAt: doc.occurred_at as string | undefined,
      }));
      return { results, elapsedMs: Date.now() - t0, queryId, azureRequestId };
    }
    const FIELDS_ID = 'id';
    ```

    **`packages/azure-search/src/upsert.ts`** — bulk upsert with batching (Azure limit: 1000 docs per batch):
    ```ts
    import { azureFetch } from './client.js';
    import { INDEX_NAME } from './schema.js';

    export interface AzureDocument { /* per <interfaces> */ }
    export interface UpsertResult { uploaded: number; failed: number; errors: { key: string; message: string }[]; }

    export async function upsertDocuments(docs: AzureDocument[]): Promise<UpsertResult> {
      const result: UpsertResult = { uploaded: 0, failed: 0, errors: [] };
      const batchSize = 500; // safe under Azure 1000 ceiling
      for (let i = 0; i < docs.length; i += batchSize) {
        const batch = docs.slice(i, i + batchSize);
        const resp = await azureFetch(`/indexes/${INDEX_NAME}/docs/index`, { method: 'POST', body: JSON.stringify({ value: batch }) });
        if (!resp.ok) {
          result.failed += batch.length;
          result.errors.push({ key: batch[0].id, message: `HTTP ${resp.status}` });
          continue;
        }
        const j = await resp.json() as { value: Array<{ key: string; status: boolean; errorMessage?: string }> };
        for (const r of j.value) {
          if (r.status) result.uploaded++;
          else { result.failed++; result.errors.push({ key: r.key, message: r.errorMessage ?? 'unknown' }); }
        }
      }
      return result;
    }
    ```
