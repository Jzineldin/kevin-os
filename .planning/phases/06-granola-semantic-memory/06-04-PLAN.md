```typescript
// Response: { items: TimelineEvent[], elapsedMs: number }
export interface TimelineEvent {
  event_id: string;
  entity_id: string;
  source: string;
  context: string | null;
  occurred_at: string;     // ISO datetime
  capture_id: string | null;

}
```

SQL query (D-26 — MV ⋃ live overlay with deduplication):
```sql
WITH mv AS (
  SELECT event_id, entity_id, source, context, occurred_at, capture_id
  FROM entity_timeline_mv
  WHERE entity_id = $1 AND owner_id = $2
  ORDER BY occurred_at DESC
  LIMIT 50
),
live AS (
  SELECT id AS event_id, entity_id, source, context, occurred_at, capture_id
  FROM mention_events
  WHERE entity_id = $1 AND owner_id = $2
    AND occurred_at > now() - interval '10 minutes'
    AND id NOT IN (SELECT event_id FROM mv)
)
SELECT *, false AS is_live_overlay FROM mv
UNION ALL
SELECT *, true AS is_live_overlay FROM live
ORDER BY occurred_at DESC
LIMIT 50;
```

    ```ts
    // apps/dashboard/src/app/api/entities/[id]/timeline/route.ts
    import { NextRequest, NextResponse } from 'next/server';
    import { getDashboardPool } from '@/lib/db';  // existing helper from Phase 3

    export const runtime = 'nodejs';
    export const dynamic = 'force-dynamic';

    interface TimelineRow {
      event_id: string;
      entity_id: string;
      source: string;
      context: string | null;
      occurred_at: string;
      capture_id: string | null;
      is_live_overlay: boolean;
    }

    export async function GET(
      _req: NextRequest,
      { params }: { params: Promise<{ id: string }> },
    ): Promise<NextResponse> {
      const { id } = await params;
      // Bearer auth handled by Phase 3 middleware; ownerId derived from session/secret
      const ownerId = process.env.KEVIN_OWNER_ID!;
      const pool = await getDashboardPool();
      const t0 = Date.now();

      const sql = `
        WITH mv AS (
          SELECT event_id, entity_id, source, context, occurred_at, capture_id
          FROM entity_timeline_mv
          WHERE entity_id = $1 AND owner_id = $2
          ORDER BY occurred_at DESC
          LIMIT 50
        ),
        live AS (
          SELECT id AS event_id, entity_id, source, context, occurred_at, capture_id
          FROM mention_events
          WHERE entity_id = $1 AND owner_id = $2
            AND occurred_at > now() - interval '10 minutes'
            AND id::text NOT IN (SELECT event_id::text FROM mv)
        )
        SELECT event_id, entity_id, source, context, occurred_at, capture_id, false AS is_live_overlay FROM mv
        UNION ALL
        SELECT event_id, entity_id, source, context, occurred_at, capture_id, true AS is_live_overlay FROM live
        ORDER BY occurred_at DESC
        LIMIT 50;
      `;
      const r = await pool.query(sql, [id, ownerId]);
      const items: TimelineRow[] = r.rows.map((row) => ({
        event_id: String(row.event_id),
        entity_id: String(row.entity_id),
        source: String(row.source),
        context: row.context,
        occurred_at: new Date(row.occurred_at).toISOString(),
        capture_id: row.capture_id ? String(row.capture_id) : null,
        is_live_overlay: Boolean(row.is_live_overlay),
      }));
      return NextResponse.json({ items, elapsedMs: Date.now() - t0 });
    }
    ```
