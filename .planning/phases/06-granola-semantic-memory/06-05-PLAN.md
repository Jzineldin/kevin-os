    import { createHash } from 'node:crypto';

    export interface CacheReadResult {
      entityId: string;
      bundle: { dossier_markdown: string; semantic_chunks: SearchHit[] };
      hashMatched: boolean;
    }
    export async function readDossierCache(pool: PgPool, entityIds: string[]): Promise<Map<string, CacheReadResult>>;

    export async function computeLastTouchHash(pool: PgPool, entityId: string): Promise<string> {
      const r = await pool.query(
        `SELECT ei.name, ei.last_touch, COALESCE(MAX(me.occurred_at)::text, '') AS max_mention
         FROM entity_index ei
         LEFT JOIN mention_events me ON me.entity_id = ei.id
         WHERE ei.id = $1
         GROUP BY ei.name, ei.last_touch`,
        [entityId]
      );
      const row = r.rows[0] ?? { name: '', last_touch: null, max_mention: '' };
      return createHash('sha256').update(`${row.name}|${row.last_touch ?? ''}|${row.max_mention}`).digest('hex');
    }

    **`packages/context-loader/src/loadContext.ts`** — main entry:
    ```ts
    import { hybridQuery } from '@kos/azure-search';
    import { loadKevinContextBlock } from './kevin.js';
    import { loadEntityDossiers, loadRecentMentions, loadLinkedProjects } from './dossiers.js';
    import { readDossierCache, computeLastTouchHash, writeDossierCache } from './cache.js';
    import { buildDossierMarkdown } from './markdown.js';
    import type { ContextBundle } from '@kos/contracts';

    export interface LoadContextInput { /* per <interfaces> */ }

    export async function loadContext(input: LoadContextInput): Promise<ContextBundle> {
      const t0 = Date.now();
      const pool = input.pool ?? await getPoolFn();   // getPoolFn imported from a context-loader-local pool.ts

      const hasEntities = input.entityIds.length > 0;

      // Parallel fetches (D-15 budget — 800ms p95 via Promise.all)
      const [kevinContext, dossiersFromDb, mentions, projects, semanticResult, cacheMap] = await Promise.all([
        loadKevinContextBlock(input.ownerId, pool),
        hasEntities ? loadEntityDossiers(pool, input.entityIds, input.ownerId) : Promise.resolve([]),
        hasEntities ? loadRecentMentions(pool, input.entityIds, input.ownerId, 20) : Promise.resolve([]),
        hasEntities ? loadLinkedProjects(pool, input.entityIds, input.ownerId) : Promise.resolve([]),
        hasEntities
          ? hybridQuery({ query: dossiersFromDb_queryText(input.entityIds, dossiersFromDb), ownerId: input.ownerId, entityIds: input.entityIds, topK: input.maxSemanticChunks ?? 10, semanticRerank: true })
          : input.rawText
            ? hybridQuery({ query: input.rawText, ownerId: input.ownerId, topK: 5, semanticRerank: true })
            : Promise.resolve({ results: [], elapsedMs: 0, queryId: 'na' }),
        hasEntities ? readDossierCache(pool, input.entityIds) : Promise.resolve(new Map()),
      ]);

      // For each entity: prefer cache hit when last_touch_hash matches; else recompute markdown.
      let cacheHit = false;
      const finalDossiers = dossiersFromDb;  // simplified — bundle assembly always uses fresh DB rows; cache is only used for the markdown block + semantic chunks
      const cacheHitsCount = cacheMap.size;
      cacheHit = cacheHitsCount > 0;

      // Assemble markdown — cache hits short-circuit re-build per entity:
      const assembledMarkdown = buildDossierMarkdown({
        entityDossiers: finalDossiers,
        recentMentions: mentions,
        semanticChunks: semanticResult.results,
        linkedProjects: projects,
      });

      // Fire-and-forget cache writes for misses (do NOT block return):
      void Promise.allSettled(
        finalDossiers
          .filter(d => !cacheMap.has(d.entity_id))
          .map(async d => {
            const hash = await computeLastTouchHash(pool, d.entity_id);
            await writeDossierCache(pool, {
              entityId: d.entity_id, ownerId: input.ownerId, lastTouchHash: hash,
              dossierMarkdown: assembledMarkdown,  // entity-scoped markdown ideally; v1 uses full assembled bundle
              semanticChunks: semanticResult.results.filter(h => h.entityIds?.includes(d.entity_id)),
              ttlHours: 1,
            });
          })
      );

      return {
        kevinContext,
        entityDossiers: finalDossiers,
        recentMentions: mentions,
        semanticChunks: semanticResult.results,
        linkedProjects: projects,
        assembledMarkdown,
        elapsedMs: Date.now() - t0,
        cacheHit,
      };
    }

    function dossiersFromDb_queryText(entityIds: string[], dossiers: EntityDossier[]): string {
      if (dossiers.length === 0) return entityIds.join(' ');
      return dossiers.map(d => `${d.name} ${d.role ?? ''} ${d.org ?? ''}`).join(' | ');
    }
    ```

               cached_at = now(),
               expires_at = EXCLUDED.expires_at,

        [args.entityId, args.ownerId, hash, args.dossierMarkdown]
