```typescript
import { z } from 'zod';

const UlidRegex = /^[0-9A-HJKMNP-TV-Z]{26}$/;

export const TopThreeItemSchema = z.object({
  title: z.string().min(1).max(200),
  entity_ids: z.array(z.string().uuid()).max(5),
  urgency: z.enum(['high', 'med', 'low']),
});

export const DroppedThreadSchema = z.object({
  title: z.string().min(1).max(200),
  entity_ids: z.array(z.string().uuid()).max(5),
  last_mentioned_at: z.string().datetime().optional(),
});

export const CalendarEventSchema = z.object({
  start: z.string().datetime(),
  end: z.string().datetime().optional(),
  title: z.string().min(1).max(200),
  attendees: z.array(z.string()).max(10).optional(),
});

export const DraftReadyListItemSchema = z.object({
  draft_id: z.string().uuid(),
  from: z.string().min(1).max(200),
  subject: z.string().min(1).max(300),
  classification: z.enum(['urgent', 'important']),
});

export const BriefCommonFieldsSchema = z.object({
  prose_summary: z.string().min(1).max(600),
  top_three: z.array(TopThreeItemSchema).min(0).max(3),
  dropped_threads: z.array(DroppedThreadSchema).max(5),
});

export const MorningBriefSchema = BriefCommonFieldsSchema.extend({
  calendar_today: z.array(CalendarEventSchema).max(20).default([]),
  calendar_tomorrow: z.array(CalendarEventSchema).max(20).default([]),
  drafts_ready: z.array(DraftReadyListItemSchema).max(10).default([]),
});

export const DayCloseBriefSchema = BriefCommonFieldsSchema.extend({
  slipped_items: z.array(z.object({
    title: z.string().min(1).max(200),
    entity_ids: z.array(z.string().uuid()).max(5),
    reason: z.string().max(200).optional(),
  })).max(5),
  recent_decisions: z.array(z.string().max(200)).max(5),
  active_threads_delta: z.array(z.object({
    thread: z.string().min(1).max(200),
    status: z.enum(['new', 'updated', 'closed']),
  })).max(10),
});

export const WeeklyReviewSchema = z.object({
  prose_summary: z.string().min(1).max(1000),
  week_recap: z.array(z.string().min(1).max(240)).max(10),
  next_week_candidates: z.array(z.object({
    title: z.string().min(1).max(200),
    why: z.string().min(1).max(200),
  })).max(7),
  active_threads_snapshot: z.array(z.object({
    thread: z.string().min(1).max(200),
    where: z.enum(['almi', 'speed', 'tale-forge', 'outbehaving', 'other']),
    status: z.string().max(200),
  })).max(20),
});

export type MorningBrief = z.infer<typeof MorningBriefSchema>;
export type DayCloseBrief = z.infer<typeof DayCloseBriefSchema>;
export type WeeklyReview = z.infer<typeof WeeklyReviewSchema>;

// Envelope wrapping brief output in an agent_runs-friendly shape:
export const BriefAgentRunOutputSchema = z.object({
  brief_kind: z.enum(['morning-brief', 'day-close', 'weekly-review']),
  brief_capture_id: z.string().regex(UlidRegex),
  rendered_at: z.string().datetime(),
  data: z.union([MorningBriefSchema, DayCloseBriefSchema, WeeklyReviewSchema]),
});
```

```sql
-- Phase 7 migration 0014 (bump to 0015 if Phase 4/6 collision at execution time)

CREATE TABLE IF NOT EXISTS "top3_membership" (
  "id"            uuid        PRIMARY KEY DEFAULT gen_random_uuid(),
  "owner_id"      uuid        NOT NULL,
  "brief_date"    date        NOT NULL,                 -- Stockholm calendar date
  "brief_kind"    text        NOT NULL CHECK (brief_kind IN ('morning-brief','day-close','weekly-review')),
  "brief_capture_id" text     NOT NULL,                 -- ULID of the brief run
  "entity_id"     uuid        NOT NULL REFERENCES "entity_index"("id") ON DELETE CASCADE,
  "top3_title"    text        NOT NULL,
  "urgency"       text        NOT NULL CHECK (urgency IN ('high','med','low')),
  "membered_at"   timestamptz NOT NULL DEFAULT now(),
  "acted_on_at"   timestamptz                           -- NULL until a matching mention_events arrives
);
CREATE INDEX IF NOT EXISTS "top3_membership_by_owner_date" ON "top3_membership" ("owner_id","brief_date");
CREATE INDEX IF NOT EXISTS "top3_membership_by_entity" ON "top3_membership" ("entity_id");
CREATE INDEX IF NOT EXISTS "top3_membership_pending" ON "top3_membership" ("owner_id","brief_date")
  WHERE "acted_on_at" IS NULL;

CREATE OR REPLACE VIEW "dropped_threads_v" AS
SELECT
  tm.owner_id,
  tm.brief_date AS membered_on,
  (CURRENT_DATE AT TIME ZONE 'Europe/Stockholm')::date AS detected_for_date,
  tm.entity_id,
  ei.name AS entity_name,
  tm.top3_title AS title,
  tm.urgency,
  MAX(me.occurred_at) AS last_mentioned_at,
  MAX(tm.acted_on_at) AS last_acted_on_at
FROM "top3_membership" tm
JOIN "entity_index" ei ON ei.id = tm.entity_id
LEFT JOIN "mention_events" me
  ON me.entity_id = tm.entity_id AND me.owner_id = tm.owner_id
WHERE tm.acted_on_at IS NULL
  AND tm.brief_date >= (CURRENT_DATE AT TIME ZONE 'Europe/Stockholm' - INTERVAL '7 days')::date
GROUP BY tm.owner_id, tm.brief_date, tm.entity_id, ei.name, tm.top3_title, tm.urgency
HAVING MAX(me.occurred_at) IS NULL
    OR MAX(me.occurred_at) < (NOW() - INTERVAL '24 hours');

-- Trigger: mark top3_membership.acted_on_at when a mention_events row arrives
CREATE OR REPLACE FUNCTION mark_top3_acted_on() RETURNS trigger AS $$
BEGIN
  IF NEW.entity_id IS NULL THEN
    RETURN NEW;
  END IF;
  UPDATE "top3_membership"
     SET "acted_on_at" = NOW()
   WHERE "owner_id" = NEW.owner_id
     AND "entity_id" = NEW.entity_id
     AND "acted_on_at" IS NULL
     AND "membered_at" < NEW.occurred_at
     AND "brief_date" >= (NEW.occurred_at AT TIME ZONE 'Europe/Stockholm')::date - INTERVAL '7 days';
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS trg_mark_top3_acted_on ON "mention_events";
CREATE TRIGGER trg_mark_top3_acted_on
  AFTER INSERT ON "mention_events"
  FOR EACH ROW EXECUTE FUNCTION mark_top3_acted_on();
```
