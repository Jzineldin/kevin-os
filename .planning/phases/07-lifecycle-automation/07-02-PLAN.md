    ```typescript
    export async function replaceActiveThreadsSection(kevinContextPageId: string, snapshot: WeeklyReview['active_threads_snapshot']): Promise<void> {
      const notion = await getNotionClient();
      // 1. List children
      const existing = await notion.blocks.children.list({ block_id: kevinContextPageId, page_size: 100 });
      // 2. Find blocks BETWEEN "Active threads" heading_2 and the next heading_2 (exclusive). Archive them.
      let inSection = false;
      const toArchive: string[] = [];
      for (const b of existing.results) {
        const blk = b as any;
        if (blk.type === 'heading_2') {
          const text = (blk.heading_2?.rich_text?.[0]?.plain_text ?? '').toLowerCase();
          if (text.startsWith('active threads')) {
            inSection = true;
            toArchive.push(blk.id);   // archive the heading too — we'll re-create
            continue;
          }
          if (inSection) break;       // hit next heading_2 — stop
        }
        if (inSection) toArchive.push(blk.id);
      }
      const sema = new Semaphore(3);
      await Promise.all(toArchive.map(id => sema.acquire().then(async release => {
        try { await notion.blocks.update({ block_id: id, archived: true } as any); } finally { release(); }
      })));
      // 3. Append new section at end
      const newBlocks: NotionBlock[] = [
        { object: 'block', type: 'heading_2', heading_2: { rich_text: [{ type: 'text', text: { content: 'Active threads' } }] } },
        ...snapshot.map(t => ({
          object: 'block' as const, type: 'bulleted_list_item' as const,
          bulleted_list_item: { rich_text: [{ type: 'text' as const, text: { content: `[${t.where}] ${t.thread} — ${t.status}` } }] },
        })),
      ];
      await notion.blocks.children.append({ block_id: kevinContextPageId, children: newBlocks });
    }
    ```
