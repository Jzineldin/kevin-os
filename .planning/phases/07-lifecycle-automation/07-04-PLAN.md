<!-- SQL patterns -->
```sql
-- Cap snapshot per day (from agent_runs)
SELECT date_trunc('day', (started_at AT TIME ZONE 'Europe/Stockholm'))::date AS stockholm_date,
       count(*) FILTER (WHERE agent_name = 'push-telegram' AND status = 'ok') AS push_ok_count
FROM agent_runs
WHERE owner_id = $1
  AND started_at >= now() - interval '14 days'
GROUP BY 1
ORDER BY 1 DESC;

-- Quiet-hours violations
SELECT started_at, capture_id,
       EXTRACT(HOUR FROM (started_at AT TIME ZONE 'Europe/Stockholm'))::int AS stockholm_hour
FROM agent_runs
WHERE owner_id = $1
  AND agent_name = 'push-telegram'
  AND status = 'ok'
  AND started_at >= now() - interval '14 days'
  AND (EXTRACT(HOUR FROM (started_at AT TIME ZONE 'Europe/Stockholm'))::int >= 20
    OR EXTRACT(HOUR FROM (started_at AT TIME ZONE 'Europe/Stockholm'))::int < 8);
```

    #!/usr/bin/env node
    import pg from 'pg';
    import { DynamoDBClient } from '@aws-sdk/client-dynamodb';
    import { DynamoDBDocumentClient, GetCommand } from '@aws-sdk/lib-dynamodb';

    const OWNER_ID = process.env.KEVIN_OWNER_ID ?? '9e4be978-cc7d-571b-98ec-a1e92373682c';
    const CAP_TABLE_NAME = process.env.CAP_TABLE_NAME;
    const DATABASE_URL = process.env.DATABASE_URL;

    if (!CAP_TABLE_NAME || !DATABASE_URL) {
      console.error('Set CAP_TABLE_NAME + DATABASE_URL (or RDS_PROXY_ENDPOINT + IAM token shell dance).');
      process.exit(2);
    }

    const pool = new pg.Pool({ connectionString: DATABASE_URL });
    const ddb = DynamoDBDocumentClient.from(new DynamoDBClient({}));

    const { rows } = await pool.query(`
      SELECT date_trunc('day', (started_at AT TIME ZONE 'Europe/Stockholm'))::date AS stockholm_date,
             count(*) FILTER (WHERE agent_name = 'push-telegram' AND status = 'ok') AS push_ok_count
        FROM agent_runs
       WHERE owner_id = $1
         AND started_at >= now() - interval '14 days'
       GROUP BY 1
       ORDER BY 1 DESC
    `, [OWNER_ID]);

    let violations = 0;
    console.log('Stockholm Date  | Pushes | DDB | Status');
    console.log('----------------+--------+-----+-------');
    for (const r of rows) {
      const date = r.stockholm_date.toISOString().slice(0, 10);
      const n = Number(r.push_ok_count);
      const dd = await ddb.send(new GetCommand({ TableName: CAP_TABLE_NAME, Key: { pk: `telegram-cap#${date}` } }));
      const ddCount = dd.Item?.count ?? null;
      const status = n > 3 ? 'VIOLATION' : 'ok';
      if (n > 3) violations++;
      console.log(`${date}  | ${String(n).padStart(6)} | ${String(ddCount ?? '—').padStart(3)} | ${status}`);
    }

    await pool.end();
    if (violations > 0) {
      console.error(`\n${violations} day(s) violated the 3-push cap.`);
      process.exit(1);
    }
    console.log('\n14-day cap invariant HOLDS.');
    process.exit(0);
    ```

    #!/usr/bin/env node
    // Full Phase 7 SC verifier. Chains 5 checks; prints a per-check PASS/FAIL table.
    import pg from 'pg';
    import { execSync } from 'node:child_process';

    const OWNER_ID = process.env.KEVIN_OWNER_ID ?? '9e4be978-cc7d-571b-98ec-a1e92373682c';
    const pool = new pg.Pool({ connectionString: process.env.DATABASE_URL });

    const results = [];

    // Check 1: morning-brief recorded in agent_runs in last 24h
    const mb = await pool.query(
      `SELECT count(*) AS n FROM agent_runs
        WHERE owner_id=$1 AND agent_name='morning-brief' AND status='ok'
          AND started_at > now() - interval '24 hours'`,
      [OWNER_ID],
    );
    results.push({ check: 'SC1 morning-brief ran in 24h', pass: Number(mb.rows[0].n) > 0, detail: `n=${mb.rows[0].n}` });

    // Check 2: day-close recorded
    const dc = await pool.query(
      `SELECT count(*) AS n FROM agent_runs
        WHERE owner_id=$1 AND agent_name='day-close' AND status='ok'
          AND started_at > now() - interval '30 hours'`,
      [OWNER_ID],
    );
    results.push({ check: 'SC2 day-close ran in 30h', pass: Number(dc.rows[0].n) > 0, detail: `n=${dc.rows[0].n}` });

    // Check 3: weekly-review recorded within last 8 days
    const wr = await pool.query(
      `SELECT count(*) AS n FROM agent_runs
        WHERE owner_id=$1 AND agent_name='weekly-review' AND status='ok'
          AND started_at > now() - interval '8 days'`,
      [OWNER_ID],
    );
    results.push({ check: 'SC2 weekly-review ran in 8d', pass: Number(wr.rows[0].n) > 0, detail: `n=${wr.rows[0].n}` });

    // Check 4: top3_membership rows exist for today (morning-brief wrote them)
    const t3 = await pool.query(
      `SELECT count(*) AS n FROM top3_membership
        WHERE owner_id=$1 AND brief_date = (CURRENT_DATE AT TIME ZONE 'Europe/Stockholm')::date`,
      [OWNER_ID],
    );
    results.push({ check: 'SC7 top3_membership today', pass: Number(t3.rows[0].n) > 0, detail: `n=${t3.rows[0].n}` });

    // Check 5+6: run other two verifiers via subprocess
    for (const [name, cmd] of [['SC5 cap invariant', 'node scripts/verify-notification-cap-14day.mjs'], ['SC6 quiet-hours invariant', 'node scripts/verify-quiet-hours-invariant.mjs']]) {
      try {
        execSync(cmd, { stdio: 'inherit' });
        results.push({ check: name, pass: true, detail: 'subprocess ok' });
      } catch (err) {
        results.push({ check: name, pass: false, detail: `subprocess exit ${(err).status ?? 'unknown'}` });
      }
    }

    await pool.end();
    console.log('\n=== Phase 7 E2E Verification ===');
    for (const r of results) {
      console.log(`  [${r.pass ? 'PASS' : 'FAIL'}] ${r.check}  (${r.detail})`);
    }
    const allPassed = results.every(r => r.pass);
    process.exit(allPassed ? 0 : 1);
    ```
