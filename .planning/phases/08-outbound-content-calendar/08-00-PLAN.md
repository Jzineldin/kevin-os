export const ContentPlatformEnum = z.enum(['instagram', 'linkedin', 'tiktok', 'reddit', 'newsletter']);
export type ContentPlatform = z.infer<typeof ContentPlatformEnum>;

  topic_id: z.string().regex(UlidRegex),

  submitted_at: z.string().datetime(),

CREATE TABLE content_drafts (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  owner_id uuid NOT NULL,
  topic_id text NOT NULL,                           -- ULID
  capture_id text NOT NULL,
  platform text NOT NULL CHECK (platform IN ('instagram','linkedin','tiktok','reddit','newsletter')),
  content text NOT NULL,
  media_urls jsonb DEFAULT '[]'::jsonb,
  status text NOT NULL DEFAULT 'draft' CHECK (status IN ('draft','edited','approved','skipped','scheduled','published','cancelled','failed')),
  schedule_time timestamptz,
  postiz_post_id text,                              -- filled by publisher on success
  created_at timestamptz NOT NULL DEFAULT now(),
  approved_at timestamptz,
  scheduled_at timestamptz,
  published_at timestamptz,
  CONSTRAINT content_drafts_topic_platform_uidx UNIQUE (topic_id, platform)
);
CREATE INDEX content_drafts_owner_status_idx ON content_drafts (owner_id, status, created_at DESC);
CREATE INDEX content_drafts_owner_topic_idx ON content_drafts (owner_id, topic_id);

CREATE TABLE content_publish_authorizations (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  owner_id uuid NOT NULL,
  draft_id uuid NOT NULL REFERENCES content_drafts(id) ON DELETE CASCADE,
  approved_by text NOT NULL DEFAULT 'kevin',
  approved_at timestamptz NOT NULL DEFAULT now(),
  schedule_time timestamptz,                        -- null = publish immediately
  consumed_at timestamptz,                          -- set by publisher after Postiz call
  publish_result jsonb
);
CREATE INDEX content_publish_authorizations_owner_draft_idx ON content_publish_authorizations (owner_id, draft_id);

CREATE TABLE pending_mutations (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  owner_id uuid NOT NULL,
  capture_id text NOT NULL,
  mutation_type text NOT NULL CHECK (mutation_type IN ('cancel_meeting','delete_task','archive_doc','cancel_content_draft','cancel_email_draft','reschedule_meeting')),
  target_kind text NOT NULL,
  target_id text NOT NULL,
  target_display text NOT NULL,
  confidence numeric(4,3) NOT NULL,
  reasoning text,
  alternatives jsonb DEFAULT '[]'::jsonb,
  status text NOT NULL DEFAULT 'proposed' CHECK (status IN ('proposed','approved','executed','skipped','failed')),
  proposed_at timestamptz NOT NULL DEFAULT now(),
  approved_at timestamptz,
  executed_at timestamptz,
  result text,                                       -- 'archived' | 'rescheduled' | 'no_op' | 'failed'
  error text
);
CREATE INDEX pending_mutations_owner_status_idx ON pending_mutations (owner_id, status, proposed_at DESC);
CREATE INDEX pending_mutations_capture_id_idx ON pending_mutations (capture_id);

CREATE TABLE document_versions (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  owner_id uuid NOT NULL,
  recipient_email text NOT NULL,
  doc_name text NOT NULL,
  sha256 text NOT NULL,
  s3_bucket text NOT NULL,
  s3_key text NOT NULL,
  version_n integer NOT NULL,
  parent_sha256 text,
  diff_summary text,
  sent_at timestamptz NOT NULL,
  capture_id text NOT NULL,
  CONSTRAINT document_versions_recipient_doc_sha_uidx UNIQUE (recipient_email, doc_name, sha256)
);
CREATE INDEX document_versions_recipient_doc_idx ON document_versions (recipient_email, doc_name, version_n DESC);
CREATE INDEX document_versions_owner_idx ON document_versions (owner_id, sent_at DESC);

CREATE TABLE calendar_events_cache (
  event_id text NOT NULL,
  account text NOT NULL CHECK (account IN ('kevin-elzarka','kevin-taleforge')),
  owner_id uuid NOT NULL,
  calendar_id text NOT NULL,
  summary text NOT NULL,
  description text,
  location text,
  start_utc timestamptz NOT NULL,
  end_utc timestamptz NOT NULL,
  timezone text NOT NULL DEFAULT 'Europe/Stockholm',
  attendees_json jsonb NOT NULL DEFAULT '[]'::jsonb,
  is_all_day boolean NOT NULL DEFAULT false,
  ignored_by_kevin boolean NOT NULL DEFAULT false,
  updated_at timestamptz NOT NULL,
  cached_at timestamptz NOT NULL DEFAULT now(),
  PRIMARY KEY (event_id, account)
);
CREATE INDEX calendar_events_cache_owner_window_idx ON calendar_events_cache (owner_id, start_utc) WHERE ignored_by_kevin = false;
CREATE INDEX calendar_events_cache_account_updated_idx ON calendar_events_cache (account, updated_at DESC);
