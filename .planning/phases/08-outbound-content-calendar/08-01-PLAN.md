    ```typescript
    import { SecretsManagerClient, GetSecretValueCommand } from '@aws-sdk/client-secrets-manager';

    const sm = new SecretsManagerClient({ region: process.env.AWS_REGION ?? 'eu-north-1' });

    interface TokenCache { accessToken: string; expiresAt: number; }
    const cache = new Map<string, TokenCache>();

    export async function getAccessToken(account: 'kevin-elzarka' | 'kevin-taleforge'): Promise<string> {
      const now = Date.now();
      const cached = cache.get(account);
      if (cached && cached.expiresAt > now + 60_000) return cached.accessToken;  // 60s safety margin

      const secretId = `kos/gcal-oauth-${account}`;
      const r = await sm.send(new GetSecretValueCommand({ SecretId: secretId }));
      if (!r.SecretString) throw new Error(`Secret ${secretId} missing — run scripts/bootstrap-gcal-oauth.mjs --account ${account}`);
      const creds: { client_id: string; client_secret: string; refresh_token: string } = JSON.parse(r.SecretString);

      const body = new URLSearchParams({
        grant_type: 'refresh_token',
        client_id: creds.client_id,
        client_secret: creds.client_secret,
        refresh_token: creds.refresh_token,
      });
      const resp = await fetch('https://oauth2.googleapis.com/token', {
        method: 'POST', headers: { 'Content-Type': 'application/x-www-form-urlencoded' }, body,
      });
      if (!resp.ok) {
        const errBody = await resp.text();
        if (resp.status === 400 && /invalid_grant/.test(errBody)) {
          throw new Error(`Google OAuth refresh token expired or revoked for ${account}. Re-run scripts/bootstrap-gcal-oauth.mjs --account ${account}.`);
        }
        throw new Error(`Google OAuth ${resp.status}: ${errBody}`);
      }
      const tok: { access_token: string; expires_in: number } = await resp.json();
      cache.set(account, { accessToken: tok.access_token, expiresAt: now + tok.expires_in * 1000 - 600_000 });  // refresh 10 min early
      return tok.access_token;
    }
    ```

    ```typescript
    export async function fetchEventsWindow(args: {
      accessToken: string; timeMinIso: string; timeMaxIso: string; calendarId?: string;
    }): Promise<GcalEvent[]> {
      const url = new URL(`https://www.googleapis.com/calendar/v3/calendars/${args.calendarId ?? 'primary'}/events`);
      url.searchParams.set('timeMin', args.timeMinIso);
      url.searchParams.set('timeMax', args.timeMaxIso);
      url.searchParams.set('singleEvents', 'true');
      url.searchParams.set('orderBy', 'startTime');
      url.searchParams.set('maxResults', '250');
      const r = await fetch(url, { headers: { Authorization: `Bearer ${args.accessToken}` } });
      if (r.status === 401) { const e: Error & { code?: string } = new Error('gcal auth stale'); e.code = 'auth_stale'; throw e; }
      if (!r.ok) throw new Error(`gcal events.list ${r.status}: ${await r.text()}`);
      const body: { items?: Array<any> } = await r.json();
      return (body.items ?? [])
        .filter(ev => ev.status !== 'cancelled')  // skip cancelled events upstream
        .map(ev => {
          const isAllDay = Boolean(ev.start?.date);
          const startUtc = isAllDay ? new Date(ev.start.date + 'T00:00:00Z').toISOString() : new Date(ev.start.dateTime).toISOString();
          const endUtc = isAllDay ? new Date(ev.end.date + 'T00:00:00Z').toISOString() : new Date(ev.end.dateTime).toISOString();
          return {
            event_id: ev.id,
            summary: ev.summary ?? '(no title)',
            description: ev.description ?? null,
            location: ev.location ?? null,
            start_utc: startUtc,
            end_utc: endUtc,
            timezone: ev.start?.timeZone ?? 'Europe/Stockholm',
            attendees: (ev.attendees ?? []).map((a: any) => ({ email: a.email, display_name: a.displayName ?? null })),
            is_all_day: isAllDay,
            updated_at: ev.updated,
          };
        });
    }
    ```

    ```typescript
    import { initSentry, wrapHandler } from '../../_shared/sentry.js';
    import { setupOtelTracingAsync, flush as langfuseFlush } from '../../_shared/tracing.js';
    import { EventBridgeClient, PutEventsCommand } from '@aws-sdk/client-eventbridge';
    import { getAccessToken } from './oauth.js';
    import { fetchEventsWindow } from './gcal.js';
    import { upsertCalendarEvents, getPool } from './persist.js';
    import { CalendarEventsReadSchema } from '@kos/contracts';

    const eb = new EventBridgeClient({ region: process.env.AWS_REGION ?? 'eu-north-1' });
    if (!process.env.AWS_REGION) process.env.AWS_REGION = 'eu-north-1';

    export const handler = wrapHandler(async (_event: unknown) => {
      await initSentry();
      await setupOtelTracingAsync();
      const ownerId = process.env.KEVIN_OWNER_ID;
      if (!ownerId) throw new Error('KEVIN_OWNER_ID not set');

      const accounts: Array<'kevin-elzarka' | 'kevin-taleforge'> = ['kevin-elzarka', 'kevin-taleforge'];
      const now = new Date();
      const timeMin = new Date(now.getTime() - 60 * 60 * 1000).toISOString();       // -1h
      const timeMax = new Date(now.getTime() + 48 * 60 * 60 * 1000).toISOString();  // +48h

      try {
        const pool = await getPool();
        const results = await Promise.allSettled(accounts.map(async (account) => {
          let attempt = 0;
          while (true) {
            try {
              const token = await getAccessToken(account);
              const events = await fetchEventsWindow({ accessToken: token, timeMinIso: timeMin, timeMaxIso: timeMax });
              const counts = await upsertCalendarEvents(pool, { ownerId, account, events, calendarId: 'primary' });
              await eb.send(new PutEventsCommand({
                Entries: [{
                  EventBusName: 'kos.capture',
                  Source: 'kos.capture',
                  DetailType: 'calendar.events.cached',
                  Detail: JSON.stringify(CalendarEventsReadSchema.parse({
                    account, window_start_utc: timeMin, window_end_utc: timeMax,
                    fetched_at: now.toISOString(), events_count: events.length,
                  })),
                }],
              }));
              return { account, events: events.length, ...counts };
            } catch (e) {
              if ((e as Error & { code?: string }).code === 'auth_stale' && attempt === 0) {
                // Invalidate module cache by re-fetching — next getAccessToken will refresh.
                attempt++;
                continue;
              }
              throw e;
            }
          }
        }));

        const ok = results.filter(r => r.status === 'fulfilled').map(r => (r as PromiseFulfilledResult<any>).value);
        const failed = results.filter(r => r.status === 'rejected');
        if (failed.length) {
          console.warn('[calendar-reader] account failures', failed.map(f => (f as PromiseRejectedResult).reason?.message ?? 'unknown'));
        }
        return { ok, failed: failed.length };
      } finally {
        await langfuseFlush();
      }
    });
    ```

    ```typescript
    import type { Pool as PgPool } from 'pg';

    export interface CalendarWindowRow {
      event_id: string;
      account: string;
      summary: string;
      start_utc: string;
      end_utc: string;
      timezone: string;
      attendees_json: Array<{ email: string; display_name: string | null }>;
      is_all_day: boolean;
    }

    export async function loadCalendarWindow(
      pool: PgPool,
      ownerId: string,
      windowHours = 48,
    ): Promise<CalendarWindowRow[]> {
      const r = await pool.query(
        `SELECT event_id, account, summary, start_utc::text, end_utc::text, timezone, attendees_json, is_all_day
         FROM calendar_events_cache
         WHERE owner_id = $1
           AND ignored_by_kevin = false
           AND start_utc BETWEEN now() AND now() + ($2::int || ' hours')::interval
         ORDER BY start_utc ASC`,
        [ownerId, windowHours],
      );
      return r.rows as CalendarWindowRow[];
    }

    export function formatCalendarMarkdown(rows: CalendarWindowRow[]): string {
      if (rows.length === 0) return '';
      const lines = ['## Today + Tomorrow Calendar'];
      for (const r of rows) {
        const when = r.is_all_day
          ? new Date(r.start_utc).toISOString().slice(0, 10) + ' (all day)'
          : new Date(r.start_utc).toLocaleString('sv-SE', { timeZone: r.timezone, hour: '2-digit', minute: '2-digit', month: 'short', day: '2-digit' });
        const who = r.attendees_json.length
          ? ` w/ ${r.attendees_json.map(a => a.display_name ?? a.email).slice(0, 4).join(', ')}`
          : '';
        lines.push(`- ${when} [${r.account}] ${r.summary}${who}`);
      }
      return lines.join('\n');
    }
    ```

    ```javascript
    #!/usr/bin/env node
    /**
     * bootstrap-gcal-oauth — one-time Google Calendar OAuth refresh-token setup.
     *
     * Usage:
     *   node scripts/bootstrap-gcal-oauth.mjs --account kevin-elzarka
     *   node scripts/bootstrap-gcal-oauth.mjs --account kevin-taleforge
     *
     * Prerequisites:
     *   - GCP project with Calendar API enabled
     *   - OAuth 2.0 Desktop client created, client_id + client_secret available
     *   - Environment: GCAL_CLIENT_ID + GCAL_CLIENT_SECRET set
     *
     * Flow:
     *   1. Opens Google consent URL (redirect to http://127.0.0.1:9788/callback)
     *   2. Spins a local HTTP listener on 127.0.0.1:9788
     *   3. User approves in browser → Google redirects with ?code=...
     *   4. Script exchanges code → refresh_token
     *   5. Writes Secrets Manager kos/gcal-oauth-<account> = JSON { client_id, client_secret, refresh_token }
     *
     * Scope: calendar.readonly ONLY — NEVER grant calendar.events write.
     */
    import { createServer } from 'node:http';
    import { SecretsManagerClient, PutSecretValueCommand, CreateSecretCommand } from '@aws-sdk/client-secrets-manager';
    import { createHash, randomBytes } from 'node:crypto';
    import process from 'node:process';

    const argv = process.argv.slice(2);
    const accountIdx = argv.indexOf('--account');
    if (accountIdx === -1 || !argv[accountIdx + 1]) { console.error('usage: --account kevin-elzarka|kevin-taleforge'); process.exit(2); }
    const account = argv[accountIdx + 1];
    if (!['kevin-elzarka','kevin-taleforge'].includes(account)) { console.error('invalid account'); process.exit(2); }

    const clientId = process.env.GCAL_CLIENT_ID;
    const clientSecret = process.env.GCAL_CLIENT_SECRET;
    if (!clientId || !clientSecret) { console.error('GCAL_CLIENT_ID and GCAL_CLIENT_SECRET must be set'); process.exit(2); }

    const redirectUri = 'http://127.0.0.1:9788/callback';
    const scope = 'https://www.googleapis.com/auth/calendar.readonly';
    const state = randomBytes(16).toString('hex');
    const authUrl = new URL('https://accounts.google.com/o/oauth2/v2/auth');
    authUrl.searchParams.set('client_id', clientId);
    authUrl.searchParams.set('redirect_uri', redirectUri);
    authUrl.searchParams.set('response_type', 'code');
    authUrl.searchParams.set('scope', scope);
    authUrl.searchParams.set('access_type', 'offline');
    authUrl.searchParams.set('prompt', 'consent');
    authUrl.searchParams.set('state', state);

    console.log(`Open this URL in your browser (or xdg-open it):\n\n${authUrl.toString()}\n`);

    const code = await new Promise((resolve, reject) => {
      const server = createServer((req, res) => {
        const url = new URL(req.url, redirectUri);
        if (url.pathname !== '/callback') { res.writeHead(404).end(); return; }
        if (url.searchParams.get('state') !== state) { res.writeHead(400).end('state mismatch'); reject(new Error('state mismatch')); return; }
        const c = url.searchParams.get('code');
        if (!c) { res.writeHead(400).end('no code'); reject(new Error('no code')); return; }
        res.writeHead(200, { 'Content-Type': 'text/html' }).end('<h1>KOS OAuth capture — you may close this tab.</h1>');
        server.close();
        resolve(c);
      }).listen(9788, '127.0.0.1');
    });

    const tokBody = new URLSearchParams({
      code, client_id: clientId, client_secret: clientSecret, redirect_uri: redirectUri, grant_type: 'authorization_code',
    });
    const tokResp = await fetch('https://oauth2.googleapis.com/token', {
      method: 'POST', headers: { 'Content-Type': 'application/x-www-form-urlencoded' }, body: tokBody,
    });
    if (!tokResp.ok) { console.error(await tokResp.text()); process.exit(3); }
    const tok = await tokResp.json();
    if (!tok.refresh_token) { console.error('No refresh_token. Ensure prompt=consent and access_type=offline.'); process.exit(3); }

    const sm = new SecretsManagerClient({ region: process.env.AWS_REGION ?? 'eu-north-1' });
    const secretId = `kos/gcal-oauth-${account}`;
    const payload = JSON.stringify({ client_id: clientId, client_secret: clientSecret, refresh_token: tok.refresh_token });
    try {
      await sm.send(new PutSecretValueCommand({ SecretId: secretId, SecretString: payload }));
      console.log(`Updated existing ${secretId}`);
    } catch (e) {
      if (e.name === 'ResourceNotFoundException') {
        await sm.send(new CreateSecretCommand({ Name: secretId, SecretString: payload, Description: `Google Calendar OAuth for ${account}` }));
        console.log(`Created ${secretId}`);
      } else {
        throw e;
      }
    }
    console.log(`\nVerify with:\n  aws secretsmanager get-secret-value --secret-id ${secretId} --query SecretString --output text | jq '.refresh_token' | head -c 20\n`);
    ```
