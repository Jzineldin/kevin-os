    import { SecretsManagerClient, GetSecretValueCommand } from '@aws-sdk/client-secrets-manager';
    import { randomUUID } from 'node:crypto';

    const sm = new SecretsManagerClient({ region: process.env.AWS_REGION ?? 'eu-north-1' });
    let cachedApiKey: string | null = null;

    async function getApiKey(): Promise<string> {
      if (cachedApiKey) return cachedApiKey;
      const arn = process.env.POSTIZ_API_KEY_SECRET_ARN;
      if (!arn) throw new Error('POSTIZ_API_KEY_SECRET_ARN not set');
      const r = await sm.send(new GetSecretValueCommand({ SecretId: arn }));
      if (!r.SecretString) throw new Error('postiz api key secret empty');
      cachedApiKey = r.SecretString.trim();
      return cachedApiKey;
    }

    export async function postizMcpCall<T>(method: string, params: unknown): Promise<T> {
      const apiKey = await getApiKey();
      const base = process.env.POSTIZ_BASE_URL ?? 'http://postiz.kos.local:3000';
      const url = `${base}/api/mcp/${apiKey}`;
      const body = { jsonrpc: '2.0', id: randomUUID(), method, params };
      const r = await fetch(url, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json', Accept: 'application/json, text/event-stream' },
        body: JSON.stringify(body),
      });
      if (!r.ok) {
        const text = await r.text();
        throw new Error(`Postiz MCP ${r.status}: ${text.slice(0, 400)}`);
      }
      const resp: { result?: T; error?: { code: number; message: string; data?: any } } = await r.json();
      if (resp.error) {
        const err: Error & { code?: number; retryAfter?: number } = new Error(`Postiz MCP error: ${resp.error.message}`);
        err.code = resp.error.code;
        if (resp.error.data?.retry_after) err.retryAfter = resp.error.data.retry_after;
        throw err;
      }
      if (resp.result === undefined) throw new Error('Postiz MCP: no result, no error — unexpected shape');
      return resp.result;
    }

    export async function createPost(input: {
      platform: 'instagram'|'linkedin'|'tiktok'|'reddit'|'newsletter';
      content: string;
      media_urls?: string[];
      schedule_time?: string;
    }) {
      return postizMcpCall<{ post_id: string; status: string; schedule_time?: string }>('create_post', input);
    }

    export async function deletePost(postId: string) {
      return postizMcpCall<{ post_id: string; status: string }>('delete_post', { post_id: postId });
    }

    import { initSentry, wrapHandler } from '../../_shared/sentry.js';
    import { setupOtelTracingAsync, flush as langfuseFlush, tagTraceWithCaptureId } from '../../_shared/tracing.js';
    import { withTimeoutAndRetry } from '../../_shared/with-timeout-retry.js';
    import { EventBridgeClient, PutEventsCommand } from '@aws-sdk/client-eventbridge';
    import { ContentApprovedSchema, ContentPublishedSchema } from '@kos/contracts';
    import { createPost, deletePost } from './postiz.js';
    import { loadDraftForPublish, markDraftScheduled, markDraftPublished, markAuthorizationConsumed, markDraftFailed, loadDraftForCancel, markDraftCancelled, getPool } from './persist.js';
    import { z } from 'zod';

    const eb = new EventBridgeClient({ region: 'eu-north-1' });
    if (!process.env.AWS_REGION) process.env.AWS_REGION = 'eu-north-1';

    const CancelRequestedSchema = z.object({
      capture_id: z.string(),
      draft_id: z.string().uuid(),
      requested_at: z.string().datetime(),
    });

    export const handler = wrapHandler(async (event: any) => {
      await initSentry();
      await setupOtelTracingAsync();
      const ownerId = process.env.KEVIN_OWNER_ID!;
      if (!ownerId) throw new Error('KEVIN_OWNER_ID not set');
      try {
        if (event['detail-type'] === 'content.approved') {
          return await handleApproved(event.detail, ownerId);
        }
        if (event['detail-type'] === 'content.cancel_requested') {
          return await handleCancel(event.detail, ownerId);
        }
        return { skipped: event['detail-type'] };
      } finally {
        await langfuseFlush();
      }
    });

    async function handleApproved(detail: unknown, ownerId: string) {
      const payload = ContentApprovedSchema.parse(detail);
      tagTraceWithCaptureId(payload.capture_id);
      const pool = await getPool();
      const draft = await loadDraftForPublish(pool, payload.draft_id, payload.authorization_id);
      if (!draft) return { skipped: 'not_found_or_consumed', draft_id: payload.draft_id };
      try {
        const result = await withTimeoutAndRetry(
          () => createPost({
            platform: draft.platform as any,
            content: draft.content,
            media_urls: draft.media_urls ?? [],
            schedule_time: payload.schedule_time ?? undefined,
          }),
          { timeoutMs: 15_000, maxRetries: 2, toolName: 'postiz.create_post', captureId: payload.capture_id, ownerId },
        );
        if (result.status === 'published' || !payload.schedule_time) {
          await markDraftPublished(pool, { draftId: payload.draft_id, postizPostId: result.post_id, publishedAt: new Date().toISOString() });
        } else {
          await markDraftScheduled(pool, { draftId: payload.draft_id, postizPostId: result.post_id, scheduleTime: result.schedule_time ?? payload.schedule_time });
        }
        await markAuthorizationConsumed(pool, payload.authorization_id, { post_id: result.post_id, status: result.status });
        await eb.send(new PutEventsCommand({
          Entries: [{
            EventBusName: 'kos.output', Source: 'kos.output', DetailType: 'content.published',
            Detail: JSON.stringify(ContentPublishedSchema.parse({
              capture_id: payload.capture_id, draft_id: payload.draft_id, platform: draft.platform,
              postiz_post_id: result.post_id, published_at: new Date().toISOString(),
            })),
          }],
        }));
        return { published: result.post_id };
      } catch (e) {
        await markDraftFailed(pool, payload.draft_id, String(e));
        throw e;
      }
    }

    async function handleCancel(detail: unknown, ownerId: string) {
      const payload = CancelRequestedSchema.parse(detail);
      tagTraceWithCaptureId(payload.capture_id);
      const pool = await getPool();
      const draft = await loadDraftForCancel(pool, payload.draft_id);
      if (!draft) return { skipped: 'not_found' };
      if (!draft.postiz_post_id) {
        // Never scheduled with Postiz; just mark local state
        await markDraftCancelled(pool, payload.draft_id);
        return { cancelled: 'local_only' };
      }
      if (draft.status === 'published') return { skipped: 'already_published' };
      await deletePost(draft.postiz_post_id);
      await markDraftCancelled(pool, payload.draft_id);
      return { cancelled: draft.postiz_post_id };
    }

    import { z } from 'zod';
    import { EventBridgeClient, PutEventsCommand } from '@aws-sdk/client-eventbridge';
    import { randomUUID } from 'node:crypto';
    import { ContentApprovedSchema } from '@kos/contracts';
    import { loadContentDraftById, insertContentPublishAuthorization, updateContentDraftForEdit, updateContentDraftStatus, getPool } from '../content-drafts-persist.js';

    const eb = new EventBridgeClient({ region: 'eu-north-1' });

    const ApproveInputSchema = z.object({ schedule_time: z.string().datetime().nullable().optional() });

    export async function approveContentDraft(req: { draftId: string; ownerId: string; body: unknown }) {
      const pool = await getPool();
      const draft = await loadContentDraftById(pool, req.draftId, req.ownerId);
      if (!draft) return { status: 404, body: { error: 'not_found' } };
      if (!['draft','edited'].includes(draft.status)) return { status: 409, body: { error: `cannot approve from status ${draft.status}` } };
      const parsed = ApproveInputSchema.safeParse(req.body);
      if (!parsed.success) return { status: 400, body: { error: 'invalid_body', issues: parsed.error.issues } };
      const scheduleTime = parsed.data.schedule_time ?? null;
      if (scheduleTime && new Date(scheduleTime).getTime() < Date.now() - 60_000) {
        return { status: 400, body: { error: 'schedule_time in the past' } };
      }

      const authId = randomUUID();
      // D-26: write authorization row BEFORE emitting event (publisher joins on it)
      await insertContentPublishAuthorization(pool, { id: authId, ownerId: req.ownerId, draftId: req.draftId, scheduleTime });

      const payload = ContentApprovedSchema.parse({
        capture_id: draft.capture_id, draft_id: req.draftId, authorization_id: authId,
        schedule_time: scheduleTime, approved_at: new Date().toISOString(),
      });
      await eb.send(new PutEventsCommand({
        Entries: [{ EventBusName: 'kos.output', Source: 'kos.output', DetailType: 'content.approved', Detail: JSON.stringify(payload) }],
      }));
      return { status: 200, body: { authorization_id: authId, schedule_time: scheduleTime } };
    }

    const EditInputSchema = z.object({ content: z.string().min(1).max(15000), media_urls: z.array(z.string().url()).max(10).optional() });
    const PLATFORM_CAPS: Record<string, number> = { instagram: 2200, linkedin: 3000, tiktok: 800, reddit: 40000, newsletter: 15000 };

    export async function editContentDraft(req: { draftId: string; ownerId: string; body: unknown }) {
      const pool = await getPool();
      const draft = await loadContentDraftById(pool, req.draftId, req.ownerId);
      if (!draft) return { status: 404, body: { error: 'not_found' } };
      if (!['draft','edited'].includes(draft.status)) return { status: 409, body: { error: `cannot edit from status ${draft.status}` } };
      const parsed = EditInputSchema.safeParse(req.body);
      if (!parsed.success) return { status: 400, body: { error: 'invalid_body' } };
      const cap = PLATFORM_CAPS[draft.platform];
      if (cap && parsed.data.content.length > cap) return { status: 400, body: { error: `content exceeds ${draft.platform} cap ${cap}` } };
      await updateContentDraftForEdit(pool, req.draftId, parsed.data.content, parsed.data.media_urls ?? []);
      await eb.send(new PutEventsCommand({
        Entries: [{ EventBusName: 'kos.output', Source: 'kos.output', DetailType: 'content_draft_edited', Detail: JSON.stringify({ capture_id: draft.capture_id, draft_id: req.draftId }) }],
      }));
      return { status: 200, body: { status: 'edited' } };
    }

    export async function skipContentDraft(req: { draftId: string; ownerId: string }) {
      const pool = await getPool();
      const draft = await loadContentDraftById(pool, req.draftId, req.ownerId);
      if (!draft) return { status: 404, body: { error: 'not_found' } };
      if (['published','scheduled'].includes(draft.status)) return { status: 409, body: { error: `cannot skip from status ${draft.status}` } };
      await updateContentDraftStatus(pool, req.draftId, 'skipped');
      await eb.send(new PutEventsCommand({
        Entries: [{ EventBusName: 'kos.output', Source: 'kos.output', DetailType: 'content_draft_skipped', Detail: JSON.stringify({ capture_id: draft.capture_id, draft_id: req.draftId }) }],
      }));
      return { status: 200, body: { status: 'skipped' } };
    }

    export async function cancelContentDraft(req: { draftId: string; ownerId: string }) {
      const pool = await getPool();
      const draft = await loadContentDraftById(pool, req.draftId, req.ownerId);
      if (!draft) return { status: 404, body: { error: 'not_found' } };
      if (draft.status === 'published') return { status: 409, body: { error: 'already_published' } };
      // Emit cancel_requested — publisher handles Postiz deletion for scheduled posts
      await eb.send(new PutEventsCommand({
        Entries: [{ EventBusName: 'kos.output', Source: 'kos.output', DetailType: 'content.cancel_requested',
          Detail: JSON.stringify({ capture_id: draft.capture_id, draft_id: req.draftId, requested_at: new Date().toISOString() }) }],
      }));
      return { status: 200, body: { status: 'cancel_requested' } };
    }
