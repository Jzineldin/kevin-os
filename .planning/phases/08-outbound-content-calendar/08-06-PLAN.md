    ```javascript
    #!/usr/bin/env node
    /**
     * Approve-gate invariant verifier.
     *
     * Static (--static): runs `pnpm --filter @kos/cdk synth` and greps the output:
     *   - publisher Lambda role: NO bedrock:* actions
     *   - content-writer-platform Lambda role: NO postiz ARNs or postiz-related IAM
     *   - mutation-executor Lambda role: NO bedrock:*, NO postiz:*, NO ses:*, NO google scope
     *   - mutation-proposer Lambda role: NO postiz:*, NO ses:*
     *
     * Live (--live): queries RDS and asserts:
     *   - SELECT COUNT(*) FROM content_drafts WHERE status='published' AND id NOT IN (SELECT draft_id FROM content_publish_authorizations WHERE consumed_at IS NOT NULL) = 0
     *   - SELECT COUNT(*) FROM pending_mutations WHERE status='executed' AND id NOT IN (SELECT mutation_id FROM pending_mutation_authorizations WHERE consumed_at IS NOT NULL) = 0
     */
    import { execFileSync } from 'node:child_process';
    import pg from 'pg';

    const mode = process.argv.includes('--live') ? 'live' : 'static';

    async function runStatic() {
      const synthOut = execFileSync('pnpm', ['--filter', '@kos/cdk', 'synth'], { encoding: 'utf8', stdio: ['ignore','pipe','pipe'] });
      // Parse CloudFormation template; extract IAM policies per Lambda; assert forbidden actions absent.
      // Simplification: grep the output for role names + check forbidden actions in JSON policy docs.
      const checks = [
        { role: 'KosIntegrations-Publisher', forbidden: ['bedrock:', 'ses:SendRawEmail'] },
        { role: 'KosIntegrations-ContentWriterPlatform', forbidden: ['postiz', 'ses:SendRawEmail', 'google'] },
        { role: 'KosIntegrations-MutationExecutor', forbidden: ['bedrock:', 'postiz', 'ses:', 'googleapis'] },
        { role: 'KosIntegrations-MutationProposer', forbidden: ['postiz', 'ses:SendRawEmail'] },
      ];
      let failed = 0;
      for (const { role, forbidden } of checks) {
        for (const f of forbidden) {
          // Very rough: find role's section in synth output, check for action
          const roleIdx = synthOut.indexOf(role);
          if (roleIdx === -1) { console.error(`role ${role} NOT FOUND in synth output`); failed++; continue; }
          const rolePolicyBlock = synthOut.slice(roleIdx, roleIdx + 4000);
          if (rolePolicyBlock.includes(f)) { console.error(`FORBIDDEN ${f} found on ${role}`); failed++; }
        }
      }
      if (failed) { console.error(`approve-gate static FAIL: ${failed}`); process.exit(1); }
      console.log('approve-gate static PASS');
    }

    async function runLive() {
      const pool = new pg.Pool({ connectionString: process.env.KOS_RDS_URL });
      const q1 = await pool.query(`
        SELECT COUNT(*) AS bad
        FROM content_drafts cd
        WHERE cd.status IN ('scheduled','published')
          AND NOT EXISTS (SELECT 1 FROM content_publish_authorizations cpa WHERE cpa.draft_id = cd.id AND cpa.consumed_at IS NOT NULL)
      `);
      const q2 = await pool.query(`
        SELECT COUNT(*) AS bad FROM pending_mutations pm
        WHERE pm.status = 'executed'
          AND NOT EXISTS (SELECT 1 FROM pending_mutation_authorizations pma WHERE pma.mutation_id = pm.id AND pma.consumed_at IS NOT NULL)
      `);
      await pool.end();
      const fail = Number(q1.rows[0].bad) + Number(q2.rows[0].bad);
      if (fail) { console.error(`approve-gate live FAIL: ${fail} orphan rows`); process.exit(1); }
      console.log('approve-gate live PASS');
    }

    if (mode === 'static') await runStatic();
    else await runLive();
    ```
