```typescript
export const VpsServiceInventorySchema = z.object({
  generated_at: z.string().datetime(),
  hostname: z.string(),           // '98.91.6.66' or 'ubuntu'
  services: z.array(z.object({
    service_name: z.string(),     // 'gmail_classifier'
    systemd_unit: z.string().optional(),  // 'gmail-classifier.service' if present
    process_name: z.string().optional(),  // as seen in ps aux
    status: z.enum(['active', 'inactive', 'failed', 'not-installed']),
    replacement_phase: z.string().optional(),   // e.g. 'phase-4-email-triage'
    replacement_artifact: z.string().optional(), // e.g. 'services/email-triage'
    retirement_category: z.enum(['replaced', 'inert', 'frozen-phase-1', 'n8n']),
    last_log_line: z.string().optional(),
  })),
});
export type VpsServiceInventory = z.infer<typeof VpsServiceInventorySchema>;
```
