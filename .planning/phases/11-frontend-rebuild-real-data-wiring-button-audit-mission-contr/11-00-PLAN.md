    ```js
    import { chromium } from '@playwright/test';
    const baseUrl = process.env.PLAYWRIGHT_BASE_URL;
    const token = process.env.KOS_DASHBOARD_BEARER_TOKEN;
    if (!baseUrl || !token) { console.error('Need PLAYWRIGHT_BASE_URL + KOS_DASHBOARD_BEARER_TOKEN'); process.exit(1); }
    const browser = await chromium.launch();
    const ctx = await browser.newContext();
    await ctx.addCookies([{ name: 'kos_session', value: token, domain: new URL(baseUrl).hostname, path: '/', secure: true, httpOnly: true, sameSite: 'Lax' }]);
    const page = await ctx.newPage();
    for (const route of ['today', 'inbox', 'entities', 'calendar']) {
      await page.goto(`${baseUrl}/${route}`, { waitUntil: 'networkidle' });
      await page.screenshot({ path: `apps/dashboard/tests/visual-baseline/${route}.png`, fullPage: true });
      console.log(`captured ${route}`);
    }
    await browser.close();
    ```
