    ```typescript
    import { sql } from 'drizzle-orm';
    import { getDb } from './db.js';
    import { OWNER_ID } from './owner-scoped.js';

    /**
     * D-03 seed names. These were inserted by a one-shot manual SQL session
     * pre-Phase 11; repo-wide grep confirms they appear in zero prod write
     * paths. This guard blocks dashboard-api from serving traffic if any
     * row with one of these titles appears under owner_id=Kevin.
     *
     * Per CONTEXT D-04 (Claude's discretion of mechanism), option (b) was
     * chosen: cheapest, fastest, fail-loud at Lambda init.
     */
    export const SEED_NAMES = [
      'Damien Carter',
      'Christina Larsson',
      'Jan Eriksson',
      'Lars Svensson',
      'Almi Företagspartner',
      'Re: Partnership proposal',
      'Re: Summer meeting',
      'Possible duplicate: Damien C.',
      'Paused: Maria vs Maria Johansson',
      'Outbehaving angel investor',
    ] as const;

    let cachedResult: 'clean' | 'polluted' | null = null;

    export async function assertNoSeedPollution(): Promise<void> {
      if (cachedResult === 'clean') return;
      if (cachedResult === 'polluted') {
        throw new Error('[dashboard-api] seed pollution detected — refusing to serve');
      }
      const db = await getDb();
      const r = (await db.execute(sql`
        SELECT 1 FROM inbox_index
        WHERE owner_id = ${OWNER_ID}::uuid
          AND title = ANY(${[...SEED_NAMES]}::text[])
        LIMIT 1
      `)) as unknown as { rows: unknown[] };
      if (r.rows.length > 0) {
        cachedResult = 'polluted';
        console.error('[dashboard-api] SEED POLLUTION DETECTED — purge required before re-enabling');
        throw new Error('[dashboard-api] seed pollution detected — refusing to serve');
      }
      cachedResult = 'clean';
    }

    /** Test-only: reset the cached result so tests can re-exercise the query path. */
    export function __resetSeedPollutionCacheForTests(): void {
      cachedResult = null;
    }
    ```
