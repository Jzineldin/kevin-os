    ```typescript
    async function queryCalendarEventsCache(
      db: NodePgDatabase,
      weekStart: string,
      weekEnd: string,
    ): Promise<CalendarEvent[]> {
      const r = (await db.execute(sql`
        SELECT
          id::text          AS id,
          start_at::text    AS start_at,
          end_at::text      AS end_at,
          title             AS title,
          account_email     AS account_email
        FROM calendar_events_cache
        WHERE owner_id = ${OWNER_ID}::uuid
          AND start_at >= ${weekStart}::timestamptz
          AND start_at < ${weekEnd}::timestamptz
        ORDER BY start_at ASC
      `)) as unknown as {
        rows: Array<{ id: string; start_at: string; end_at: string | null; title: string; account_email: string | null }>;
      };
      return r.rows.map((r) => ({
        id: r.id,
        start_at: r.start_at,
        end_at: r.end_at,
        title: r.title,
        source: 'google_calendar' as const,
        account_email: r.account_email,
      }));
    }

    /**
     * Merge two event lists, deduplicating by (start_at minute, title-normalized).
     * Google Calendar source wins over Notion CC when both contain a matching event.
     * (Per RESEARCH: Google is canonical for actual meetings; Notion CC for deadlines.)
     */
    function mergeAndDedupeEvents(notion: CalendarEvent[], google: CalendarEvent[]): CalendarEvent[] {
      const keyOf = (e: CalendarEvent): string => {
        const startMinute = new Date(e.start_at).toISOString().slice(0, 16);  // YYYY-MM-DDTHH:MM
        const norm = e.title.trim().toLowerCase().replace(/\s+/g, ' ');
        return `${startMinute}|${norm}`;
      };
      const merged = new Map<string, CalendarEvent>();
      // Insert Notion first (lower priority); Google then overrides.
      for (const e of notion) merged.set(keyOf(e), { ...e, source: e.source ?? 'notion_cc' });
      for (const e of google) merged.set(keyOf(e), e);
      return Array.from(merged.values()).sort(
        (a, b) => new Date(a.start_at).getTime() - new Date(b.start_at).getTime(),
      );
    }
    ```

    async function calendarWeekHandler(ctx: Ctx): Promise<RouteResponse> {
      const { weekStart, weekEnd } = computeWeekWindow(ctx);
      const db = await getDb();
      const [notionEvents, googleEvents] = await Promise.all([
