    Step 2 — Create `services/dashboard-api/src/handlers/integrations.ts`:

    ```typescript
    import { sql } from 'drizzle-orm';
    import {
      ChannelHealthItemSchema,
      IntegrationsHealthResponseSchema,
      SchedulerHealthItemSchema,
      type ChannelHealthItem,
      type SchedulerHealthItem,
    } from '@kos/contracts/dashboard';
    import { register, type Ctx, type RouteResponse } from '../router.js';
    import { getDb } from '../db.js';
    import { OWNER_ID } from '../owner-scoped.js';

    /** Known capture channels and their expected interval in minutes. */
    const CHANNEL_SPEC: Array<{ name: string; agent_name: string; max_age_min: number; type: 'capture' }> = [
      { name: 'Telegram', agent_name: 'telegram-bot', max_age_min: 1440, type: 'capture' },        // 1 day; only fires when Kevin sends
      { name: 'Gmail', agent_name: 'gmail-poller', max_age_min: 30, type: 'capture' },             // 5 min cron, allow 30
      { name: 'Granola', agent_name: 'granola-poller', max_age_min: 60, type: 'capture' },         // 15 min cron, allow 60
      { name: 'Google Calendar', agent_name: 'calendar-reader', max_age_min: 90, type: 'capture' },// 30 min cron, allow 90
      { name: 'Chrome extension', agent_name: 'chrome-webhook', max_age_min: 4320, type: 'capture' }, // 3 days
      { name: 'LinkedIn', agent_name: 'linkedin-webhook', max_age_min: 4320, type: 'capture' },    // 3 days
    ];

    /** Known schedulers + their cron expectations. */
    const SCHEDULER_SPEC: Array<{ name: string; agent_name: string }> = [
      { name: 'morning-brief', agent_name: 'morning-brief' },
      { name: 'day-close', agent_name: 'day-close' },
      { name: 'weekly-review', agent_name: 'weekly-review' },
      { name: 'gmail-poller', agent_name: 'gmail-poller' },
      { name: 'granola-poller', agent_name: 'granola-poller' },
      { name: 'calendar-reader', agent_name: 'calendar-reader' },
    ];

    function classifyChannel(lastOkAt: string | null, maxAgeMin: number): ChannelHealthItem['status'] {
      if (lastOkAt == null) return 'down';
      const ageMin = (Date.now() - new Date(lastOkAt).getTime()) / 60000;
      if (ageMin > maxAgeMin * 2) return 'down';
      if (ageMin > maxAgeMin) return 'degraded';
      return 'healthy';
    }

    async function loadAgentRunsAggregate(): Promise<Map<string, { last_ok: string | null; last_any: string | null; last_status: 'ok' | 'fail' | 'pending' | null }>> {
      const db = await getDb();
      const r = (await db.execute(sql`
        SELECT
          agent_name,
          MAX(CASE WHEN status = 'ok' THEN finished_at END) AS last_ok,
          MAX(finished_at) AS last_any,
          (
            SELECT status FROM agent_runs ar2
            WHERE ar2.agent_name = ar.agent_name AND ar2.owner_id = ${OWNER_ID}::uuid
            ORDER BY ar2.finished_at DESC NULLS LAST LIMIT 1
          ) AS last_status
        FROM agent_runs ar
        WHERE owner_id = ${OWNER_ID}::uuid
        GROUP BY agent_name
      `)) as unknown as {
        rows: Array<{ agent_name: string; last_ok: string | null; last_any: string | null; last_status: string | null }>;
      };
      const map = new Map<string, { last_ok: string | null; last_any: string | null; last_status: 'ok' | 'fail' | 'pending' | null }>();
      for (const row of r.rows) {
        map.set(row.agent_name, {
          last_ok: row.last_ok,
          last_any: row.last_any,
          last_status: (row.last_status as 'ok' | 'fail' | 'pending' | null) ?? null,
        });
      }
      return map;
    }

    async function loadChannels(agg: Map<string, { last_ok: string | null }>): Promise<ChannelHealthItem[]> {
      return CHANNEL_SPEC.map((spec) => {
        const a = agg.get(spec.agent_name);
        const last_ok = a?.last_ok ?? null;
        return ChannelHealthItemSchema.parse({
          name: spec.name,
          type: spec.type,
          status: classifyChannel(last_ok, spec.max_age_min),
          last_event_at: last_ok,
        });
      });
    }

    async function loadSchedulers(agg: Map<string, { last_any: string | null; last_status: 'ok' | 'fail' | 'pending' | null }>): Promise<SchedulerHealthItem[]> {
      return SCHEDULER_SPEC.map((spec) => {
        const a = agg.get(spec.agent_name);
        return SchedulerHealthItemSchema.parse({
          name: spec.name,
          last_run_at: a?.last_any ?? null,
          next_run_at: null,                          // EventBridge Scheduler next-fire is not exposed via SDK without a separate call; null for now.
          last_status: a?.last_status ?? null,
        });
      });
    }

    async function integrationsHealthHandler(_ctx: Ctx): Promise<RouteResponse> {
      const agg = await loadAgentRunsAggregate();
      const [channels, schedulers] = await Promise.all([loadChannels(agg), loadSchedulers(agg)]);
      const payload = IntegrationsHealthResponseSchema.parse({ channels, schedulers });
      return {
        statusCode: 200,
        body: JSON.stringify(payload),
        headers: { 'cache-control': 'private, max-age=0, stale-while-revalidate=60' },
      };
    }

    register('GET', '/integrations/health', integrationsHealthHandler);
    export { integrationsHealthHandler };
    ```

    function timeAgo(iso: string | null): string {
      if (!iso) return '—';
      const ms = Date.now() - new Date(iso).getTime();
      const min = Math.floor(ms / 60000);
      if (min < 1) return 'just now';
      if (min < 60) return `${min}m ago`;
      const hr = Math.floor(min / 60);
      if (hr < 24) return `${hr}h ago`;
      return `${Math.floor(hr / 24)}d ago`;
    }
